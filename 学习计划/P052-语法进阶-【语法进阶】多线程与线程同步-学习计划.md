## 一、线程基础

### 1.1 线程 vs 进程

| 对比项 | 进程 | 线程 |
|--------|------|------|
| 资源占用 | 独立内存空间 | 共享进程内存 |
| 创建开销 | 大 | 小 |
| 通信方式 | IPC复杂 | 直接共享数据 |
| 切换开销 | 大 | 小 |
| 适用场景 | CPU 密集 | I/O 密集 |

### 1.2 threading 模块

```python
import threading
import time

def worker(name, seconds):
    """工作线程函数"""
    print(f"线程 {name} 开始")
    time.sleep(seconds)
    print(f"线程 {name} 结束")

# 方法一：创建 Thread 对象
t1 = threading.Thread(target=worker, args=("A", 1))
t1.start()  # 启动线程
t1.join()  # 等待线程结束

# 方法二：继承 Thread 类
class MyThread(threading.Thread):
    def __init__(self, name, seconds):
        super().__init__()
        self.name = name
        self.seconds = seconds
    
    def run(self):  # 线程执行的入口
        print(f"线程 {self.name} 开始")
        time.sleep(self.seconds)
        print(f"线程 {self.name} 结束")

t2 = MyThread("B", 2)
t2.start()
t2.join()
```

## 二、线程同步原语

### 2.1 Lock 互斥锁

```python
import threading

counter = 0
lock = threading.Lock()

def increment():
    global counter
    for _ in range(1000000):
        with lock:  # 获取锁
            counter += 1

# 不使用锁：结果不确定
# 使用锁：结果正确
threads = [threading.Thread(target=increment) for _ in range(4)]
for t in threads:
    t.start()
for t in threads:
    t.join()

print(f"最终计数：{counter}")  # 应该是 4000000
```

### 2.2 RLock 可重入锁

```python
import threading

lock = threading.RLock()

def outer():
    with lock:
        print("外层获取锁")
        inner()

def inner():
    # 递归调用，同一线程需要再次获取锁
    with lock:
        print("内层获取锁")

outer()  # RLock 允许同一线程多次获取
```

### 2.3 Condition 条件变量

```python
import threading

buffer = []
MAX_SIZE = 5
not_full = threading.Condition()
not_empty = threading.Condition()

def producer():
    for i in range(10):
        with not_full:
            while len(buffer) >= MAX_SIZE:
                not_full.wait()
            buffer.append(i)
            print(f"生产: {i}")
            not_empty.notify()

def consumer():
    for i in range(10):
        with not_empty:
            while not buffer:
                not_empty.wait()
            item = buffer.pop(0)
            print(f"消费: {item}")
            not_full.notify()

t1 = threading.Thread(target=producer)
t2 = threading.Thread(target=consumer)
t1.start()
t2.start()
t1.join()
t2.join()
```

### 2.4 Semaphore 信号量

```python
import threading

# 控制同时访问的线程数
pool_size = 3
pool = threading.Semaphore(pool_size)

def access_resource(name):
    with pool:
        print(f"{name} 获取资源")
        time.sleep(1)
        print(f"{name} 释放资源")

threads = [threading.Thread(target=access_resource, args=(f"Worker-{i}",)) for i in range(6)]
for t in threads:
    t.start()
for t in threads:
    t.join()
```

### 2.5 Event 事件

```python
import threading

event = threading.Event()

def waiter(name, wait_time):
    print(f"{name} 等待中...")
    event.wait(wait_time)
    if event.is_set():
        print(f"{name} 收到信号")
    else:
        print(f"{name} 超时")

def setter():
    time.sleep(2)
    print("设置事件")
    event.set()

t1 = threading.Thread(target=waiter, args=("W1", 5))
t2 = threading.Thread(target=waiter, args=("W2", 5))
t3 = threading.Thread(target=setter)

t1.start()
t2.start()
t3.start()

t1.join()
t2.join()
t3.join()
```

## 三、Queue 线程安全队列

### 3.1 生产者-消费者模式

```python
import threading
import queue

q = queue.Queue(maxsize=10)

def producer(n):
    for i in range(n):
        q.put(i)
        print(f"生产: {i}")

def consumer(n):
    for _ in range(n):
        item = q.get()
        print(f"消费: {item}")
        q.task_done()

t1 = threading.Thread(target=producer, args=(10,))
t2 = threading.Thread(target=consumer, args=(10,))

t1.start()
t2.start()

t1.join()
t2.join()
```

pq.put((3, "低优先级"))

## 四、GIL 原理

### 4.1 什么是 GIL

```python
import dis

def count_loop():
    """CPU 密集型任务"""
    total = 0
    for i in range(1000000):
        total += i
    return total

# 查看字节码
print(dis.dis(count_loop))
```

GIL（Global Interpreter Lock）导致同一时刻只有一个线程执行 Python 字节码。

### 4.2 GIL 影响

```python
import threading
import time

# CPU 密集型：GIL 限制，无法利用多核
def cpu_task():
    return sum(i * i for i in range(10000000))

# I/O 密集型：GIL 在等待 I/O 时释放，可以利用多核
def io_task():
    time.sleep(1)  # 模拟 I/O

# CPU 密集型测试
start = time.time()
threads = [threading.Thread(target=cpu_task) for _ in range(4)]
for t in threads:
    t.start()
for t in threads:
    t.join()
print(f"CPU 密集型（4线程）用时：{time.time() - start:.2f}s")

# 单线程对比
start = time.time()
cpu_task()
print(f"CPU 密集型（1线程）用时：{time.time() - start:.2f}s")
# 两者时间相近，说明 GIL 限制了并行
```

### 4.3 何时不用关心 GIL

```python
# I/O 密集型：网络请求、文件读写
import threading
import requests

def fetch_url(url):
    response = requests.get(url)
    return len(response.content)

urls = ["https://example.com"] * 10
threads = [threading.Thread(target=fetch_url, args=(url,)) for url in urls]

start = time.time()
for t in threads:
    t.start()
for t in threads:
    t.join()
print(f"I/O 密集型用时：{time.time() - start:.2f}s")
```

## 五、实战案例

### 5.1 多线程下载器

```python
import threading
import queue
import time

class DownloadManager:
    def __init__(self, max_workers=5):
        self.queue = queue.Queue()
        self.max_workers = max_workers
        self.results = []
        self.lock = threading.Lock()
    
    def add_task(self, url):
        self.queue.put(url)
    
    def worker(self):
        """下载线程"""
        while True:
            try:
                url = self.queue.get(timeout=1)
            except queue.Empty:
                break
            
            try:
                # 模拟下载
                time.sleep(0.5)
                result = f"Downloaded: {url}"
                
                with self.lock:
                    self.results.append(result)
            finally:
                self.queue.task_done()
    
    def download_all(self):
        """启动下载"""
        threads = []
        for _ in range(self.max_workers):
            t = threading.Thread(target=self.worker)
            t.start()
            threads.append(t)
        
        self.queue.join()
        
        for t in threads:
            t.join()
        
        return self.results

manager = DownloadManager(max_workers=3)
for i in range(10):
    manager.add_task(f"https://example.com/file{i}.txt")

results = manager.download_all()
print(f"完成 {len(results)} 个下载")
```

### 5.2 线程池

```python
from concurrent.futures import ThreadPoolExecutor
import threading

def task(n):
    return f"任务 {n} 完成，当前线程: {threading.current_thread().name}"

# 使用 ThreadPoolExecutor
with ThreadPoolExecutor(max_workers=4) as executor:
    # 提交多个任务
    futures = [executor.submit(task, i) for i in range(8)]
    
    # 获取结果
    for future in futures:
        print(future.result())

# map 用法
with ThreadPoolExecutor(max_workers=4) as executor:
    results = executor.map(task, range(8))
    for result in results:
        print(result)
```

### 5.3 进度显示

```python
import threading
import time
import sys

def long_running_task(task_id, progress_queue):
    """长时间运行的任务"""
    total = 100
    for i in range(total):
        time.sleep(0.05)
        progress_queue.put((task_id, i + 1, total))

def progress_reporter(progress_queue, num_tasks):
    """进度汇报线程"""
    completed = 0
    total_progress = {i: 0 for i in range(num_tasks)}
    
    while completed < num_tasks:
        try:
            task_id, current, total = progress_queue.get(timeout=0.1)
            total_progress[task_id] = current
            completed = sum(1 for p in total_progress.values() if p >= total)
            
            # 显示进度条
            progress = sum(total_progress.values()) / (num_tasks * total) * 100
            bar = "=" * int(progress // 2) + ">" + " " * (50 - int(progress // 2))
            sys.stdout.write(f"\r[{bar}] {progress:.1f}%")
            sys.stdout.flush()
        except:
            pass
    
    print()

if __name__ == "__main__":
    num_tasks = 5
    progress_queue = queue.Queue()
    
    reporter = threading.Thread(target=progress_reporter, args=(progress_queue, num_tasks))
    reporter.start()
    
    threads = [
        threading.Thread(target=long_running_task, args=(i, progress_queue))
        for i in range(num_tasks)
    ]
    
    for t in threads:
        t.start()
    for t in threads:
        t.join()
    
    reporter.join()
    print("所有任务完成")
```

## 六、死锁与避免

### 6.1 死锁原因

```python
import threading

# 死锁示例：两个线程互相等待对方释放锁
lock1 = threading.Lock()
lock2 = threading.Lock()

def thread1():
    with lock1:
        time.sleep(0.1)
        with lock2:  # 等待 lock2
            print("线程1完成")

def thread2():
    with lock2:
        time.sleep(0.1)
        with lock1:  # 等待 lock1（但线程1持有lock1）
            print("线程2完成")

# t1 = threading.Thread(target=thread1)
# t2 = threading.Thread(target=thread2)
# t1.start()
# t2.start()
# 死锁！线程1持有lock1等待lock2，线程2持有lock2等待lock1
```

### 6.2 避免死锁

```python
# 方法一：始终按相同顺序获取锁
lock1 = threading.Lock()
lock2 = threading.Lock()

def safe_thread1():
    with lock1:
        time.sleep(0.1)
        with lock2:  # 始终先获取 lock1，再获取 lock2
            print("线程1完成")

def safe_thread2():
    with lock1:  # 始终按相同顺序
        time.sleep(0.1)
        with lock2:
            print("线程2完成")

# 方法二：使用超时
def with_timeout(lock, timeout=1):
    return lock.acquire(timeout=timeout)

# 方法三：使用 RLock（但要注意顺序）
```

## 八、易错点与避坑指南

1. **GIL 误解**：CPU 密集型用多进程，I/O 密集型用多线程
2. **共享状态**：全局变量在线程间共享，注意竞争条件
3. **守护线程**：主线程结束，守护线程也会被强制结束
4. **死锁**：避免多个线程以不同顺序获取锁
5. **资源泄露**：确保锁在使用后被释放
