### 第 2 步：使用外链 API（15 分钟）

```python
import requests
import os

def get_song_url(song_id):
    """获取歌曲播放 URL（使用外链 API）"""
    # 外链 API
    url = f"https://music.163.com/song/media/outer/url?id={song_id}.mp3"

    headers = {
        "User-Agent": "Mozilla/5.0 (Windows NT 10.0; Win64; x64) AppleWebKit/537.36",
        "Referer": "https://music.163.com/",
    }

    response = requests.get(url, headers=headers, allow_redirects=False)

    if response.status_code == 302:
        # 重定向到真实 URL
        real_url = response.headers.get("Location")
        return real_url
    elif response.status_code == 200:
        return url
    else:
        return None

def download_song(song_id, song_name=None, save_dir="songs"):
    """下载网易云歌曲"""
    os.makedirs(save_dir, exist_ok=True)

    # 获取歌曲 URL
    song_url = get_song_url(song_id)

    if song_url is None:
        print(f"[失败] {song_id} - 无法获取歌曲 URL（可能是 VIP 或下架）")
        return None

    # 下载歌曲
    headers = {
        "User-Agent": "Mozilla/5.0 (Windows NT 10.0; Win64; x64) AppleWebKit/537.36",
        "Referer": "https://music.163.com/",
    }

    print(f"[下载中] {song_id}...")

    response = requests.get(song_url, headers=headers, stream=True)
    response.raise_for_status()

    # 保存文件
    if song_name is None:
        song_name = f"song_{song_id}"

    filename = os.path.join(save_dir, f"{song_name}.mp3")
    with open(filename, "wb") as f:
        for chunk in response.iter_content(chunk_size=8192):
            if chunk:
                f.write(chunk)

    print(f"[成功] {song_id} -> {filename}")
    return filename
```

```python
import requests

def get_song_info(song_id):
    """获取歌曲详细信息"""
    api_url = f"https://music.163.com/api/song/detail?id={song_id}&ids=%5B{song_id}%5D"

    headers = {
        "User-Agent": "Mozilla/5.0 (Windows NT 10.0; Win64; x64) AppleWebKit/537.36",
        "Referer": "https://music.163.com/",
    }

    response = requests.get(api_url, headers=headers)
    data = response.json()

    if data["code"] == 200:
        song = data["songs"][0]
        return {
            "id": song["id"],
            "name": song["name"],
            "artist": song["artists"][0]["name"],
            "album": song["album"]["name"],
            "duration": song["duration"] / 1000,  # 毫秒转秒
            "pic_url": song["album"]["picUrl"],
        }
    return None

    # 这个 API 需要登录 Cookie 或复杂的加密参数
```

### 第 4 步：批量下载歌单歌曲（15 分钟）

```python
import requests
import os
import time

def get_playlist_songs(playlist_id):
    """获取歌单中的所有歌曲"""
    api_url = f"https://music.163.com/api/playlist/detail?id={playlist_id}"

    headers = {
        "User-Agent": "Mozilla/5.0 (Windows NT 10.0; Win64; x64) AppleWebKit/537.36",
        "Referer": "https://music.163.com/",
    }

    response = requests.get(api_url, headers=headers)
    data = response.json()

    if data["code"] == 200:
        songs = data["result"]["tracks"]
        return [
            {
                "id": song["id"],
                "name": song["name"],
                "artist": song["artists"][0]["name"],
            }
            for song in songs
        ]
    return []

def download_playlist(playlist_id, save_dir=None):
    """下载整个歌单"""
    print(f"获取歌单 {playlist_id} 的歌曲...")

    songs = get_playlist_songs(playlist_id)
    print(f"找到 {len(songs)} 首歌曲")

    if save_dir is None:
        save_dir = f"playlist_{playlist_id}"
    os.makedirs(save_dir, exist_ok=True)

    success_count = 0
    for i, song in enumerate(songs):
        print(f"\n[{i+1}/{len(songs)}] {song['name']} - {song['artist']}")

        # 清理文件名
        safe_name = "".join(c for c in song["name"] if c not in r'\/:*?"<>|')
        safe_name = f"{song['id']}_{safe_name}"

        try:
            result = download_song(song["id"], safe_name, save_dir)
            if result:
                success_count += 1
        except Exception as e:
            print(f"[失败] {e}")

        # 避免请求过快
        time.sleep(0.5)

    print(f"\n完成: {success_count}/{len(songs)} 首歌")
    return success_count
```
