## 一、装饰器核心概念

### 1.1 什么是装饰器

装饰器是一种高级函数，它可以在不修改原函数代码的情况下，给函数添加额外的功能。

```
未使用装饰器：
函数 → 功能

使用装饰器：
函数 → 装饰器包装 → 增强后的功能
```

```python
# 最简单的装饰器
def my_decorator(func):
    """装饰器函数"""
    def wrapper():
        print("调用函数之前")
        func()
        print("调用函数之后")
    return wrapper

def say_hello():
    print("你好！")

# 手动装饰
decorated = my_decorator(say_hello)
decorated()

# 输出：
# 调用函数之前
# 你好！
# 调用函数之后
```

### 1.2 装饰器的等价形式

```python
# 语法糖形式
@my_decorator
def say_hello():
    print("你好！")

# 等价于
def say_hello():
    print("你好！")
say_hello = my_decorator(say_hello)

# 调用
say_hello()
```

### 1.3 为什么需要装饰器

| 场景 | 不用装饰器 | 用装饰器 |
|------|------------|----------|
| 日志记录 | 在每个函数内加 print | 一个装饰器，所有函数复用 |
| 性能计时 | 每个函数内单独计时 | 一个装饰器，统一处理 |
| 权限验证 | 每个函数内单独检查 | 一个装饰器，统一验证 |
| 缓存结果 | 每个函数内单独实现 | 一个装饰器，通用实现 |

## 二、函数即对象

### 2.1 函数是一等公民

在 Python 中，函数和其他对象一样，可以：
- 被赋值给变量
- 作为参数传递给其他函数
- 作为返回值
- 存储在数据结构中

```python
# 函数可以赋值给变量
def greet():
    print("Hello!")

say_hello = greet  # 注意：不要加括号
say_hello()  # Hello!

# 函数可以作为参数
def execute_twice(func):
    func()
    func()

execute_twice(greet)
# 输出：Hello!\nHello!

# 函数可以作为返回值
def create_multiplier(factor):
    def multiply(number):
        return number * factor
    return multiply

double = create_multiplier(2)
print(double(5))  # 10
```

### 2.2 闭包回顾

```python
# 闭包 = 函数 + 环境（外部变量引用）
def outer():
    x = 10
    
    def inner():
        print(x)  # 引用外部变量 x
    
    return inner

closure = outer()
closure()  # 输出 10

# 闭包会"记住"环境
def make_counter():
    count = 0
    
    def counter():
        nonlocal count
        count += 1
        return count
    
    return counter

c1 = make_counter()
c2 = make_counter()

print(c1())  # 1
print(c1())  # 2
print(c2())  # 1（独立的计数器）
```

## 三、基础装饰器实现

### 3.1 日志装饰器

```python
import functools
from datetime import datetime

def log_calls(func):
    """记录函数调用日志"""
    @functools.wraps(func)  # 保留原函数信息
    def wrapper(*args, **kwargs):
        print(f"[{datetime.now().strftime('%Y-%m-%d %H:%M:%S')}] 调用 {func.__name__}")
        print(f"  参数：args={args}, kwargs={kwargs}")
        result = func(*args, **kwargs)
        print(f"  返回：{result}")
        return result
    return wrapper

@log_calls
def add(a, b):
    """两数相加"""
    return a + b

@log_calls
def greet(name, greeting="Hello"):
    """打招呼"""
    return f"{greeting}, {name}!"

# 使用
print(add(3, 5))
print(greet("Alice"))
```

### 3.2 计时装饰器

```python
import functools
import time

def timing(func):
    """函数执行计时"""
    @functools.wraps(func)
    def wrapper(*args, **kwargs):
        start = time.perf_counter()
        result = func(*args, **kwargs)
        end = time.perf_counter()
        print(f"{func.__name__} 执行用时：{end - start:.4f}秒")
        return result
    return wrapper

@timing
def slow_function():
    time.sleep(0.5)
    return "完成"

@timing
def fibonacci(n):
    """递归计算斐波那契数"""
    if n <= 1:
        return n
    return fibonacci(n - 1) + fibonacci(n - 2)

slow_function()
print(f"fib(20) = {fibonacci(20)}")
```

### 3.3 验证装饰器

```python
import functools

def validate_positive(func):
    """验证参数为正数"""
    @functools.wraps(func)
    def wrapper(*args, **kwargs):
        for i, arg in enumerate(args):
            if isinstance(arg, (int, float)) and arg <= 0:
                raise ValueError(f"参数 {i+1} 必须为正数，当前值：{arg}")
        for key, val in kwargs.items():
            if isinstance(val, (int, float)) and val <= 0:
                raise ValueError(f"参数 {key} 必须为正数，当前值：{val}")
        return func(*args, **kwargs)
    return wrapper

@validate_positive
def calculate_area(radius):
    """计算圆面积"""
    return 3.14159 * radius ** 2

@validate_positive
def divide(a, b):
    """两数相除"""
    return a / b

# 测试
print(calculate_area(5))     # 78.53975
# print(calculate_area(-5))   # ValueError

print(divide(10, 2))         # 5.0
# print(divide(10, -2))      # ValueError
```

## 四、带参数的装饰器

### 4.1 装饰器工厂函数

```python
import functools

def repeat(times):
    """重复执行函数指定次数"""
    def decorator(func):
        @functools.wraps(func)
        def wrapper(*args, **kwargs):
            results = []
            for _ in range(times):
                result = func(*args, **kwargs)
                results.append(result)
            return results
        return wrapper
    return decorator

@repeat(times=3)
def say_hello():
    return "Hello!"

print(say_hello())  # ['Hello!', 'Hello!', 'Hello!']

@repeat(5)
def get_random():
    import random
    return random.randint(1, 10)

print(get_random())  # [7, 3, 9, 2, 5]
```

### 4.2 条件装饰器

```python
import functools

def debug(enabled=True):
    """调试装饰器，可开关"""
    def decorator(func):
        @functools.wraps(func)
        def wrapper(*args, **kwargs):
            if enabled:
                print(f"DEBUG: 调用 {func.__name__}")
            return func(*args, **kwargs)
        return wrapper
    return decorator

@debug(enabled=True)
def process():
    print("处理中...")

@debug(enabled=False)
def skip():
    print("跳过处理")

process()  # DEBUG: 调用 process\n处理中...
skip()     # 跳过处理
```

### 4.3 缓存装饰器

```python
import functools

def memoize(func):
    """缓存装饰器"""
    cache = {}
    
    @functools.wraps(func)
    def wrapper(*args):
        if args in cache:
            print(f"缓存命中：{args}")
            return cache[args]
        result = func(*args)
        cache[args] = result
        return result
    
    # 给 wrapper 添加方法以便查看缓存
    wrapper.cache = cache
    wrapper.clear_cache = lambda: cache.clear()
    return wrapper

@memoize
def fibonacci(n):
    """斐波那契数列（带缓存）"""
    if n <= 1:
        return n
    return fibonacci(n - 1) + fibonacci(n - 2)

print(fibonacci(10))  # 55
print(fibonacci(10))  # 缓存命中
print(f"缓存大小：{len(fibonacci.cache)}")
```

## 五、类装饰器

### 5.1 使用类的装饰器

```python
class CallCounter:
    """计数器装饰器（类形式）"""
    
    def __init__(self, func):
        self.func = func
        self.count = 0
        functools.update_wrapper(self, func)
    
    def __call__(self, *args, **kwargs):
        self.count += 1
        print(f"{self.func.__name__} 被调用了 {self.count} 次")
        return self.func(*args, **kwargs)

@CallCounter
def add(a, b):
    return a + b

@CallCounter
def multiply(a, b):
    return a * b

add(1, 2)
add(3, 4)
multiply(5, 6)
add(7, 8)
print(f"add 被调用了 {add.count} 次")
```

### 5.2 带参数的类装饰器

```python
class RateLimit:
    """限流装饰器（类形式）"""
    
    def __init__(self, max_calls, period=60):
        self.max_calls = max_calls
        self.period = period
        self.calls = []
    
    def __call__(self, func):
        @functools.wraps(func)
        def wrapper(*args, **kwargs):
            import time
            now = time.time()
            
            # 清理超出时间窗口的调用记录
            self.calls = [t for t in self.calls if now - t < self.period]
            
            if len(self.calls) >= self.max_calls:
                raise RuntimeError(f"调用频率超限：最多 {self.max_calls} 次/{self.period}秒")
            
            self.calls.append(now)
            return func(*args, **kwargs)
        
        return wrapper

@RateLimit(max_calls=3, period=5)
def send_message(msg):
    print(f"发送消息：{msg}")

send_message("消息1")
send_message("消息2")
send_message("消息3")
# send_message("消息4")  # RuntimeError
```

## 六、functools.wraps 的重要性

### 6.1 不使用 wraps 的问题

```python
# ❌ 错误：不保留原函数信息
def bad_decorator(func):
    def wrapper(*args, **kwargs):
        return func(*args, **kwargs)
    return wrapper

@bad_decorator
def add(a, b):
    """两数相加"""
    return a + b

print(add.__name__)       # wrapper（应该是 add）
print(add.__doc__)        # None（应该是"两数相加"）
print(add.__module__)     # __main__（正确）
```

### 6.2 正确使用 wraps

```python
import functools

# ✅ 正确：保留原函数信息
def good_decorator(func):
    @functools.wraps(func)  # 必须放在 wrapper 上面
    def wrapper(*args, **kwargs):
        return func(*args, **kwargs)
    return wrapper

@good_decorator
def add(a, b):
    """两数相加"""
    return a + b

print(add.__name__)       # add
print(add.__doc__)        # 两数相加
print(add.__module__)     # __main__
```

## 七、多重装饰器

### 7.1 装饰器执行顺序

```python
def decorator_a(func):
    def wrapper(*args, **kwargs):
        print("A before")
        result = func(*args, **kwargs)
        print("A after")
        return result
    return wrapper

def decorator_b(func):
    def wrapper(*args, **kwargs):
        print("B before")
        result = func(*args, **kwargs)
        print("B after")
        return result
    return wrapper

@decorator_a
@decorator_b
def say():
    print("Hello!")

say()

# 执行顺序：
# A before
# B before
# Hello!
# B after
# A after
```

### 7.2 组合使用

```python
import functools
import time

def log(func):
    """日志装饰器"""
    @functools.wraps(func)
    def wrapper(*args, **kwargs):
        print(f"调用 {func.__name__}")
        return func(*args, **kwargs)
    return wrapper

def timing(func):
    """计时装饰器"""
    @functools.wraps(func)
    def wrapper(*args, **kwargs):
        start = time.time()
        result = func(*args, **kwargs)
        print(f"用时：{time.time() - start:.4f}秒")
        return result
    return wrapper

def validate(func):
    """验证装饰器"""
    @functools.wraps(func)
    def wrapper(*args, **kwargs):
        print("验证通过")
        return func(*args, **kwargs)
    return wrapper

@log
@timing
@validate
def process_data(data):
    """处理数据"""
    time.sleep(0.1)
    return f"处理了 {data}"

print(process_data("测试数据"))
```

## 八、实战应用

### 8.1 注册函数系统

```python
import functools

class FunctionRegistry:
    """函数注册表"""
    
    _registry = {}
    
    @classmethod
    def register(cls, name=None):
        """注册装饰器"""
        def decorator(func):
            func_name = name or func.__name__
            cls._registry[func_name] = func
            
            @functools.wraps(func)
            def wrapper(*args, **kwargs):
                return func(*args, **kwargs)
            
            return wrapper
        return decorator
    
    @classmethod
    def call(cls, name, *args, **kwargs):
        """调用已注册的函数"""
        if name not in cls._registry:
            raise ValueError(f"未找到函数：{name}")
        return cls._registry[name](*args, **kwargs)
    
    @classmethod
    def list_functions(cls):
        """列出所有已注册函数"""
        return list(cls._registry.keys())

# 使用示例
@FunctionRegistry.register("add")
def add(a, b):
    return a + b

@FunctionRegistry.register("multiply")
def multiply(a, b):
    return a * b

@FunctionRegistry.register("greet")
def greet(name):
    return f"Hello, {name}!"

print(FunctionRegistry.list_functions())
print(FunctionRegistry.call("add", 3, 5))
print(FunctionRegistry.call("greet", "Alice"))
```

### 8.2 自动重试装饰器

```python
import functools
import time

def retry(max_attempts=3, delay=1, exceptions=(Exception,)):
    """自动重试装饰器"""
    def decorator(func):
        @functools.wraps(func)
        def wrapper(*args, **kwargs):
            last_exception = None
            for attempt in range(max_attempts):
                try:
                    return func(*args, **kwargs)
                except exceptions as e:
                    last_exception = e
                    if attempt < max_attempts - 1:
                        print(f"尝试 {attempt + 1} 失败，{delay}秒后重试...")
                        time.sleep(delay)
            raise last_exception
        return wrapper
    return decorator

@retry(max_attempts=3, delay=1)
def fetch_data():
    import random
    if random.random() < 0.7:  # 70% 概率失败
        raise ConnectionError("网络连接失败")
    return "数据获取成功"

# 尝试执行
for i in range(5):
    print(f"第 {i+1} 次尝试：")
    try:
        result = fetch_data()
        print(f"成功：{result}")
        break
    except ConnectionError as e:
        print(f"失败：{e}")
```

### 8.3 检查返回类型

```python
import functools

def returns(expected_type):
    """检查返回类型的装饰器"""
    def decorator(func):
        @functools.wraps(func)
        def wrapper(*args, **kwargs):
            result = func(*args, **kwargs)
            if not isinstance(result, expected_type):
                raise TypeError(
                    f"{func.__name__} 应返回 {expected_type.__name__}，"
                    f"实际返回 {type(result).__name__}"
                )
            return result
        return wrapper
    return decorator

@returns(int)
def add(a, b):
    return a + b

@returns(list)
def get_items():
    return [1, 2, 3]

print(add(1, 2))           # 3
print(get_items())         # [1, 2, 3]
# add("a", "b")           # TypeError（返回了 str）
```

## 十、易错点与避坑指南

1. **忘记 functools.wraps**：导致函数元数据丢失
2. **装饰器顺序错误**：装饰器从下到上执行，从上到下理解
3. **闭包变量未声明**：使用 `nonlocal` 修改外部变量
4. **参数传递问题**：使用 `*args, **kwargs` 传递所有参数
5. **装饰器修改函数签名**：导致 `inspect` 模块无法正确识别
