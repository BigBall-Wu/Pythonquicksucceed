```python
import requests
import re
import json

def get_mv_url(mv_id):
    """获取 MV 的播放 URL"""
    url = f"https://music.163.com/mv?id={mv_id}"

    headers = {
        "User-Agent": "Mozilla/5.0 (Windows NT 10.0; Win64; x64) AppleWebKit/537.36",
        "Referer": "https://music.163.com/",
    }

    response = requests.get(url, headers=headers)
    response.encoding = "utf-8"

    # 方法 1：从页面提取 JSON 数据
    # 网易云 MV 页面通常包含视频 URL
    patterns = [
        r'<script>window\.__INITIAL_STATE__=(.+?)</script>',
        r'"url"\s*:\s*"([^"]+mp4[^"]*)"',
        r'"brUrl"\s*:\s*"([^"]+)"',
    ]

    for pattern in patterns:
        match = re.search(pattern, response.text)
        if match:
            data_str = match.group(1)
            try:
                # 尝试解析 JSON
                data = json.loads(data_str)
                # 从复杂结构中提取 URL
                # ...
            except:
                # 可能直接匹配到 URL
                url = match.group(1).replace("\\/", "/")
                return url

    return None

def get_mv_url_api(mv_id):
    """使用 API 获取 MV URL（推荐）"""
    # 网易云有 MV 详情 API
    api_url = f"https://music.163.com/api/mv/detail?id={mv_id}&ids=%5B{mv_id}%5D"

    headers = {
        "User-Agent": "Mozilla/5.0 (Windows NT 10.0; Win64; x64) AppleWebKit/537.36",
        "Referer": "https://music.163.com/",
    }

    response = requests.get(api_url, headers=headers)
    data = response.json()

    if data["code"] == 200:
        mv_data = data["data"]
        # 获取不同清晰度的 URL
        urls = mv_data.get("brs", {})  # 不同码率的 URL
        if urls:
            # 返回最高清晰度的 URL
            return urls.get("1080") or urls.get("720") or urls.get("480")
        return mv_data.get("url")
    return None
```

### 第 3 步：下载 MV（15 分钟）

```python
import requests
import os

def download_mv(mv_id, save_dir="mvs", quality="1080p"):
    """下载网易云 MV"""
    os.makedirs(save_dir, exist_ok=True)

    # 获取 MV 信息
    api_url = f"https://music.163.com/api/mv/detail?id={mv_id}&ids=%5B{mv_id}%5D"

    headers = {
        "User-Agent": "Mozilla/5.0 (Windows NT 10.0; Win64; x64) AppleWebKit/537.36",
        "Referer": "https://music.163.com/",
    }

    print(f"获取 MV {mv_id} 信息...")
    response = requests.get(api_url, headers=headers)
    data = response.json()

    if data["code"] != 200:
        print("获取 MV 信息失败")
        return None

    mv_data = data["data"]
    mv_name = mv_data.get("name", f"mv_{mv_id}")

    # 获取 URL（按清晰度）
    quality_map = {"1080p": "1080", "720p": "720", "480p": "480", "360p": "360"}
    br_url_key = quality_map.get(quality, "720")

    urls = mv_data.get("brs", {})
    mv_url = urls.get(br_url_key)

    if not mv_url:
        # 尝试其他清晰度
        for q in ["720", "480", "360"]:
            if q in urls:
                mv_url = urls[q]
                print(f"使用 {q}p 清晰度")
                break

    if not mv_url:
        print("无法获取 MV 视频 URL（可能需要 VIP）")
        return None

    print(f"MV 名称: {mv_name}")
    print(f"清晰度: {quality}")
    print(f"下载地址: {mv_url}")

    # 下载视频
    filename = f"{mv_name}_{mv_id}.mp4"
    filepath = os.path.join(save_dir, filename)

    print(f"\n开始下载...")
    response = requests.get(mv_url, headers=headers, stream=True)
    response.raise_for_status()

    total_size = int(response.headers.get("Content-Length", 0))
    downloaded = 0

    with open(filepath, "wb") as f:
        for chunk in response.iter_content(chunk_size=1024 * 1024):  # 1MB
            f.write(chunk)
            downloaded += len(chunk)
            if total_size:
                pct = downloaded / total_size * 100
                print(f"\r下载进度: {pct:.1f}%", end="", flush=True)

    print(f"\n保存到: {filepath}")
    return filepath
```

### 第 4 步：ffmpeg 处理视频（15 分钟）

```python
# ffmpeg 安装：
# Mac: brew install ffmpeg

# Windows: https://ffmpeg.org/download.html

import subprocess
import os

def extract_audio(mv_path, output_path=None):
    """从 MV 中提取音频"""
    if output_path is None:
        output_path = mv_path.replace(".mp4", ".mp3")

    cmd = ["ffmpeg", "-i", mv_path, "-vn", "-c:a", "mp3", output_path]
    subprocess.run(cmd, capture_output=True)
    print(f"音频提取到: {output_path}")

def convert_format(input_path, output_path, codec="libx264"):
    """转换视频格式"""
    cmd = ["ffmpeg", "-i", input_path, "-c:v", codec, "-c:a", "copy", output_path]
    subprocess.run(cmd, capture_output=True)
    print(f"转换完成: {output_path}")

def compress_video(input_path, output_path, crf=23):
    """压缩视频（降低文件大小）"""
    # CRF 值：18-28，越小质量越高
    cmd = [
        "ffmpeg", "-i", input_path,
        "-c:v", "libx264",
        "-crf", str(crf),
        "-c:a", "aac",
        "-b:a", "128k",
        output_path
    ]
    subprocess.run(cmd, capture_output=True)
    print(f"压缩完成: {output_path}")

def get_video_info(path):
    """获取视频信息"""
    cmd = ["ffprobe", "-v", "quiet", "-print_format", "json",
           "-show_format", "-show_streams", path]
    result = subprocess.run(cmd, capture_output=True, text=True)
    import json
    return json.loads(result.stdout)
```
