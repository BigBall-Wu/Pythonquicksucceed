### 第 1 步：单例模式（10 分钟）

```python
# ❌ 普通类：每次实例化都创建新对象
class Database:
    def __init__(self):
        print("创建新连接")
        self.connection = "DB connection"

d1 = Database()                 # 创建
d2 = Database()                 # 又创建
print(d1 is d2)                  # False（不同对象）

# ✓ 单例：整个程序只有一个实例
class Singleton:
    _instance = None

    def __new__(cls, *args, **kwargs):
        if cls._instance is None:
            cls._instance = super().__new__(cls)
            cls._instance._initialized = False
        return cls._instance

    def __init__(self):
        if not self._initialized:
            print("初始化")
            self.connection = "DB connection"
            self._initialized = True

s1 = Singleton()
s2 = Singleton()
print(s1 is s2)                  # True（同一对象！）
```

### 第 2 步：3 种单例实现方式（10 分钟）

```python
# 方式 1：__new__ 方法（标准）
class Singleton1:
    _instance = None
    def __new__(cls, *args, **kwargs):
        if cls._instance is None:
            cls._instance = super().__new__(cls)
        return cls._instance

# 方式 2：装饰器
def singleton(cls):
    instances = {}
    def wrapper(*args, **kwargs):
        if cls not in instances:
            instances[cls] = cls(*args, **kwargs)
        return instances[cls]
    return wrapper

@singleton
class Singleton2:
    pass

s1 = Singleton2()
s2 = Singleton2()
print(s1 is s2)                  # True

# 方式 3：元类（高级）
class SingletonMeta(type):
    _instances = {}
    def __call__(cls, *args, **kwargs):
        if cls not in SingletonMeta._instances:
            SingletonMeta._instances[cls] = super().__call__(*args, **kwargs)
        return SingletonMeta._instances[cls]

class Singleton3(metaclass=SingletonMeta):
    pass
```

### 第 3 步：理解魔法方法（10 分钟）

**魔法方法（Dunder Methods）**：以双下划线开头和结尾的方法，Python 在特定场景自动调用。

```python
class Vector:
    def __init__(self, x, y):
        """构造器"""
        self.x = x
        self.y = y

    def __repr__(self):
        """开发者字符串（repr() 调用）"""
        return f"Vector({self.x}, {self.y})"

    def __str__(self):
        """用户字符串（str()、print() 调用）"""
        return f"({self.x}, {self.y})"

    def __add__(self, other):
        """加法：v1 + v2"""
        return Vector(self.x + other.x, self.y + other.y)

    def __eq__(self, other):
        """相等：v1 == v2"""
        return self.x == other.x and self.y == other.y

    def __len__(self):
        """长度：len(v)"""
        return 2

# 用
v1 = Vector(1, 2)
v2 = Vector(3, 4)
print(v1)                        # (1, 2)（用 __str__）
print(repr(v1))                  # Vector(1, 2)（用 __repr__）
v3 = v1 + v2                     # 用 __add__
print(v3)                        # (4, 6)
print(v1 == Vector(1, 2))        # True（用 __eq__）
print(len(v1))                    # 2（用 __len__）
```

### 第 5 步：实战 - 单词统计类（10 分钟）

```python
class WordCounter:
    def __init__(self, text):
        self.text = text
        self.words = text.split()
        self._count = len(self.words)

    def __len__(self):
        return self._count

    def __getitem__(self, index):
        return self.words[index]

    def __contains__(self, word):
        return word in self.words

    def __iter__(self):
        return iter(self.words)

    def __add__(self, other):
        new_text = self.text + " " + other.text
        return WordCounter(new_text)

    def __repr__(self):
        return f"WordCounter({len(self)} words)"

# 用
wc1 = WordCounter("Python is awesome")
wc2 = WordCounter("Python is fun")
print(len(wc1))                   # 3
print(wc1[0])                     # Python
print("is" in wc1)                 # True
for word in wc1:                  # 迭代
    print(word, end=" ")
print()
print(wc1 + wc2)                  # WordCounter(5 words)
```

### 第 6 步：常见错误（5 分钟）

```python
# 错误 1：单例变多例
class Bug:
    _instance = None
    def __new__(cls):
        return object.__new__(cls)  # ❌ 每次创建新对象

# 正确：要先判断 _instance
def __new__(cls):
    if cls._instance is None:
        cls._instance = object.__new__(cls)
    return cls._instance

# 错误 2：魔法方法名打错
class Typo:
    def _init_(self):               # ❌ 单下划线
        pass
    # 正确：__init__（双下划线）
```

class Complex:
    def __init__(self, real, imag):
        self.real = real
        self.imag = imag

    def __add__(self, other):
        return Complex(self.real + other.real, self.imag + other.imag)

    def __repr__(self):
        return f"{self.real}+{self.imag}i"

c1 = Complex(1, 2)
c2 = Complex(3, 4)
print(c1 + c2)                    # 4+6i
