### 1.1 瓶颈识别

```python
import time
import functools
from contextlib import contextmanager

@contextmanager
def timer(name: str):
    """计时上下文管理器"""
    start = time.time()
    yield
    elapsed = time.time() - start
    print(f"{name} 耗时: {elapsed:.3f}s")

# 使用
with timer("抓取页面"):
    response = fetch(url)

# 装饰器版本
def timed(func):
    """计时装饰器"""
    @functools.wraps(func)
    def wrapper(*args, **kwargs):
        start = time.time()
        result = func(*args, **kwargs)
        elapsed = time.time() - start
        print(f"{func.__name__} 耗时: {elapsed:.3f}s")
        return result
    return wrapper
```

```python
import asyncio
import aiohttp

async def batch_fetch(urls, concurrency=50):
    """批量并发请求"""
    semaphore = asyncio.Semaphore(concurrency)
    
    async def fetch_with_limit(url):
        async with semaphore:
            async with aiohttp.ClientSession() as session:
                async with session.get(url) as response:
                    return await response.text()
    
    tasks = [fetch_with_limit(url) for url in urls]
    results = await asyncio.gather(*tasks, return_exceptions=True)
    
    return results
```

### 2.3 异步批量写入

```python
import asyncio
from motor.motor_asyncio import AsyncIOMotorClient

class AsyncMongoWriter:
    """异步 MongoDB 写入"""
    
    def __init__(self, mongo_uri, db_name, collection_name, batch_size=100):
        self.client = AsyncIOMotorClient(mongo_uri)
        self.collection = self.client[db_name][collection_name]
        self.batch_size = batch_size
        self.buffer = []
    
    async def write(self, item):
        """写入缓冲区"""
        self.buffer.append(item)
        
        if len(self.buffer) >= self.batch_size:
            await self.flush()
    
    async def flush(self):
        """刷新缓冲区"""
        if not self.buffer:
            return
        
        await self.collection.insert_many(self.buffer)
        self.buffer = []
    
    async def close(self):
        """关闭时刷新剩余数据"""
        await self.flush()
```

## 四、错误处理与重试

### 4.1 指数退避

```python
import asyncio
import aiohttp

async def fetch_with_retry(url, max_retries=5, base_delay=1):
    """指数退避重试"""
    for attempt in range(max_retries):
        try:
            async with aiohttp.ClientSession() as session:
                async with session.get(url) as response:
                    if response.status == 200:
                        return await response.text()
                    elif response.status == 429:
                        delay = base_delay * (2 ** attempt)
                        print(f"频率限制，等待 {delay}s")
                        await asyncio.sleep(delay)
                    else:
                        raise aiohttp.ClientError(f"HTTP {response.status}")
        except Exception as e:
            if attempt == max_retries - 1:
                raise
            delay = base_delay * (2 ** attempt)
            await asyncio.sleep(delay)
    
    return None
```

### 4.2 断点续爬

```python
import json
from pathlib import Path

class CheckpointManager:
    """断点续爬管理"""
    
    def __init__(self, checkpoint_file):
        self.checkpoint_file = Path(checkpoint_file)
        self.checkpoint = self.load()
    
    def load(self):
        """加载检查点"""
        if self.checkpoint_file.exists():
            with open(self.checkpoint_file, "r") as f:
                return json.load(f)
        return {"completed": [], "failed": [], "last_index": 0}
    
    def save(self):
        """保存检查点"""
        with open(self.checkpoint_file, "w") as f:
            json.dump(self.checkpoint, f, indent=2)
    
    def mark_completed(self, item_id):
        """标记完成"""
        if item_id not in self.checkpoint["completed"]:
            self.checkpoint["completed"].append(item_id)
        self.save()
    
    def mark_failed(self, item_id, error):
        """标记失败"""
        self.checkpoint["failed"].append({
            "id": item_id,
            "error": str(error)
        })
        self.save()
    
    def get_pending(self, all_items):
        """获取待处理项"""
        completed = set(self.checkpoint["completed"])
        return [item for item in all_items if item["id"] not in completed]
```

### 5.1 指标采集

```python
import time
from dataclasses import dataclass
from datetime import datetime

@dataclass
class Metrics:
    """爬虫指标"""
    start_time: float
    success_count: int = 0
    failed_count: int = 0
    total_count: int = 0
    
    @property
    def elapsed_time(self):
        return time.time() - self.start_time
    
    @property
    def success_rate(self):
        if self.total_count == 0:
            return 0
        return self.success_count / self.total_count * 100
    
    @property
    def speed(self):
        if self.elapsed_time == 0:
            return 0
        return self.total_count / self.elapsed_time
    
    def to_dict(self):
        return {
            "elapsed_time": f"{self.elapsed_time:.2f}s",
            "success_count": self.success_count,
            "failed_count": self.failed_count,
            "total_count": self.total_count,
            "success_rate": f"{self.success_rate:.1f}%",
            "speed": f"{self.speed:.2f}/s"
        }

# 使用
metrics = Metrics(start_time=time.time())

for url in urls:
    try:
        crawl(url)
        metrics.success_count += 1
    except:
        metrics.failed_count += 1
    finally:
        metrics.total_count += 1

print(metrics.to_dict())
```

```python
import logging
from logging.handlers import RotatingFileHandler

def setup_logger(name, log_file, max_bytes=10*1024*1024, backup_count=5):
    """配置日志（带轮转）"""
    logger = logging.getLogger(name)
    logger.setLevel(logging.INFO)
    
    # 格式
    formatter = logging.Formatter(
        "%(asctime)s - %(name)s - %(levelname)s - %(message)s"
    )
    
    # 控制台
    console = logging.StreamHandler()
    console.setFormatter(formatter)
    logger.addHandler(console)
    
    # 文件（轮转）
    file_handler = RotatingFileHandler(
        log_file,
        maxBytes=max_bytes,
        backupCount=backup_count
    )
    file_handler.setFormatter(formatter)
    logger.addHandler(file_handler)
    
    return logger
```

### 6.2 结构化日志

```python
import json
import logging

class JSONFormatter(logging.Formatter):
    """JSON 日志格式"""
    
    def format(self, record):
        log_data = {
            "timestamp": self.formatTime(record),
            "level": record.levelname,
            "logger": record.name,
            "message": record.getMessage(),
        }
        
        if hasattr(record, "spider"):
            log_data["spider"] = record.spider
        
        if hasattr(record, "url"):
            log_data["url"] = record.url
        
        if record.exc_info:
            log_data["exception"] = self.formatException(record.exc_info)
        
        return json.dumps(log_data)

# 使用
logger = logging.getLogger("crawler")
handler = logging.StreamHandler()
handler.setFormatter(JSONFormatter())
logger.addHandler(handler)

logger.info("开始爬取", extra={"spider": "news", "url": "https://example.com"})
```

## 七、日常维护

### 7.1 维护脚本

```python
# maintenance.py
import redis
from pymongo import MongoClient
from datetime import datetime, timedelta

class Maintenance:
    """日常维护"""
    
    def __init__(self):
        self.r = redis.Redis(decode_responses=True)
        self.mongo = MongoClient()
        self.db = self.mongo["crawler_db"]
    
    def cleanup_old_data(self, days=30):
        """清理过期数据"""
        cutoff = datetime.now() - timedelta(days=days)
        
        # MongoDB
        result = self.db.articles.delete_many({
            "created_at": {"$lt": cutoff}
        })
        print(f"删除 MongoDB 数据: {result.deleted_count}")
        
        # Redis 清理过期键
        for key in self.r.scan_iter("crawler:*:dupefilter"):
            self.r.delete(key)
    
    def rebuild_dupefilter(self, spider_name):
        """重建去重过滤器"""
        # 从已爬取 URL 重建
        cursor = self.db[spider_name].find({}, {"url": 1})
        
        count = 0
        for doc in cursor:
            self.r.sadd(f"{spider_name}:dupefilter", doc["url"])
            count += 1
        
        print(f"重建去重过滤器: {count} 条")
    
    def stats_report(self):
        """统计报告"""
        stats = {
            "mongodb": {
                "articles": self.db.articles.count_documents({}),
                "crawled_today": self.db.articles.count_documents({
                    "created_at": {"$gte": datetime.now().replace(hour=0, minute=0, second=0)}
                })
            },
            "redis": {
                "keys": self.r.dbsize(),
                "memory": self.r.info("memory")["used_memory_human"]
            }
        }
        
        print("=== 统计报告 ===")
        print(f"MongoDB 文章数: {stats['mongodb']['articles']}")
        print(f"今日爬取: {stats['mongodb']['crawled_today']}")
        print(f"Redis 键数: {stats['redis']['keys']}")
        print(f"Redis 内存: {stats['redis']['memory']}")
        
        return stats

if __name__ == "__main__":
    m = Maintenance()
    m.stats_report()
```
