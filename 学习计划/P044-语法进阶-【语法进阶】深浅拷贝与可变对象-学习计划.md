## 一、Python 对象模型

### 1.1 一切皆对象

在 Python 中，一切皆为对象。每个对象都有三个特征：
- **身份（id）**：对象的唯一标识，类似于内存地址
- **类型（type）**：对象的数据类型
- **值（value）**：对象存储的数据

```python
# 对象的三个特征
x = 42

print(id(x))   # 身份：例如 140712345678912
print(type(x)) # 类型：<class 'int'>
print(x)       # 值：42

# 列表也是对象
lst = [1, 2, 3]
print(id(lst))  # 与 x 不同的 id
print(type(lst)) # <class 'list'>
```

### 1.2 变量与对象的关系

```python
# 变量是对象的引用（标签）
a = [1, 2, 3]  # 创建列表对象，a 指向它
b = a           # b 也指向同一个对象

print(id(a))   # 140xxx
print(id(b))   # 140xxx（相同！）

# 修改通过任一变量都影响对象
b.append(4)
print(a)  # [1, 2, 3, 4]（a 也被改了）
```

## 二、可变对象 vs 不可变对象

### 2.1 分类概览

| 类型 | 可变/不可变 | 示例 |
|------|------------|------|
| int | 不可变 | `42` |
| float | 不可变 | `3.14` |
| str | 不可变 | `"hello"` |
| tuple | 不可变 | `(1, 2, 3)` |
| list | 可变 | `[1, 2, 3]` |
| dict | 可变 | `{"a": 1}` |
| set | 可变 | `{1, 2, 3}` |

### 2.2 不可变对象特性

```python
# 不可变对象无法直接修改
s = "hello"
# s[0] = "H"  # ❌ TypeError

# 实际上是通过创建新对象来实现"修改"
s = "Hello"  # 创建新字符串，旧对象被回收
print(s)  # Hello

# 字符串拼接也会创建新对象
text = ""
for i in range(1000):
    text += str(i)  # 每次都创建新对象，效率低！

# 正确做法：使用列表
parts = [str(i) for i in range(1000)]
text = "".join(parts)  # 效率高得多
```

### 2.3 可变对象特性

```python
# 列表可以直接修改
lst = [1, 2, 3]
lst[0] = 10          # ✅ 直接修改
lst.append(4)         # ✅ 添加元素
lst.extend([5, 6])   # ✅ 扩展列表

print(lst)  # [10, 2, 3, 4, 5, 6]

# 字典可以直接修改
d = {"a": 1}
d["b"] = 2           # 添加键值对
d["a"] = 10          # 修改值
print(d)  # {'a': 10, 'b': 2}

# 集合可以直接修改
s = {1, 2, 3}
s.add(4)             # 添加
s.remove(2)          # 删除
print(s)  # {1, 3, 4}
```

### 2.4 不可变对象的"陷阱"

```python
# 列表中存储的是引用
a = [1, 2, 3]
b = [4, 5, 6]
c = [a, b]  # c 存储的是 a 和 b 的引用

# 修改 a 会影响 c
a.append(100)
print(c)  # [[1, 2, 3, 100], [4, 5, 6]]

# 元组虽然是不可变的，但可以存储可变对象
t = ([1, 2], [3, 4])
# t[0] = [5, 6]  # ❌ TypeError：元组不可变
t[0].append(5)     # ✅ 可以修改列表
print(t)  # ([1, 2, 5], [3, 4])
```

## 三、引用传递

### 3.1 函数参数的引用传递

```python
def modify_list(lst):
    lst.append(100)  # 修改原对象

original = [1, 2, 3]
modify_list(original)
print(original)  # [1, 2, 3, 100]（被修改了！）

# 对比：创建新列表不会影响原对象
def replace_list(lst):
    lst = [4, 5, 6]  # 这只是让 lst 指向新对象，不影响 original

original = [1, 2, 3]
replace_list(original)
print(original)  # [1, 2, 3]（未被修改）
```

### 3.2 常见误区

```python
# 误区：以为参数赋值会修改原对象
def wrong_append(lst, item):
    lst = lst + [item]  # 创建新列表！

original = [1, 2, 3]
wrong_append(original, 4)
print(original)  # [1, 2, 3]（未被修改！）

# 正确：使用 extend 或 append
def correct_append(lst, item):
    lst.append(item)  # 直接修改原对象

original = [1, 2, 3]
correct_append(original, 4)
print(original)  # [1, 2, 3, 4]
```

## 四、浅拷贝（Shallow Copy）

### 4.1 拷贝方法

```python
# 方法一：切片拷贝
original = [1, 2, 3]
copy = original[:]

# 方法二：list() 拷贝
copy = list(original)

# 方法三：copy() 方法
copy = original.copy()

# 方法四：copy 模块
import copy
copy = copy.copy(original)
```

### 4.2 浅拷贝的特点

```python
# 浅拷贝：只拷贝第一层
original = [1, 2, 3]
copy = original.copy()

print(original is copy)  # False（不同对象）
print(original == copy)  # True（值相同）

# 修改第一层不影响对方
copy.append(4)
print(original)  # [1, 2, 3]
print(copy)      # [1, 2, 3, 4]
```

### 4.3 浅拷贝的"陷阱"

```python
# 浅拷贝只复制一层，内部对象仍然是共享引用！
original = [[1, 2], [3, 4]]
copy = original.copy()

print(original[0] is copy[0])  # True（内部列表是同一个对象！）

# 修改内部列表会影响原对象
copy[0].append(100)
print(original)  # [[1, 2, 100], [3, 4]]（原对象也被改了！）
print(copy)     # [[1, 2, 100], [3, 4]]
```

### 4.4 嵌套结构的浅拷贝问题

```python
# 常见场景：配置对象
config1 = {
    "name": "app1",
    "settings": {
        "debug": True,
        "port": 8000
    }
}

# 错误做法：直接复制
# config2 = config1  # 这是赋值，不是拷贝！

# 浅拷贝
import copy
config2 = copy.copy(config1)

# 修改外层不影响
config2["name"] = "app2"
print(config1["name"])  # "app1"（不受影响）

# 修改嵌套层会影响
config2["settings"]["port"] = 9000
print(config1["settings"]["port"])  # 9000！（被改了！）
```

## 五、深拷贝（Deep Copy）

### 5.1 深拷贝方法

```python
import copy

original = [[1, 2], [3, 4]]

# 深拷贝：递归复制所有层级
deep = copy.deepcopy(original)

print(original[0] is deep[0])  # False（内部列表也是新对象）
```

### 5.2 深拷贝 vs 浅拷贝对比

```python
import copy

data = [1, 2, [3, 4], {"a": 5}]

# 原始对象
print("原始对象:", data)

# 浅拷贝
shallow = copy.copy(data)
print("浅拷贝:", shallow)
print("浅拷贝第一层相同:", data[0] is shallow[0])  # True
print("浅拷贝嵌套列表相同:", data[2] is shallow[2])  # True（危险！）
print("浅拷贝字典相同:", data[3] is shallow[3])  # True（危险！）

# 深拷贝
deep = copy.deepcopy(data)
print("深拷贝:", deep)
print("深拷贝第一层相同:", data[0] is deep[0])  # True（不可变对象）
print("深拷贝嵌套列表相同:", data[2] is deep[2])  # False（安全！）
print("深拷贝字典相同:", data[3] is deep[3])  # False（安全！）

# 修改验证
shallow[2].append(999)
print("浅拷贝修改后原始:", data)  # data[2] 也被改了！

deep2 = copy.deepcopy(data)
deep2[2].append(999)
print("深拷贝修改后原始:", data)  # data 不受影响
```

### 5.3 深拷贝的注意事项

```python
import copy

# 注意一：循环引用会导致问题
circular = [1, 2]
circular.append(circular)  # 循环引用

# deep copy 能处理循环引用，但要注意
deep = copy.deepcopy(circular)
print(deep)
print(deep[2] is deep)  # True

# 注意二：自定义对象需要正确实现 __copy__ 和 __deepcopy__
class MyClass:
    def __init__(self, data):
        self.data = data
    
    def __copy__(self):
        """自定义浅拷贝"""
        return MyClass(self.data)  # data 本身如果是可变对象也会有问题
    
    def __deepcopy__(self, memo):
        """自定义深拷贝"""
        return MyClass(copy.deepcopy(self.data, memo))

# 注意三：共享单例对象
class Singleton:
    _instance = None
    
    def __new__(cls):
        if cls._instance is None:
            cls._instance = super().__new__(cls)
        cls._instance.value = 0
        return cls._instance

# deep copy 不会复制单例（取决于实现）
```

## 六、实战应用

### 6.1 函数参数保护

```python
import copy

def process_data(data, deep_copy=True):
    """
    处理数据
    
    Args:
        data: 输入数据
        deep_copy: 是否深拷贝，True 表示不修改原数据
    """
    working_data = copy.deepcopy(data) if deep_copy else data
    
    # 执行各种修改操作
    if isinstance(working_data, list):
        working_data.append("processed")
    elif isinstance(working_data, dict):
        working_data["status"] = "processed"
    
    return working_data

# 使用
original = {"items": [1, 2, 3], "count": 3}
result = process_data(original, deep_copy=True)
print("原始数据:", original)   # 不受影响
print("处理结果:", result)     # 已处理
```

### 6.2 配置管理

```python
import copy

class Config:
    """配置管理器"""
    
    def __init__(self):
        self._config = {}
    
    def set(self, key, value):
        """设置配置"""
        self._config[key] = value
    
    def get(self, key, default=None):
        """获取配置"""
        return self._config.get(key, default)
    
    def create_snapshot(self):
        """创建配置快照（深拷贝）"""
        return copy.deepcopy(self._config)
    
    def restore_snapshot(self, snapshot):
        """恢复配置快照"""
        self._config = copy.deepcopy(snapshot)
    
    def merge_update(self, updates):
        """
        合并更新配置
        """
        new_config = copy.deepcopy(self._config)
        new_config.update(updates)
        return new_config

# 使用示例
config = Config()
config.set("database", {"host": "localhost", "port": 5432})
config.set("cache", {"enabled": True, "ttl": 3600})

# 创建快照
snapshot = config.create_snapshot()

# 修改配置
config.set("database", {"host": "production", "port": 5433})

# 恢复快照
config.restore_snapshot(snapshot)
print(config.get("database"))  # 恢复到 localhost
```

### 6.3 撤销/重做功能

```python
import copy

class UndoManager:
    """撤销/重做管理器"""
    
    def __init__(self):
        self.history = []
        self.current = 0
    
    def save_state(self, state):
        """保存状态"""
        # 清除当前位置之后的历史
        self.history = self.history[:self.current]
        # 深拷贝保存新状态
        self.history.append(copy.deepcopy(state))
        self.current = len(self.history)
    
    def undo(self):
        """撤销"""
        if self.current > 1:
            self.current -= 1
            return copy.deepcopy(self.history[self.current - 1])
        return None
    
    def redo(self):
        """重做"""
        if self.current < len(self.history):
            self.current += 1
            return copy.deepcopy(self.history[self.current - 1])
        return None

# 使用示例
manager = UndoManager()
document = {"content": "Hello", "cursor": 5}

manager.save_state(document)

document["content"] = "Hello World"
manager.save_state(document)

document["content"] = "Hello World!"
manager.save_state(document)

print(manager.undo())  # 恢复到 "Hello World"
print(manager.undo())  # 恢复到 "Hello"
print(manager.redo())  # 恢复到 "Hello World"
```

## 七、is vs == 的区别

```python
# is：检查身份（id）是否相同
# ==：检查值是否相等

a = [1, 2, 3]
b = [1, 2, 3]
c = a

print(a is b)  # False（不同对象）
print(a is c)  # True（同一个对象）

print(a == b)  # True（值相同）
print(a == c)  # True

# 特殊情况：Python 内部缓存了小整数和短字符串
x = 256
y = 256
print(x is y)  # True（缓存范围内）

x = 257
y = 257
print(x is y)  # False（超出缓存范围）

# 小字符串缓存
s1 = "hello"
s2 = "hello"
print(s1 is s2)  # True（字符串字面量）
```

## 八、总结对比表

| 操作 | 拷贝层级 | 适用场景 |
|------|----------|----------|
| 赋值 `b = a` | 无（共享引用） | 临时使用同一对象 |

| 深拷贝 `b = copy.deepcopy(a)` | 全部层级 | 嵌套可变对象 |

## 十、易错点与避坑指南

1. **混淆赋值和拷贝**：`b = a` 不是拷贝，是共享引用
2. **浅拷贝陷阱**：嵌套的可变对象仍然共享引用
3. **元组误解**：元组不可变，但内部可变对象可变
4. **循环引用**：深拷贝能处理，但要注意性能
5. **性能问题**：深拷贝递归复制，大数据结构注意性能
