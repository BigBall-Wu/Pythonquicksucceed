## 一、multiprocessing 基础

### 1.1 为什么需要多进程

```python
# GIL（全局解释器锁）限制了多线程的并行执行
# CPU 密集型任务需要使用多进程来利用多核 CPU

# 单进程：只能使用一个 CPU 核心
def cpu_task(n):
    return sum(i * i for i in range(n))

# 多进程：可以使用多个 CPU 核心
from multiprocessing import Process

def parallel_task(n, result_queue):
    result = sum(i * i for i in range(n))
    result_queue.put(result)

if __name__ == "__main__":
    # 在 Windows 上必须使用 if __name__ == "__main__"
    process = Process(target=cpu_task, args=(1000000,))
    process.start()
    process.join()
```

### 1.2 Process 基本用法

```python
from multiprocessing import Process
import os

def print_process_info(name):
    """子进程执行的函数"""
    print(f"子进程 {name}: PID = {os.getpid()}")
    print(f"父进程 PID = {os.getppid()}")

if __name__ == "__main__":
    print(f"主进程 PID = {os.getpid()}")
    
    # 创建进程
    processes = []
    for i in range(3):
        p = Process(target=print_process_info, args=(f"Worker-{i}",))
        processes.append(p)
        p.start()  # 启动进程
    
    # 等待所有子进程结束
    for p in processes:
        p.join()
    
    print("所有子进程已结束")
```

### 1.3 进程生命周期

```python
from multiprocessing import Process
import time

def long_running_task(name):
    print(f"{name} 开始")
    time.sleep(2)
    print(f"{name} 结束")

if __name__ == "__main__":
    p = Process(target=long_running_task, args=("Task",))
    
    print(f"进程状态 - 创建前：{p.is_alive()}")
    
    p.start()
    print(f"进程状态 - 启动后：{p.is_alive()}")
    
    p.join()  # 等待进程结束
    print(f"进程状态 - 结束后：{p.is_alive()}")
    print(f"退出码：{p.exitcode}")
```

## 二、进程间通信

### 2.1 Queue 队列

```python
from multiprocessing import Process, Queue

def producer(queue, items):
    """生产者：向队列添加数据"""
    for item in items:
        queue.put(item)
        print(f"生产: {item}")
    queue.put(None)  # 发送结束信号

def consumer(queue):
    """消费者：从队列获取数据"""
    while True:
        item = queue.get()
        if item is None:  # 收到结束信号
            break
        print(f"消费: {item}")

if __name__ == "__main__":
    queue = Queue()
    
    # 启动生产者和消费者
    p1 = Process(target=producer, args=(queue, range(5)))
    p2 = Process(target=consumer, args=(queue,))
    
    p1.start()
    p2.start()
    
    p1.join()
    p2.terminate()
    
    print("处理完成")
```

### 2.2 Pipe 管道

```python
from multiprocessing import Process, Pipe

def sender(conn, messages):
    """发送消息"""
    for msg in messages:
        conn.send(msg)
        print(f"发送: {msg}")
    conn.close()

def receiver(conn):
    """接收消息"""
    while True:
        try:
            msg = conn.recv()
            print(f"接收: {msg}")
        except EOFError:
            break

if __name__ == "__main__":
    parent_conn, child_conn = Pipe()  # 创建双向管道
    
    messages = ["Hello", "World", "123"]
    
    p = Process(target=sender, args=(child_conn, messages))
    p.start()
    
    receiver(parent_conn)
    
    p.join()
```

### 2.3 共享内存

```python
from multiprocessing import Process, Value, Array, Lock

def increment(shared_value, lock):
    """增加共享值"""
    for _ in range(1000000):
        with lock:  # 使用锁保护共享变量
            shared_value.value += 1

if __name__ == "__main__":
    # 创建共享内存
    counter = Value('i', 0)  # 'i' 表示有符号整数
    lock = Lock()
    
    # 创建多个进程
    processes = [
        Process(target=increment, args=(counter, lock))
        for _ in range(4)
    ]
    
    for p in processes:
        p.start()
    
    for p in processes:
        p.join()
    
    print(f"最终计数：{counter.value}")  # 应该是 4000000
```

### 2.4 Manager 托管对象

```python
from multiprocessing import Process, Manager

def worker(shared_dict, shared_list, lock):
    """使用托管的共享对象"""
    with lock:
        shared_dict["count"] = shared_dict.get("count", 0) + 1
        shared_list.append(os.getpid())

import os

if __name__ == "__main__":
    manager = Manager()
    
    shared_dict = manager.dict()
    shared_list = manager.list()
    lock = manager.Lock()
    
    processes = [
        Process(target=worker, args=(shared_dict, shared_list, lock))
        for _ in range(5)
    ]
    
    for p in processes:
        p.start()
    
    for p in processes:
        p.join()
    
    print(f"共享字典：{dict(shared_dict)}")
    print(f"共享列表：{list(shared_list)}")
```

## 三、进程池 Pool

### 3.1 Pool 基本用法

```python
from multiprocessing import Pool

def square(n):
    """计算平方"""
    return n * n

if __name__ == "__main__":
    with Pool(processes=4) as pool:  # 创建 4 个进程的进程池
        # map：返回结果列表
        results = pool.map(square, range(10))
        print(f"map 结果：{results}")
        
        # apply：同步执行单个任务
        result = pool.apply(square, (5,))
        print(f"apply 结果：{result}")
        
        # apply_async：异步执行
        async_result = pool.apply_async(square, (7,))
        print(f"等待结果：{async_result.get()}")
```

### 3.2 map vs starmap

```python
from multiprocessing import Pool

def add(a, b):
    return a + b

if __name__ == "__main__":
    with Pool(4) as pool:
        # map：处理单参数列表
        result1 = pool.map(square := lambda x: x**2, range(5))
        print(f"map 结果：{result1}")
        
        # starmap：处理元组参数
        pairs = [(1, 2), (3, 4), (5, 6)]
        result2 = pool.starmap(add, pairs)
        print(f"starmap 结果：{result2}")
        
        # imap：惰性迭代器
        print("imap 结果：")
        for result in pool.imap(square := lambda x: x**2, range(5)):
            print(f"  {result}")
```

### 3.3 回调函数

```python
from multiprocessing import Pool

def heavy_task(n):
    """耗时任务"""
    import time
    time.sleep(0.5)
    return n * n

def callback(result):
    """回调函数：任务完成后自动调用"""
    print(f"任务完成，结果：{result}")

if __name__ == "__main__":
    with Pool(4) as pool:
        # apply_async 带回调
        for i in range(5):
            pool.apply_async(heavy_task, (i,), callback=callback)
        
        pool.close()
        pool.join()
        print("所有任务完成")
```

### 3.4 错误处理

```python
from multiprocessing import Pool

def task_with_error(n):
    if n == 3:
        raise ValueError("n 不能是 3")
    return n * n

if __name__ == "__main__":
    with Pool(4) as pool:
        # 使用 map：会抛出第一个异常
        try:
            results = pool.map(task_with_error, range(5))
        except ValueError as e:
            print(f"map 捕获异常：{e}")
        
        # 使用 imap_unordered：处理所有结果
        for i, result in enumerate(pool.imap_unordered(task_with_error, range(5))):
            print(f"结果 {i}: {result}")
```

## 四、实战应用

### 4.1 并行计算

```python
from multiprocessing import Pool, cpu_count
import math

def calculate_partial_integral(args):
    """计算定积分的一部分"""
    start, end, steps = args
    step_size = (end - start) / steps
    total = 0
    x = start
    for _ in range(steps):
        total += math.sin(x) * step_size
        x += step_size
    return total

def parallel_integrate(func, start, end, total_steps, num_processes=None):
    """并行计算定积分"""
    if num_processes is None:
        num_processes = cpu_count()
    
    steps_per_process = total_steps // num_processes
    ranges = [
        (start + i * steps_per_process * (end - start) / total_steps,
         start + (i + 1) * steps_per_process * (end - start) / total_steps,
         steps_per_process)
        for i in range(num_processes)
    ]
    
    with Pool(num_processes) as pool:
        results = pool.map(calculate_partial_integral, ranges)
    
    return sum(results)

# 使用
import time
result = parallel_integrate(math.sin, 0, math.pi, 1000000)
print(f"积分结果：{result:.6f}")
print(f"实际 sin(x) 积分：{-math.cos(math.pi) + math.cos(0):.6f}")
```

### 4.2 批量文件处理

```python
from multiprocessing import Pool
from pathlib import Path
import hashlib

def process_file(filepath):
    """处理单个文件"""
    try:
        with open(filepath, 'rb') as f:
            content = f.read()
        return {
            "path": str(filepath),
            "size": len(content),
            "md5": hashlib.md5(content).hexdigest()
        }
    except Exception as e:
        return {"path": str(filepath), "error": str(e)}

def batch_process_files(directory, extension="*.py", max_workers=4):
    """批量处理目录中的文件"""
    dir_path = Path(directory)
    files = list(dir_path.rglob(extension))
    
    with Pool(max_workers) as pool:
        results = pool.map(process_file, files)
    
    return results

if __name__ == "__main__":
    results = batch_process_files(".", "*.py")
    for r in results[:5]:
        if "error" not in r:
            print(f"{r['path']}: {r['size']} bytes, MD5={r['md5'][:8]}")
```

### 4.3 并行爬虫

```python
from multiprocessing import Pool
import requests
from urllib.parse import urljoin

def fetch_page(args):
    """获取单个页面"""
    base_url, url = args
    full_url = urljoin(base_url, url)
    try:
        response = requests.get(full_url, timeout=5)
        return {
            "url": full_url,
            "status": response.status_code,
            "content_length": len(response.content)
        }
    except Exception as e:
        return {"url": full_url, "error": str(e)}

def parallel_fetch(base_url, urls, max_workers=10):
    """并行抓取多个页面"""
    args = [(base_url, url) for url in urls]
    
    with Pool(max_workers) as pool:
        results = pool.map(fetch_page, args)
    
    return results

if __name__ == "__main__":
    urls = [
        "https://example.com/page1",
        "https://example.com/page2",
        "https://example.com/page3"
    ]
    
    results = parallel_fetch("https://example.com", urls)
    for r in results:
        if "error" not in r:
            print(f"{r['url']}: {r['status']}, {r['content_length']} bytes")
```

## 五、进程同步

### 5.1 Lock 互斥锁

```python
from multiprocessing import Process, Lock

def task_with_lock(lock, shared_counter):
    """使用锁的任务"""
    for _ in range(100):
        with lock:
            shared_counter.value += 1

if __name__ == "__main__":
    counter = Value('i', 0)
    lock = Lock()
    
    processes = [
        Process(target=task_with_lock, args=(lock, counter))
        for _ in range(4)
    ]
    
    for p in processes:
        p.start()
    for p in processes:
        p.join()
    
    print(f"最终计数：{counter.value}")  # 应该是 400
```

### 5.2 Event 事件

```python
from multiprocessing import Process, Event
import time

def waiter(event, name):
    """等待事件"""
    print(f"{name} 等待中...")
    event.wait()  # 阻塞直到事件被设置
    print(f"{name} 收到信号，继续执行")

def setter(event):
    """设置事件"""
    time.sleep(2)
    print("设置事件")
    event.set()

if __name__ == "__main__":
    event = Event()
    
    p1 = Process(target=waiter, args=(event, "Worker-1"))
    p2 = Process(target=waiter, args=(event, "Worker-2"))
    p3 = Process(target=setter, args=(event,))
    
    p1.start()
    p2.start()
    p3.start()
    
    p1.join()
    p2.join()
    p3.join()
```

### 5.3 Condition 条件

```python
from multiprocessing import Process, Condition, Value

def producer(condition, buffer, max_size):
    """生产者"""
    for i in range(20):
        with condition:
            while len(buffer) >= max_size:
                condition.wait()
            buffer.append(i)
            print(f"生产: {i}")
            condition.notify()

def consumer(condition, buffer):
    """消费者"""
    while True:
        with condition:
            while not buffer:
                condition.wait()
            item = buffer.pop(0)
            print(f"消费: {item}")
            condition.notify()
            if item == 19:  # 结束条件
                break

if __name__ == "__main__":
    condition = Condition()
    buffer = []
    
    p1 = Process(target=producer, args=(condition, buffer, 5))
    p2 = Process(target=consumer, args=(condition, buffer))
    
    p1.start()
    p2.start()
    
    p1.join()
    p2.join()
```

## 六、注意事项

### 6.1 Windows 特殊处理

```python
# Windows 上必须使用 if __name__ == "__main__"
if __name__ == "__main__":
    # 所有 multiprocessing 代码必须在这里
    from multiprocessing import Pool
    
    with Pool(4) as pool:
        print(pool.map(lambda x: x**2, range(10)))
```

### 6.2 避免共享状态

```python
# ❌ 不好：共享状态导致竞争
counter = 0

def bad_task():
    global counter
    for _ in range(100):
        counter += 1  # 竞争条件！

# ✅ 更好：使用进程安全的共享内存
from multiprocessing import Value, Lock

counter = Value('i', 0)
lock = Lock()

def good_task():
    global counter, lock
    for _ in range(100):
        with lock:
            counter.value += 1
```

### 6.3 进程数选择

```python
from multiprocessing import cpu_count

# CPU 密集型：进程数 = CPU 核心数
cpu_count = cpu_count()
print(f"CPU 核心数：{cpu_count}")

# I/O 密集型：进程数可以更多（2-4 倍 CPU 核心数）
# 因为 I/O 时进程会阻塞
io_processes = cpu_count() * 2
```

## 八、易错点与避坑指南

1. **Windows 平台**：必须使用 `if __name__ == "__main__"`
2. **GIL 误解**：多进程用于 CPU 密集型，多线程用于 I/O 密集型
3. **进程开销**：进程创建开销大，避免频繁创建销毁
4. **数据序列化**：进程间传递数据需要 pickle 序列化
5. **共享状态**：避免共享状态，使用消息传递
