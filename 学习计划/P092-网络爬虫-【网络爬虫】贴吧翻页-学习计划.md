```python
# 贴吧 URL 规律：
# https://tieba.baidu.com/f?kw={贴吧名称}&pn={偏移量}

# 分页公式：
# 第 1 页：pn=0
# 第 2 页：pn=50
# 第 3 页：pn=100
# ...
# 第 n 页：pn=(n-1)*50

# 示例：爬取 "python" 贴吧前 10 页
tieba_name = "python"
base_url = "https://tieba.baidu.com/f"

for page in range(1, 11):
    pn = (page - 1) * 50
    url = f"{base_url}?kw={tieba_name}&pn={pn}"
    print(f"Page {page}: {url}")
```

### 第 2 步：解析帖子列表（15 分钟）

```python
import requests
from bs4 import BeautifulSoup

def parse_tieba_page(html):
    """解析贴吧帖子列表"""
    soup = BeautifulSoup(html, "lxml")
    threads = []

    # 帖子列表选择器
    for item in soup.select(".threadlist_title"):
        title = item.get_text().strip()
        link = item.get("href", "")

        if title and link:
            threads.append({
                "title": title,
                "link": f"https://tieba.baidu.com{link}" if not link.startswith("http") else link
            })

    return threads

# 测试
url = "https://tieba.baidu.com/f?kw=python&pn=0"
headers = {"User-Agent": "Mozilla/5.0 ..."}
r = requests.get(url, headers=headers)
threads = parse_tieba_page(r.text)
print(f"找到 {len(threads)} 个帖子")
```

### 第 3 步：完整翻页爬虫（15 分钟）

```python
import requests
from bs4 import BeautifulSoup
import time
import json

def scrape_tieba(tieba_name, start_page=1, end_page=10, delay=2):
    """翻页爬取贴吧"""
    all_threads = []
    base_url = f"https://tieba.baidu.com/f?kw={tieba_name}"

    headers = {
        "User-Agent": "Mozilla/5.0 (Windows NT 10.0; Win64; x64) ...",
        "Referer": "https://tieba.baidu.com/",
    }

    for page in range(start_page, end_page + 1):
        pn = (page - 1) * 50
        url = f"{base_url}&pn={pn}"

        print(f"爬取第 {page} 页...")
        try:
            response = requests.get(url, headers=headers, timeout=10)
            response.encoding = "utf-8"

            soup = BeautifulSoup(response.text, "lxml")

            # 解析帖子
            for item in soup.select(".threadlist_title"):
                title = item.get_text().strip()
                link = item.get("href", "")

                if title and link:
                    full_link = f"https://tieba.baidu.com{link}"
                    all_threads.append({
                        "page": page,
                        "title": title,
                        "link": full_link
                    })

            print(f"  第 {page} 页: {len(all_threads)} 条")

        except Exception as e:
            print(f"  第 {page} 页失败: {e}")

        # 控制请求频率
        time.sleep(delay)

    return all_threads

# 运行
threads = scrape_tieba("python", 1, 5)
print(f"\n总计: {len(threads)} 条帖子")

# 保存结果
with open("tieba_python.json", "w", encoding="utf-8") as f:
    json.dump(threads, f, ensure_ascii=False, indent=2)
```

### 第 4 步：防封策略（10 分钟）

```python
# 防封策略：

防封措施 = """
1. 控制请求频率
   - 每页间隔 2-5 秒
   - 随机间隔更安全

2. 随机 User-Agent
   - 使用 UA 池
   - 定期更新 UA

3. 设置超时
   - timeout=10 秒
   - 自动重试

4. 异常处理
   - 网络错误跳过
   - 重定向处理

5. 代理 IP（高级）
   - 使用代理池
   - 自动切换
"""

# 进阶防封代码
import random
import time
from fake_useragent import UserAgent

class TiebaSpider:
    """贴吧爬虫（防封版）"""

    def __init__(self):
        self.ua = UserAgent()
        self.session = requests.Session()

    def get_headers(self):
        return {
            "User-Agent": self.ua.random,
            "Referer": "https://tieba.baidu.com/",
        }

    def fetch(self, url, retry=3):
        for i in range(retry):
            try:
                response = self.session.get(
                    url,
                    headers=self.get_headers(),
                    timeout=10
                )
                response.raise_for_status()
                return response
            except Exception as e:
                print(f"请求失败 (尝试 {i+1}/{retry}): {e}")
                time.sleep(5)
        return None

    def delay(self):
        """随机延迟"""
        time.sleep(random.uniform(2, 5))
```

### 第 5 步：数据存储（10 分钟）

```python
import json
import csv
import sqlite3
from pathlib import Path

def save_to_json(threads, filepath):
    """保存为 JSON"""
    with open(filepath, "w", encoding="utf-8") as f:
        json.dump(threads, f, ensure_ascii=False, indent=2)
    print(f"保存到 JSON: {filepath}")

def save_to_csv(threads, filepath):
    """保存为 CSV"""
    with open(filepath, "w", encoding="utf-8", newline="") as f:
        writer = csv.DictWriter(f, fieldnames=["page", "title", "link"])
        writer.writeheader()
        writer.writerows(threads)
    print(f"保存到 CSV: {filepath}")

def save_to_sqlite(threads, db_path):
    """保存到 SQLite"""
    conn = sqlite3.connect(db_path)
    cursor = conn.cursor()

    cursor.execute("""
        CREATE TABLE IF NOT EXISTS threads (
            id INTEGER PRIMARY KEY AUTOINCREMENT,
            page INTEGER,
            title TEXT,
            link TEXT
        )
    """)

    for thread in threads:
        cursor.execute(
            "INSERT INTO threads (page, title, link) VALUES (?, ?, ?)",
            (thread["page"], thread["title"], thread["link"])
        )

    conn.commit()
    conn.close()
    print(f"保存到 SQLite: {db_path}")
```
