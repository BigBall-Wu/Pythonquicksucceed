## 二、日志系统

### 2.1 日志配置

```python
# src/utils/logger.py
import logging
import os
from pathlib import Path
from logging.handlers import RotatingFileHandler
from datetime import datetime

class Logger:
    """日志管理器"""
    
    _loggers = {}
    
    @classmethod
    def get_logger(cls, name, log_file=None, level=logging.INFO):
        if name in cls._loggers:
            return cls._loggers[name]
        
        logger = logging.getLogger(name)
        logger.setLevel(level)
        
        # 格式化
        formatter = logging.Formatter(
            "%(asctime)s - %(name)s - %(levelname)s - %(message)s"
        )
        
        # 控制台输出
        console_handler = logging.StreamHandler()
        console_handler.setFormatter(formatter)
        logger.addHandler(console_handler)
        
        # 文件输出
        if log_file:
            log_dir = Path("logs")
            log_dir.mkdir(exist_ok=True)
            
            file_handler = RotatingFileHandler(
                log_dir / log_file,
                maxBytes=10 * 1024 * 1024,  # 10MB
                backupCount=5
            )
            file_handler.setFormatter(formatter)
            logger.addHandler(file_handler)
        
        cls._loggers[name] = logger
        return logger

# 使用
logger = Logger.get_logger("crawler", "crawler.log")
logger.info("爬虫启动")
logger.warning("警告信息")
logger.error("错误信息", exc_info=True)
```

### 2.2 爬虫日志装饰器

```python
import functools
import time
from logger import Logger

def log_spider_call(func):
    """爬虫方法日志装饰器"""
    logger = Logger.get_logger("spider")
    
    @functools.wraps(func)
    def wrapper(self, *args, **kwargs):
        spider_name = self.__class__.__name__
        logger.info(f"[{spider_name}] 开始执行 {func.__name__}")
        
        start_time = time.time()
        try:
            result = func(self, *args, **kwargs)
            elapsed = time.time() - start_time
            logger.info(f"[{spider_name}] {func.__name__} 完成，耗时 {elapsed:.2f}s")
            return result
        except Exception as e:
            elapsed = time.time() - start_time
            logger.error(f"[{spider_name}] {func.__name__} 失败，耗时 {elapsed:.2f}s: {e}")
            raise
    
    return wrapper
```

```python
# src/utils/error_handler.py
import functools
import asyncio
from logger import Logger

logger = Logger.get_logger("error_handler")

def handle_spider_errors(func):
    """爬虫方法统一错误处理"""
    @functools.wraps(func)
    def wrapper(*args, **kwargs):
        try:
            return func(*args, **kwargs)
        except NetworkError as e:
            logger.error(f"网络错误: {e}")
            return None
        except ParseError as e:
            logger.error(f"解析错误: {e}")
            return None
        except RateLimitError as e:
            logger.warning(f"频率限制: {e}")
            return None
        except Exception as e:
            logger.exception(f"未知错误: {e}")
            return None
    
    return wrapper

def async_handle_spider_errors(func):
    """异步爬虫方法统一错误处理"""
    @functools.wraps(func)
    async def wrapper(*args, **kwargs):
        try:
            return await func(*args, **kwargs)
        except Exception as e:
            logger.exception(f"异步任务错误: {e}")
            return None
    
    return wrapper
```

## 四、基类设计

### 4.1 爬虫基类

```python
# src/spiders/base.py
import asyncio
import aiohttp
from abc import ABC, abstractmethod
from typing import List, Dict, Any, Optional
from dataclasses import dataclass, field
from datetime import datetime

from src.utils.logger import Logger
from src.utils.error_handler import async_handle_spider_errors
from src.exceptions import NetworkError, ParseError

@dataclass
class SpiderConfig:
    """爬虫配置"""
    name: str
    start_urls: List[str]
    concurrency: int = 10
    delay: float = 1.0
    timeout: int = 30
    max_retries: int = 3
    retry_delay: float = 5.0
    headers: Dict[str, str] = field(default_factory=dict)

class BaseSpider(ABC):
    """爬虫基类"""
    
    def __init__(self, config: SpiderConfig):
        self.config = config
        self.logger = Logger.get_logger(config.name)
        self.session: Optional[aiohttp.ClientSession] = None
        self.results: List[Dict] = []
        self.errors: List[Dict] = []
    
    async def init_session(self):
        """初始化 Session"""
        timeout = aiohttp.ClientTimeout(total=self.config.timeout)
        connector = aiohttp.TCPConnector(limit=self.config.concurrency)
        
        self.session = aiohttp.ClientSession(
            timeout=timeout,
            connector=connector,
            headers=self.config.headers
        )
    
    async def close_session(self):
        """关闭 Session"""
        if self.session:
            await self.session.close()
    
    @abstractmethod
    async def parse(self, response: aiohttp.ClientResponse, url: str) -> List[Dict]:
        """解析页面，返回数据列表"""
        pass
    
    async def fetch(self, url: str) -> Optional[str]:
        """抓取单个页面"""
        for attempt in range(self.config.max_retries):
            try:
                async with self.session.get(url) as response:
                    if response.status == 200:
                        return await response.text()
                    elif response.status == 429:
                        self.logger.warning(f"频率限制，等待 {self.config.retry_delay}s")
                        await asyncio.sleep(self.config.retry_delay)
                    else:
                        raise NetworkError(f"HTTP {response.status}: {url}")
            except Exception as e:
                if attempt < self.config.max_retries - 1:
                    await asyncio.sleep(self.config.retry_delay)
                    continue
                raise
    
    async def crawl(self):
        """爬取入口"""
        await self.init_session()
        
        try:
            semaphore = asyncio.Semaphore(self.config.concurrency)
            
            async def limited_crawl(url):
                async with semaphore:
                    try:
                        html = await self.fetch(url)
                        if html:
                            items = await self.parse(self.session, url)
                            self.results.extend(items)
                            self.logger.info(f"成功抓取 {url}，获取 {len(items)} 条数据")
                    except Exception as e:
                        self.errors.append({"url": url, "error": str(e)})
                        self.logger.error(f"抓取失败 {url}: {e}")
                    
                    await asyncio.sleep(self.config.delay)
            
            tasks = [limited_crawl(url) for url in self.config.start_urls]
            await asyncio.gather(*tasks)
            
        finally:
            await self.close_session()
        
        return {
            "success": len(self.results),
            "errors": len(self.errors),
            "data": self.results
        }
    
    @async_handle_spider_errors
    async def run(self):
        """运行爬虫"""
        self.logger.info(f"爬虫 {self.config.name} 启动")
        start_time = datetime.now()
        
        result = await self.crawl()
        
        elapsed = (datetime.now() - start_time).total_seconds()
        self.logger.info(f"爬虫完成，耗时 {elapsed:.2f}s，成功 {result['success']}，失败 {result['errors']}")
        
        return result
```

```python
# src/spiders/news.py
import asyncio
from bs4 import BeautifulSoup
from dataclasses import dataclass, field

from src.spiders.base import BaseSpider, SpiderConfig

@dataclass
class NewsItem:
    """新闻数据模型"""
    title: str = ""
    url: str = ""
    author: str = ""
    publish_date: str = ""
    content: str = ""
    source: str = ""
    tags: list = field(default_factory=list)

class NewsSpider(BaseSpider):
    """新闻爬虫"""
    
    def __init__(self):
        config = SpiderConfig(
            name="news_spider",
            start_urls=[
                "https://news.example.com/",
                "https://news.example.com/page/2",
            ],
            concurrency=10,
            delay=1.0,
            headers={
                "User-Agent": "Mozilla/5.0 (Windows NT 10.0; Win64; x64) AppleWebKit/537.36"
            }
        )
        super().__init__(config)
    
    async def parse(self, session, url: str) -> list:
        html = await self.fetch(url)
        soup = BeautifulSoup(html, "lxml")
        
        items = []
        for article in soup.select("div.article"):
            item = {
                "title": article.select_one("h2.title").text.strip() if article.select_one("h2.title") else "",
                "url": article.select_one("a.title").get("href", "") if article.select_one("a.title") else "",
                "author": article.select_one("span.author").text.strip() if article.select_one("span.author") else "",
                "publish_date": article.select_one("time").get("datetime", "") if article.select_one("time") else "",
                "content": article.select_one("div.content").text.strip() if article.select_one("div.content") else "",
                "source": "example.com",
            }
            items.append(item)
        
        return items
```

### 5.2 MongoDB Pipeline

```python
# src/pipelines/mongodb.py
from pymongo import MongoClient
from datetime import datetime

from src.pipelines.base import BasePipeline

class MongoPipeline(BasePipeline):
    """MongoDB 数据存储"""
    
    def __init__(self, uri: str, db_name: str, collection: str):
        super().__init__()
        self.uri = uri
        self.db_name = db_name
        self.collection_name = collection
        self.client = None
        self.db = None
    
    async def open_spider(self, spider):
        self.client = MongoClient(self.uri)
        self.db = self.client[self.db_name]
        self.collection = self.db[self.collection_name]
        spider.logger.info(f"MongoDB 连接成功: {self.db_name}")
    
    async def process_item(self, item: Dict, spider) -> Dict:
        item["crawled_at"] = datetime.now()
        
        self.collection.update_one(
            {"url": item.get("url")},
            {"$set": item},
            upsert=True
        )
        
        return item
    
    async def close_spider(self, spider):
        if self.client:
            self.client.close()
        spider.logger.info("MongoDB 连接关闭")
```

## 六、入口文件

```python
# run.py
import asyncio
import sys
from pathlib import Path

# 添加项目根目录到 Python 路径
sys.path.insert(0, str(Path(__file__).parent))

from src.spiders.news import NewsSpider
from src.pipelines.mongodb import MongoPipeline
from config.settings import MONGO_URI, MONGO_DB

async def main():
    spider = NewsSpider()
    
    # 添加 Pipeline
    pipeline = MongoPipeline(
        uri=MONGO_URI,
        db_name=MONGO_DB,
        collection="news"
    )
    
    # 启动 Pipeline
    await pipeline.open_spider(spider)
    
    try:
        # 运行爬虫
        result = await spider.run()
        
        # 处理数据
        for item in result["data"]:
            await pipeline.process_item(item, spider)
        
        print(f"爬取完成: 成功 {result['success']}, 失败 {result['errors']}")
    
    finally:
        await pipeline.close_spider(spider)

if __name__ == "__main__":
    asyncio.run(main())
```
