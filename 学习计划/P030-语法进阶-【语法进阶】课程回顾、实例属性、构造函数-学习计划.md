## 五、具体学习步骤

### 第 1 步：实例属性 vs 类属性（10 分钟）

```python
class Dog:
    species = "Canis familiaris"     # 类属性（所有实例共享）

    def __init__(self, name, age):
        self.name = name              # 实例属性（每个实例独立）
        self.age = age

# 创建实例
dog1 = Dog("小白", 3)
dog2 = Dog("小黑", 5)

print(dog1.species)        # Canis familiaris（来自类）
print(dog2.species)        # Canis familiaris
Dog.species = "Canis"      # 改类属性
print(dog1.species)        # Canis（所有实例都变）

# 实例属性
print(dog1.name)           # 小白
print(dog2.name)           # 小黑
dog1.name = "白小白"      # 改实例属性
print(dog1.name)           # 白小白（只影响 dog1）
print(dog2.name)           # 小黑（不变）
```

### 第 2 步：理解 self（10 分钟）

```python
class Cat:
    def __init__(self, name):
        self.name = name

    def meow(self):
        print(f"{self.name}: 喵！")

cat = Cat("小花")
cat.meow()                  # 等价于 Cat.meow(cat)

# self 就是当前实例
# 调用 cat.meow() 时 Python 自动把 cat 作为 self 传进去
```

### 第 3 步：构造函数重载（Python 不支持）（10 分钟）

```python
# Python 不能像 Java 那样有多个 __init__
# 替代方案：默认参数 + isinstance 判断

class Point:
    def __init__(self, x, y=0, z=0):
        self.x = x
        self.y = y
        self.z = z

# 多种构造方式
p1 = Point(1)                          # 1, 0, 0
p2 = Point(1, 2)                       # 1, 2, 0
p3 = Point(1, 2, 3)                    # 1, 2, 3

# 或者用类方法构造
class Point:
    def __init__(self, x, y):
        self.x = x
        self.y = y

    @classmethod
    def from_tuple(cls, t):
        """从元组构造"""
        return cls(t[0], t[1])

    @classmethod
    def from_dict(cls, d):
        """从字典构造"""
        return cls(d["x"], d["y"])

p1 = Point(1, 2)
p2 = Point.from_tuple((3, 4))
p3 = Point.from_dict({"x": 5, "y": 6})
```

### 第 4 步：默认值的陷阱（10 分钟）

```python
# ❌ 默认值用可变对象
class Team:
    def __init__(self, name, members=[]):
        self.name = name
        self.members = members

t1 = Team("A")
t1.members.append("Alice")

t2 = Team("B")
print(t2.members)          # ['Alice']（累积！）

# ✓ 用 None 修正
class Team:
    def __init__(self, name, members=None):
        self.name = name
        self.members = members if members is not None else []

t1 = Team("A")
t1.members.append("Alice")

t2 = Team("B")
print(t2.members)          # []（独立）
```

### 第 5 步：实战 - 用户类（10 分钟）

```python
class User:
    """用户类"""

    # 类属性
    total_users = 0

    def __init__(self, username, email, age=18):
        """构造函数"""
        User.total_users += 1
        self.username = username
        self.email = email
        self.age = age
        self.is_active = True
        self.posts = []            # 每个实例独立

    def deactivate(self):
        """停用账号"""
        self.is_active = False

    def add_post(self, post):
        """添加文章"""
        if self.is_active:
            self.posts.append(post)

    def __str__(self):
        return f"User({self.username}, {self.email})"

# 测试
u1 = User("alice", "alice@example.com", 25)
u2 = User("bob", "bob@example.com")

print(User.total_users)        # 2

u1.add_post("第一篇博客")
u1.add_post("第二篇博客")
print(f"{u1} 有 {len(u1.posts)} 篇文章")
```

### 第 6 步：常见错误（5 分钟）

```python
# 错误 1：忘记 self 参数
class Bad:
    def greet():                # ❌ 缺 self
        print("hi")

b = Bad()
# b.greet()                     # ❌ TypeError

# 正确
class Good:
    def greet(self):             # ✓
        print("hi")

# 错误 2：在 __init__ 之前访问属性
class Student:
    def study(self):
        print(self.name)         # ❌ AttributeError（没 __init__）

s = Student()
# s.study()                      # ❌
```
