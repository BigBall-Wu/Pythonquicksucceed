## 一、functools 模块深入

### 1.1 functools.wraps 详解

```python
import functools

# functools.wraps 的作用
def decorator(func):
    @functools.wraps(func)  # 这行必须放在 wrapper 上面
    def wrapper(*args, **kwargs):
        return func(*args, **kwargs)
    return wrapper

@decorator
def original_func(a, b):
    """这是原始函数的文档字符串"""
    return a + b

# 使用 wraps 后，wrapper 会继承原函数的元数据
print(original_func.__name__)           # original_func
print(original_func.__doc__)            # 这是原始函数的文档字符串
print(original_func.__module__)         # __main__
print(original_func.__qualname__)       # original_func
print(original_func.__annotations__)     # {}

# wraps 复制的属性
print(original_func.__wrapped__)         # <function original_func at ...>
```

### 1.2 functools.lru_cache

```python
import functools

# LRU 缓存装饰器（最近最少使用）
@functools.lru_cache(maxsize=128)
def fibonacci(n):
    """斐波那契数列（自动缓存）"""
    if n <= 1:
        return n
    return fibonacci(n - 1) + fibonacci(n - 2)

# 缓存信息
print(f"Fib(100) = {fibonacci(100)}")
print(f"缓存命中：{fibonacci.cache_info()}")

# 清除缓存
fibonacci.cache_clear()
print(f"缓存清除后：{fibonacci.cache_info()}")
```

### 1.3 functools.partial

```python
import functools

# partial：创建一个"冻结"了部分参数的新函数
def power(base, exponent):
    return base ** exponent

# 冻结 exponent 参数
square = functools.partial(power, exponent=2)
cube = functools.partial(power, exponent=3)

print(square(5))    # 25 (5^2)
print(cube(5))      # 125 (5^3)

# 冻结 base 参数
of_ten = functools.partial(power, 10)
print(of_ten(2))   # 100 (10^2)
print(of_ten(3))   # 1000 (10^3)
```

## 二、类装饰器高级用法

### 2.1 装饰器接收类

```python
import functools

def logged(cls):
    """日志记录类装饰器"""
    for name, method in cls.__dict__.items():
        if callable(method) and not name.startswith('_'):
            setattr(cls, name, _log_wrapper(method, name))
    return cls

def _log_wrapper(method, method_name):
    @functools.wraps(method)
    def wrapper(*args, **kwargs):
        print(f"调用 {method_name}")
        return method(*args, **kwargs)
    return wrapper

@logged
class Calculator:
    def add(self, a, b):
        return a + b
    
    def multiply(self, a, b):
        return a * b

calc = Calculator()
calc.add(1, 2)        # 调用 add
calc.multiply(3, 4)   # 调用 multiply
```

### 2.2 类作为装饰器

```python
class CountCalls:
    """计数器装饰器"""
    
    def __init__(self, func):
        functools.update_wrapper(self, func)
        self.func = func
        self.count = 0
    
    def __call__(self, *args, **kwargs):
        self.count += 1
        print(f"{self.func.__name__} 被调用 {self.count} 次")
        return self.func(*args, **kwargs)

@CountCalls
def greet():
    print("Hello!")

greet()  # greet 被调用 1 次\nHello!
greet()  # greet 被调用 2 次\nHello!
greet()  # greet 被调用 3 次\nHello!
```

### 2.3 带参数的类装饰器

```python
class Retry:
    """带参数的重试装饰器（类形式）"""
    
    def __init__(self, max_attempts=3, delay=1):
        self.max_attempts = max_attempts
        self.delay = delay
    
    def __call__(self, func):
        @functools.wraps(func)
        def wrapper(*args, **kwargs):
            last_exception = None
            for attempt in range(self.max_attempts):
                try:
                    return func(*args, **kwargs)
                except Exception as e:
                    last_exception = e
                    print(f"尝试 {attempt + 1}/{self.max_attempts} 失败")
                    if attempt < self.max_attempts - 1:
                        import time
                        time.sleep(self.delay)
            raise last_exception
        return wrapper

@Retry(max_attempts=5, delay=0.5)
def unreliable_operation():
    import random
    if random.random() < 0.8:
        raise ValueError("随机失败")
    return "操作成功"

print(unreliable_operation())
```

## 三、保留函数签名

### 3.1 inspect 模块

```python
import inspect

def my_decorator(func):
    @functools.wraps(func)
    def wrapper(*args, **kwargs):
        return func(*args, **kwargs)
    return wrapper

@my_decorator
def add(a: int, b: int) -> int:
    """两数相加"""
    return a + b

# 检查签名
sig = inspect.signature(add)
print(sig)                    # (a: int, b: int) -> int
print(sig.parameters)         # OrderedDict([('a', <int>), ('b', <int>)])

# 绑定参数
bound = sig.bind(1, 2)
print(bound.arguments)        # Arguments(a=1, b=2)
```

### 3.2 使用 wraps 保留签名

```python
import functools
import inspect

def simple_decorator(func):
    @functools.wraps(func)
    def wrapper(*args, **kwargs):
        return func(*args, **kwargs)
    return wrapper

@simple_decorator
def original(a: int, b: str = "default") -> bool:
    """原始函数文档"""
    return True

# 检查装饰后的函数签名
print(inspect.signature(original))
# (a: int, b: str = 'default') -> bool

# 动态调用
sig = inspect.signature(original)
bound = sig.bind(a=10, b="test")
print(original(*bound.args, **bound.kwargs))  # True
```

## 四、装饰器进阶模式

### 4.1 装饰器传递额外信息

```python
import functools

def with_cache(cache_attr="_cache"):
    """通过属性传递缓存"""
    def decorator(func):
        @functools.wraps(func)
        def wrapper(*args, **kwargs):
            if not hasattr(wrapper, cache_attr):
                setattr(wrapper, cache_attr, {})
            
            cache = getattr(wrapper, cache_attr)
            key = str(args) + str(kwargs)
            
            if key in cache:
                return cache[key]
            
            result = func(*args, **kwargs)
            cache[key] = result
            return result
        return wrapper
    return decorator

@with_cache(cache_attr="_memo")
def expensive_computation(n):
    print(f"执行计算 n={n}")
    return n ** 2

print(expensive_computation(5))  # 执行计算
print(expensive_computation(5))  # 使用缓存
print(expensive_computation._memo)  # {'(...)': 25}
```

### 4.2 装饰器链式调用

```python
import functools

def decorator_maker(name):
    """装饰器工厂"""
    def decorator(func):
        @functools.wraps(func)
        def wrapper(*args, **kwargs):
            print(f"[{name}] 开始")
            result = func(*args, **kwargs)
            print(f"[{name}] 结束")
            return result
        return wrapper
    return decorator

# 链式调用
@decorator_maker("LOG")
@decorator_maker("TIMER")
@decorator_maker("VALIDATE")
def process():
    print("处理中...")

process()

# 输出顺序：
# [LOG] 开始
# [TIMER] 开始
# [VALIDATE] 开始
# 处理中...
# [VALIDATE] 结束
# [TIMER] 结束
# [LOG] 结束
```

### 4.3 状态保持装饰器

```python
import functools

class DecoratorState:
    """装饰器状态管理器"""
    
    def __init__(self):
        self._states = {}
    
    def __call__(self, func):
        state_key = f"_state_{func.__name__}"
        
        @functools.wraps(func)
        def wrapper(*args, **kwargs):
            if not hasattr(wrapper, state_key):
                setattr(wrapper, state_key, {})
            
            state = getattr(wrapper, state_key)
            wrapper.state = state
            
            return func(*args, **kwargs)
        
        return wrapper

# 使用
state_manager = DecoratorState()

@state_manager
def counter():
    """带状态的计数器"""
    if not hasattr(counter, '_counter'):
        counter._counter = 0
    counter._counter += 1
    return counter._counter

print(counter())  # 1
print(counter())  # 2
print(counter())  # 3
```

## 五、实际应用场景

### 5.1 方法分发装饰器

```python
import functools

class Router:
    """路由装饰器"""
    
    def __init__(self):
        self.routes = {}
    
    def route(self, path, method="GET"):
        """注册路由"""
        def decorator(func):
            key = f"{method}:{path}"
            self.routes[key] = func
            
            @functools.wraps(func)
            def wrapper(*args, **kwargs):
                return func(*args, **kwargs)
            
            return wrapper
        return decorator
    
    def dispatch(self, path, method="GET"):
        """分发请求"""
        key = f"{method}:{path}"
        if key not in self.routes:
            raise ValueError(f"未找到路由：{key}")
        return self.routes[key]()

# 使用
router = Router()

@router.route("/users", "GET")
def get_users():
    return ["user1", "user2"]

@router.route("/users", "POST")
def create_user():
    return "User created"

print(router.dispatch("/users", "GET"))
print(router.dispatch("/users", "POST"))
```

### 5.2 属性代理装饰器

```python
import functools

def cached_property(func):
    """缓存属性装饰器"""
    @functools.wraps(func)
    def wrapper(self):
        if not hasattr(wrapper, '_cache'):
            wrapper._cache = {}
        
        if func.__name__ not in wrapper._cache:
            wrapper._cache[func.__name__] = func(self)
        
        return wrapper._cache[func.__name__]
    
    return property(wrapper)

class DataProcessor:
    def __init__(self, data):
        self.data = data
    
    @cached_property
    def expensive_result(self):
        """耗时的计算属性"""
        print("计算中...")
        return sum(self.data) / len(self.data)
    
    @cached_property
    def processed_data(self):
        """处理后的数据"""
        print("处理中...")
        return [x * 2 for x in self.data]

processor = DataProcessor([1, 2, 3, 4, 5])

print(processor.expensive_result)  # 计算中...\n3.0
print(processor.expensive_result)  # 直接返回缓存，无打印
```

### 5.3 上下文管理器装饰器

```python
import functools
from contextlib import contextmanager

@contextmanager
def transaction(commit=True):
    """事务上下文管理器装饰器"""
    print("开始事务")
    try:
        yield {"committed": False}
        if commit:
            print("提交事务")
    except Exception as e:
        print(f"回滚事务：{e}")
        raise

@functools.wraps(transaction)
def with_transaction(func):
    """事务装饰器"""
    @functools.wraps(func)
    def wrapper(*args, **kwargs):
        with transaction():
            return func(*args, **kwargs)
    return wrapper

@with_transaction
def transfer_money(from_account, to_account, amount):
    print(f"转账 ${amount} 从 {from_account} 到 {to_account}")
    return True

transfer_money("A", "B", 100)
```

## 六、装饰器与类型提示

### 6.1 使用 typing.wraps

```python
from functools import wraps
from typing import Callable, TypeVar, ParamSpec

P = ParamSpec('P')
T = TypeVar('T')

def type_safe(func: Callable[P, T]) -> Callable[P, T]:
    """类型安全的装饰器"""
    @wraps(func)
    def wrapper(*args: P.args, **kwargs: P.kwargs) -> T:
        return func(*args, **kwargs)
    return wrapper

@type_safe
def add(a: int, b: int) -> int:
    return a + b

# 类型提示保持不变
print(add.__annotations__)  # {'a': int, 'b': int, 'return': int}
```

### 6.2 装饰器组合

```python
import functools
from typing import Callable, Any

def compose(*decorators):
    """组合多个装饰器"""
    def composed_decorator(func):
        result = func
        for decorator in decorators:
            result = decorator(result)
        return result
    return composed_decorator

def log(func):
    @functools.wraps(func)
    def wrapper(*args, **kwargs):
        print(f"LOG: {func.__name__}")
        return func(*args, **kwargs)
    return wrapper

def timing(func):
    @functools.wraps(func)
    def wrapper(*args, **kwargs):
        import time
        start = time.time()
        result = func(*args, **kwargs)
        print(f"TIME: {time.time() - start:.4f}s")
        return result
    return wrapper

def validate(func):
    @functools.wraps(func)
    def wrapper(*args, **kwargs):
        print("VALIDATE: 参数检查通过")
        return func(*args, **kwargs)
    return wrapper

@compose(log, timing, validate)
def process_data(data):
    return data.upper()

print(process_data("test"))
```

## 七、常见问题处理

### 7.1 装饰器与类方法

```python
import functools

def log_method(func):
    """方法日志装饰器"""
    @functools.wraps(func)
    def wrapper(self, *args, **kwargs):
        print(f"调用 {func.__name__}")
        return func(self, *args, **kwargs)
    return wrapper

class MyClass:
    def __init__(self, value):
        self.value = value
    
    @log_method
    def get_value(self):
        return self.value
    
    @classmethod
    @log_method
    def create(cls):
        return cls(0)
    
    @staticmethod
    @log_method
    def helper():
        return "helper"

obj = MyClass(42)
print(obj.get_value())  # 调用 get_value\n42
print(MyClass.create())  # 调用 create\n<MyClass instance>
print(MyClass.helper())  # 调用 helper\nhelper
```

### 7.2 装饰器处理异常

```python
import functools

def handle_errors(func):
    """统一异常处理装饰器"""
    @functools.wraps(func)
    def wrapper(*args, **kwargs):
        try:
            return func(*args, **kwargs)
        except ValueError as e:
            return {"error": "ValueError", "message": str(e)}
        except TypeError as e:
            return {"error": "TypeError", "message": str(e)}
        except Exception as e:
            return {"error": "Unknown", "message": str(e)}
    return wrapper

@handle_errors
def parse_number(s):
    return int(s)

print(parse_number("42"))      # 42
print(parse_number("abc"))     # {'error': 'ValueError', ...}
```

### 7.3 装饰器可选参数

```python
import functools

def debug(print_args=True, print_result=True):
    """可选参数的调试装饰器"""
    def decorator(func):
        @functools.wraps(func)
        def wrapper(*args, **kwargs):
            if print_args:
                print(f"参数：args={args}, kwargs={kwargs}")
            result = func(*args, **kwargs)
            if print_result:
                print(f"结果：{result}")
            return result
        return wrapper
    return decorator

@debug()                    # 使用默认参数
@debug(print_args=False)    # 不打印参数
@debug(print_result=False)  # 不打印结果
@debug(print_args=False, print_result=False)  # 两个都不打印
def compute(x, y):
    return x + y
```

## 九、易错点与避坑指南

1. **装饰器修改了被装饰对象**：使用 `functools.wraps` 保留元数据
2. **类方法装饰器顺序**：staticmethod/classmethod 应在内层
3. **闭包变量引用**：使用 `nonlocal` 声明
4. **装饰器无限递归**：注意装饰器自身的实现
5. **性能开销**：装饰器本身有调用开销
