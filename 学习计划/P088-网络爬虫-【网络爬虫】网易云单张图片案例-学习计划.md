```python
# 网易云音乐的反爬机制：

反爬机制 = """
1. Referer 检查
   - 必须有 Referer 指向 music.163.com
   - 否则返回 403

2. iframe 嵌套
   - 歌曲页面中的封面可能在 iframe 里
   - 直接请求歌曲页面拿不到完整内容

3. API 限制
   - 部分 API 需要登录
   - 部分 API 有频率限制

4. 加密参数
   - 搜索等 API 需要加密参数
   - 需要分析 JS 代码
"""

# 应对策略：
# 1. 设置正确的 Referer
# 2. 使用官方 API（推荐）
# 3. 使用 Selenium 渲染（备选）
```

### 第 2 步：方法 1 - 使用官方 API（15 分钟）

```python
import requests
from pathlib import Path

def download_cover(song_id, save_path=None):
    """使用官方 API 下载网易云歌曲封面"""
    # 官方 API
    api_url = f"https://music.163.com/api/song/detail?id={song_id}&ids=%5B{song_id}%5D"

    headers = {
        "User-Agent": "Mozilla/5.0 (Windows NT 10.0; Win64; x64) AppleWebKit/537.36",
        "Referer": "https://music.163.com/",
    }

    response = requests.get(api_url, headers=headers)
    data = response.json()

    if data["code"] == 200:
        song = data["songs"][0]
        song_name = song["name"]
        artist = song["artists"][0]["name"]
        cover_url = song["album"]["picUrl"]

        print(f"歌曲: {song_name}")
        print(f"歌手: {artist}")
        print(f"封面: {cover_url}")

        # 下载封面
        if save_path is None:
            save_path = f"cover_{song_id}.jpg"

        img_response = requests.get(cover_url, headers=headers)
        with open(save_path, "wb") as f:
            f.write(img_response.content)

        print(f"封面已保存到: {save_path}")
        return save_path
    else:
        print(f"获取失败: {data}")
        return None

# 测试

```

### 第 3 步：方法 2 - 解析网页源码（15 分钟）

```python
import requests
import re
from pathlib import Path

def download_cover_from_page(song_id, save_path=None):
    """从歌曲页面解析封面 URL"""
    url = f"https://music.163.com/song?id={song_id}"

    headers = {
        "User-Agent": "Mozilla/5.0 (Windows NT 10.0; Win64; x64) AppleWebKit/537.36",
        "Referer": "https://music.163.com/",
    }

    response = requests.get(url, headers=headers)
    response.encoding = "utf-8"

    # 用正则提取封面 URL
    # 网易云页面中封面通常在 meta 或 img 标签中
    patterns = [
        r'<img src="([^"]+)"[^>]*class="j-img"',  # 旧版
        r'"picUrl"\s*:\s*"([^"]+)"',  # JSON 格式
        r'<meta property="og:image" content="([^"]+)"',  # Open Graph
    ]

    for pattern in patterns:
        match = re.search(pattern, response.text)
        if match:
            cover_url = match.group(1)
            print(f"找到封面: {cover_url}")

            # 下载
            if save_path is None:
                save_path = f"cover_{song_id}.jpg"

            img_response = requests.get(cover_url, headers=headers)
            with open(save_path, "wb") as f:
                f.write(img_response.content)

            print(f"保存到: {save_path}")
            return save_path

    print("未找到封面")
    return None
```

### 第 4 步：方法 3 - 使用 Selenium（备选）（15 分钟）

```python
# 备选方案：使用 Selenium 处理 JS 渲染
# 当 API 和网页解析都失效时使用

from selenium import webdriver
from selenium.webdriver.common.by import By
import time

def download_cover_selenium(song_id, save_path=None):
    """使用 Selenium 获取封面（处理 iframe）"""
    driver = webdriver.Chrome()

    try:
        # 打开歌曲页面
        url = f"https://music.163.com/song?id={song_id}"
        driver.get(url)

        # 等待页面加载
        time.sleep(3)

        # 查找封面元素
        # 可能需要切换 iframe
        try:
            # 尝试直接查找
            img = driver.find_element(By.CSS_SELECTOR, ".j-img")
        except:
            # 可能需要切换 iframe
            driver.switch_to.frame("contentFrame")
            img = driver.find_element(By.CSS_SELECTOR, ".j-img")

        # 获取截图
        if save_path is None:
            save_path = f"cover_{song_id}.png"

        img.screenshot(save_path)
        print(f"封面已保存到: {save_path}")

        return save_path

    finally:
        driver.quit()
```

### 第 5 步：批量下载工具（10 分钟）

```python
import requests
import os

def batch_download_covers(song_ids, save_dir="covers"):
    """批量下载多首歌曲的封面"""
    os.makedirs(save_dir, exist_ok=True)

    success_count = 0
    for song_id in song_ids:
        try:
            save_path = os.path.join(save_dir, f"{song_id}.jpg")
            if os.path.exists(save_path):
                print(f"[跳过] {song_id} 已存在")
                continue

            result = download_cover(song_id, save_path)
            if result:
                success_count += 1

        except Exception as e:
            print(f"[失败] {song_id}: {e}")

    print(f"\n完成: {success_count}/{len(song_ids)}")

# 测试
song_ids = [
    476875460,  # 歌曲1
    1894905497, # 歌曲2
    1422024148, # 歌曲3
]
batch_download_covers(song_ids)
```
