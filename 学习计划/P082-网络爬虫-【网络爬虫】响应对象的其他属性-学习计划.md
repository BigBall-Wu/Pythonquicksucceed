## 二、为什么先学这一节

Response 对象不仅仅是 `response.text`。理解 Response 的完整属性可以：
- 检查请求是否成功（status_code）
- 获取响应头信息（Content-Type、Set-Cookie）
- 调试请求问题（elapsed、history）
- 处理复杂的响应（links、cookies）

## 五、具体学习步骤

### 第 1 步：Response 状态属性（10 分钟）

```python
import requests

r = requests.get("https://api.github.com")

# 状态相关属性
print(r.status_code)          # 200（HTTP 状态码）
print(r.reason)               # "OK"（状态描述）
print(r.ok)                   # True（状态码 < 400）

# 检查状态码的快捷方式
r.raise_for_status()          # 状态码 >= 400 时抛出异常

# 常用状态码判断
if r.status_code == 200:
    print("请求成功")
elif r.status_code == 404:
    print("资源不存在")
elif r.status_code >= 500:
    print("服务器错误")

# 重定向相关
print(r.is_redirect)           # False（是否重定向）
print(r.is_permanent_redirect) # False（是否永久重定向）
print(r.url)                  # 最终 URL（可能经过重定向）
print(r.history)              # 重定向历史列表
```

### 第 2 步：响应头属性（15 分钟）

```python
import requests

r = requests.get("https://api.github.com")

# 响应头（dict 类型）
print(r.headers)

# 常用响应头
print(r.headers["Content-Type"])        # "application/json; charset=utf-8"
print(r.headers.get("Content-Length"))  # "1234"
print(r.headers.get("Set-Cookie"))      # Cookie 字符串

# 检查内容类型
content_type = r.headers.get("Content-Type", "")
if "text/html" in content_type:
    print("HTML 页面")
elif "application/json" in content_type:
    print("JSON 数据")
elif "image/" in content_type:
    print("图片")

# 检查编码
print(r.encoding)  # "utf-8"（requests 自动检测）
r.encoding = "gbk"  # 手动指定编码
```

### 第 3 步：Cookie 属性（15 分钟）

```python
import requests

r = requests.get("https://www.example.com")

# Cookie 属性
print(r.cookies)                  # RequestsCookieJar 对象
print(r.cookies.get_dict())        # 转换为字典
print(r.cookies.get("session"))     # 获取特定 Cookie

# 使用 Cookie 进行后续请求
session_id = r.cookies.get("session_id")
if session_id:
    r2 = requests.get(
        "https://www.example.com/dashboard",
        cookies={"session_id": session_id}
    )

# 从响应头解析 Cookie
set_cookie = r.headers.get("Set-Cookie")
# 格式：name=value; Path=/; HttpOnly; Secure
```

### 第 4 步：请求信息属性（10 分钟）

```python
import requests

r = requests.get(
    "https://api.github.com",
    headers={"User-Agent": "MyBot/1.0"}
)

# 请求信息（通过 response.request 访问）
print(r.request)              # PreparedRequest 对象
print(r.request.headers)       # 发送的请求头
print(r.request.method)        # "GET"
print(r.request.url)          # 请求 URL
print(r.request.body)          # 请求体（POST 时有）

# 用途：对比发送的请求和接收的响应
print(f"请求: {r.request.url}")
print(f"响应: {r.url}")  # 可能因为重定向不同
```

### 第 5 步：时间和历史属性（10 分钟）

```python
import requests

# 请求耗时
r = requests.get("https://httpbin.org/delay/1")
print(r.elapsed)               # datetime.timedelta 对象
print(r.elapsed.total_seconds())  # 转换为秒（如 1.023 秒）

# 重定向历史
r = requests.get("http://httpbin.org/redirect/2")
print(r.url)                   # 最终 URL
print(r.history)               # [Response(302), Response(302)]

for resp in r.history:
    print(f"  {resp.status_code} -> {resp.url}")

# 应用：检查是否被重定向到错误页面
if r.is_redirect:
    print(f"被重定向到: {r.url}")
```

### 第 6 步：高级用法（10 分钟）

```python
import requests

# 原始响应（socket level）
r = requests.get("https://httpbin.org/bytes/100", stream=True)
print(r.raw)                   # urllib3.HTTPResponse
print(r.raw.read(50))          # 读原始字节

# 解析 Link 头（分页用）
r = requests.get("https://api.github.com/repos/psf/requests/issues")
print(r.links)                 # 解析 Link 头
# {'next': {'url': '...', 'rel': 'next'}, 'last': {...}}

if "next" in r.links:
    next_url = r.links["next"]["url"]
    print(f"下一页: {next_url}")

# JSON 解析（带错误处理）
r = requests.get("https://api.github.com")
try:
    data = r.json()           # 返回 dict 或 list
except ValueError as e:
    print(f"JSON 解析失败: {e}")
```

### 练习 1：响应诊断函数

```python
def diagnose_response(r):
    """诊断响应对象，打印所有重要信息"""
    print("=== 响应诊断 ===")
    print(f"URL: {r.url}")
    print(f"状态码: {r.status_code}")
    print(f"状态描述: {r.reason}")
    print(f"请求成功: {r.ok}")
    print(f"内容类型: {r.headers.get('Content-Type')}")
    print(f"内容长度: {r.headers.get('Content-Length')}")
    print(f"请求耗时: {r.elapsed.total_seconds():.3f}秒")
    if r.history:
        print(f"重定向次数: {len(r.history)}")
    # ... 更多诊断信息

# 使用
r = requests.get("https://httpbin.org/get")
diagnose_response(r)
```

### 练习 2：分页获取

```python
# 使用 Link 头实现分页获取
def get_all_pages(start_url):
    """获取所有分页数据"""
    results = []
    url = start_url

    while url:
        r = requests.get(url, headers={"User-Agent": "MyBot/1.0"})
        results.extend(r.json())  # 假设返回列表

        # 获取下一页
        if "next" in r.links:
            url = r.links["next"]["url"]
        else:
            url = None

    return results
```
