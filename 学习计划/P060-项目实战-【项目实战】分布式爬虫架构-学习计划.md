## 一、分布式爬虫概述

### 1.1 为什么需要分布式

| 单机爬虫 | 分布式爬虫 |
|----------|------------|
| 受限于单机带宽 | 多机带宽聚合 |
| 爬取速度有限 | 并行爬取提速 |
| 单点故障 | 无中心节点 |
| 扩展困难 | 线性扩展 |

### 1.2 架构组件

```
分布式爬虫架构
├── URL 管理器（Redis）
│   ├── 已爬取 URL 集合
│   └── 待爬取 URL 队列
├── 任务调度器
│   ├── 分配爬取任务
│   └── 监控爬取状态
├── 爬虫节点
│   ├── Worker 进程
│   └── 数据处理
├── 数据存储
│   ├── MongoDB
│   └── Elasticsearch
└── 监控平台
    └── 状态可视化
```

## 三、URL 管理器

### 3.1 基于 Redis 的 URL 管理

```python
import redis
from typing import Optional, Set

class RedisURLManager:
    """基于 Redis 的 URL 管理器"""
    
    def __init__(self, host="localhost", port=6379, db=0):
        self.r = redis.Redis(host=host, port=port, db=db, decode_responses=True)
        self.crawled_key = "urls:crawled"      # 已爬取
        self.pending_key = "urls:pending"        # 待爬取
        self.error_key = "urls:error"           # 错误 URL
    
    def add_url(self, url: str) -> bool:
        """添加 URL 到待爬队列"""
        # 已爬取的不再添加
        if self.r.sismember(self.crawled_key, url):
            return False
        # 已在队列的不再添加
        if self.r.sismember(f"{self.pending_key}:set", url):
            return False
        
        # 添加到队列和已见集合
        self.r.lpush(self.pending_key, url)
        self.r.sadd(f"{self.pending_key}:set", url)
        return True
    
    def add_urls(self, urls: list) -> int:
        """批量添加 URL"""
        count = 0
        for url in urls:
            if self.add_url(url):
                count += 1
        return count
    
    def get_url(self) -> Optional[str]:
        """获取一个待爬 URL"""
        url = self.r.rpop(self.pending_key)
        if url:
            self.r.srem(f"{self.pending_key}:set", url)
        return url
    
    def mark_crawled(self, url: str):
        """标记为已爬取"""
        self.r.sadd(self.crawled_key, url)
        self.r.srem(f"{self.pending_key}:set", url)
    
    def mark_error(self, url: str):
        """标记为错误"""
        self.r.lpush(self.error_key, url)
        self.r.srem(f"{self.pending_key}:set", url)
    
    def get_pending_count(self) -> int:
        """待爬取数量"""
        return self.r.llen(self.pending_key)
    
    def get_crawled_count(self) -> int:
        """已爬取数量"""
        return self.r.scard(self.crawled_key)
    
    def get_error_count(self) -> int:
        """错误数量"""
        return self.r.llen(self.error_key)
    
    def is_crawled(self, url: str) -> bool:
        """检查是否已爬取"""
        return self.r.sismember(self.crawled_key, url)
```

### 3.2 去重机制

```python
class DistributedDedup:
    """分布式去重"""
    
    def __init__(self, redis_client):
        self.r = redis_client
    
    def url_fingerprint(self, url: str) -> str:
        """生成 URL 指纹"""
        import hashlib
        return hashlib.md5(url.encode()).hexdigest()
    
    def check_and_add(self, url: str) -> bool:
        """
        检查并添加
        返回 True 表示新 URL（未见过）
        返回 False 表示重复
        """
        fp = self.url_fingerprint(url)
        
        # 使用 SET NX（不存在才添加，原子操作）
        added = self.r.set(f"fp:{fp}", "1", nx=True)
        return bool(added)
```

## 四、消息队列架构

### 4.1 简单消息队列实现

```python
import redis
import json
import time
from typing import Any, Dict, Optional
from dataclasses import dataclass, asdict
from datetime import datetime

@dataclass
class Task:
    """爬取任务"""
    task_id: str
    url: str
    priority: int = 0
    retry: int = 0
    created_at: str = ""
    
    def __post_init__(self):
        if not self.created_at:
            self.created_at = datetime.now().isoformat()

class TaskQueue:
    """任务队列"""
    
    def __init__(self, host="localhost", port=6379, db=0):
        self.r = redis.Redis(host=host, port=port, db=db, decode_responses=True)
    
    def push(self, task: Task):
        """添加任务"""
        data = json.dumps(asdict(task))
        # 优先级队列：使用有序集合
        score = -task.priority  # 负数使高优先级在前
        self.r.zadd("task:queue", {data: score})
    
    def pop(self, timeout=0) -> Optional[Task]:
        """获取任务（阻塞）"""
        if timeout > 0:
            result = self.r.bzpopmin("task:queue", timeout=timeout)
            if result:
                _, data, score = result
                return Task(**json.loads(data))
        else:
            result = self.r.zpopmin("task:queue", count=1)
            if result:
                data, score = result[0]
                return Task(**json.loads(data))
        return None
    
    def size(self) -> int:
        """队列大小"""
        return self.r.zcard("task:queue")
    
    def requeue(self, task: Task):
        """重新入队"""
        task.retry += 1
        self.push(task)
```

### 4.2 分布式锁

```python
import redis
import time
import uuid
from contextlib import contextmanager

class DistributedLock:
    """分布式锁"""
    
    def __init__(self, redis_client, lock_name, timeout=30):
        self.r = redis_client
        self.lock_name = f"lock:{lock_name}"
        self.timeout = timeout
        self.token = str(uuid.uuid4())
    
    def acquire(self, blocking=True, blocking_timeout=10) -> bool:
        """获取锁"""
        end_time = time.time() + blocking_timeout
        
        while True:
            # SET NX EX（原子操作）
            if self.r.set(self.lock_name, self.token, nx=True, ex=self.timeout):
                return True
            
            if not blocking:
                return False
            
            if time.time() > end_time:
                return False
            
            time.sleep(0.01)
    
    def release(self):
        """释放锁"""
        # Lua 脚本保证原子性
        lua = """
        if redis.call("get", KEYS[1]) == ARGV[1] then
            return redis.call("del", KEYS[1])
        else
            return 0
        end
        """
        self.r.eval(lua, 1, self.lock_name, self.token)
    
    @contextmanager
    def lock(self):
        """上下文管理器用法"""
        if self.acquire():
            try:
                yield
            finally:
                self.release()
        else:
            raise Exception(f"获取锁失败: {self.lock_name}")
```

## 五、Worker 节点实现

### 5.1 Worker 架构

```python
import asyncio
import aiohttp
import redis
import json
import signal
import logging
from datetime import datetime
from typing import Optional

from src.url_manager import RedisURLManager
from src.task_queue import TaskQueue

logger = logging.getLogger("worker")

class CrawlerWorker:
    """爬虫 Worker 节点"""
    
    def __init__(self, worker_id: str, redis_host="localhost"):
        self.worker_id = worker_id
        self.url_manager = RedisURLManager(host=redis_host)
        self.task_queue = TaskQueue(host=redis_host)
        self.r = redis.Redis(host=redis_host, decode_responses=True)
        
        self.running = True
        self.session: Optional[aiohttp.ClientSession] = None
    
    async def init(self):
        """初始化"""
        self.session = aiohttp.ClientSession(
            headers={
                "User-Agent": "Distributed Crawler Worker/1.0"
            }
        )
        
        # 注册 Worker
        self.r.sadd("workers:online", self.worker_id)
        self.r.hset(f"worker:{self.worker_id}", mapping={
            "status": "running",
            "started_at": datetime.now().isoformat(),
            "count": 0
        })
        
        logger.info(f"Worker {self.worker_id} 启动")
    
    async def close(self):
        """关闭"""
        if self.session:
            await self.session.close()
        
        self.r.srem("workers:online", self.worker_id)
        self.r.hset(f"worker:{self.worker_id}", "status", "stopped")
        
        logger.info(f"Worker {self.worker_id} 停止")
    
    async def fetch(self, url: str) -> Optional[str]:
        """抓取页面"""
        try:
            async with self.session.get(url, timeout=aiohttp.ClientTimeout(total=30)) as response:
                if response.status == 200:
                    return await response.text()
                else:
                    logger.warning(f"HTTP {response.status}: {url}")
                    return None
        except Exception as e:
            logger.error(f"请求失败 {url}: {e}")
            return None
    
    async def process(self, url: str):
        """处理单个 URL"""
        try:
            # 标记正在处理
            self.r.hset(f"worker:{self.worker_id}:processing", url, datetime.now().isoformat())
            
            # 抓取
            html = await self.fetch(url)
            if html:
                # 标记已爬取
                self.url_manager.mark_crawled(url)
                
                # 更新计数
                self.r.hincrby(f"worker:{self.worker_id}", "count", 1)
                
                return {"url": url, "html": html, "success": True}
            
            return {"url": url, "success": False}
        
        except Exception as e:
            logger.error(f"处理失败 {url}: {e}")
            self.url_manager.mark_error(url)
            return {"url": url, "success": False, "error": str(e)}
        
        finally:
            self.r.hdel(f"worker:{self.worker_id}:processing", url)
    
    async def run(self):
        """运行 Worker"""
        await self.init()
        
        try:
            while self.running:
                # 获取任务（阻塞等待）
                task = self.task_queue.pop(timeout=5)
                
                if task:
                    await self.process(task.url)
                else:
                    # 检查是否还有其他 Worker
                    online_workers = self.r.scard("workers:online")
                    if online_workers == 0:
                        logger.info("没有在线 Worker，退出")
                        break
        
        except KeyboardInterrupt:
            logger.info("收到停止信号")
        
        finally:
            await self.close()
    
    def stop(self):
        """停止 Worker"""
        self.running = False
```

### 6.1 主调度器

```python
import redis
import json
import asyncio
from datetime import datetime
from typing import List

class TaskScheduler:
    """任务调度器"""
    
    def __init__(self, redis_host="localhost"):
        self.r = redis.Redis(host=redis_host, decode_responses=True)
        self.url_manager = RedisURLManager(host=redis_host)
        self.task_queue = TaskQueue(host=redis_host)
    
    def add_seed_urls(self, urls: List[str]):
        """添加种子 URL"""
        count = self.url_manager.add_urls(urls)
        print(f"添加 {count} 个种子 URL")
    
    def distribute_tasks(self, count=100):
        """分发任务到队列"""
        distributed = 0
        for _ in range(count):
            url = self.url_manager.get_url()
            if not url:
                break
            
            from src.task_queue import Task
            task = Task(
                task_id=f"task_{datetime.now().timestamp()}",
                url=url,
                priority=0
            )
            self.task_queue.push(task)
            distributed += 1
        
        print(f"分发 {distributed} 个任务")
        return distributed
    
    def get_status(self):
        """获取状态"""
        return {
            "online_workers": self.r.scard("workers:online"),
            "pending_urls": self.url_manager.get_pending_count(),
            "crawled_urls": self.url_manager.get_crawled_count(),
            "error_urls": self.url_manager.get_error_count(),
            "queued_tasks": self.task_queue.size(),
            "workers": list(self.r.smembers("workers:online"))
        }
    
    def monitor(self):
        """监控状态"""
        import time
        
        while True:
            status = self.get_status()
            print(f"""
=============================================
  分布式爬虫状态
=============================================
  在线 Worker: {status['online_workers']}
  待爬 URL: {status['pending_urls']}
  已爬 URL: {status['crawled_urls']}
  错误 URL: {status['error_urls']}
  队列任务: {status['queued_tasks']}
=============================================
            """)
            time.sleep(10)
```

### 7.2 监控告警

```python
import redis
from datetime import datetime

class Monitor:
    """监控系统"""
    
    def __init__(self, redis_host="localhost"):
        self.r = redis.Redis(host=redis_host, decode_responses=True)
    
    def check_health(self):
        """健康检查"""
        issues = []
        
        # 检查 Redis 连接
        try:
            self.r.ping()
        except:
            issues.append("Redis 连接失败")
        
        # 检查 Worker 状态
        workers = self.r.smembers("workers:online")
        if not workers:
            issues.append("没有在线 Worker")
        
        # 检查爬取进度
        pending = self.r.llen("urls:pending")
        crawled = self.r.scard("urls:crawled")
        
        if pending == 0 and crawled == 0:
            issues.append("没有爬取任务")
        
        if issues:
            print(f"[ALERT] {'; '.join(issues)}")
            return False
        
        return True
    
    def get_metrics(self):
        """获取指标"""
        return {
            "timestamp": datetime.now().isoformat(),
            "workers": self.r.scard("workers:online"),
            "pending": self.r.llen("urls:pending"),
            "crawled": self.r.scard("urls:crawled"),
            "errors": self.r.llen("urls:error")
        }
```
