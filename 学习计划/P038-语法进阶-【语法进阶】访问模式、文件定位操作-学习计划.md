### 第 1 步：open() 的 8 种模式（10 分钟）

```python
# 文本模式
'r'   # 只读（默认）
'w'   # 写入（覆盖）
'a'   # 追加
'x'   # 排他创建（已存在则报错）
'r+'  # 读写
'w+'  # 读写（覆盖）
'a+'  # 读追加

# 二进制模式（加 'b'）
'rb'  # 二进制读
'wb'  # 二进制写
'ab'  # 二进制追加
'rb+' # 二进制读写

# 用
with open("file.txt", "r") as f:        # 读
    content = f.read()

with open("file.txt", "w") as f:        # 写（覆盖）
    f.write("new content")

with open("file.txt", "a") as f:        # 追加
    f.write("more content")
```

### 第 2 步：seek() 和 tell()（10 分钟）

```python
with open("file.txt", "r") as f:
    print(f.tell())        # 0（当前位置）

    content = f.read(5)    # 读 5 个字符
    print(f.tell())        # 5

    f.seek(0)              # 回到开头
    print(f.tell())        # 0

    f.seek(10)             # 跳到第 10 个字符
    content = f.read()      # 从第 10 个开始读

# 二进制 seek
with open("file.bin", "rb") as f:
    f.seek(100)            # 跳到第 100 字节
    data = f.read(50)      # 读 50 字节

# seek 偏移量
with open("file.txt", "r") as f:
    f.seek(0, 2)           # whence=2 表示从末尾，0 表示偏移 0 → 文件末尾
    f.seek(-10, 2)         # 从末尾往前 10 字节
```

### 第 3 步：逐行 vs 全部读取（10 分钟）

```python
# 方式 1：readline() 逐行
with open("big.log") as f:
    line = f.readline()
    while line:
        process(line)
        line = f.readline()

# 方式 2：for 循环（推荐，内存友好）
with open("big.log") as f:
    for line in f:           # 内部用 buffer，每次读一行
        process(line)

# 方式 3：readlines() 读所有行到列表
with open("small.txt") as f:
    lines = f.readlines()    # 列表
    # 适合小文件

# 方式 4：read() 读所有内容
with open("small.txt") as f:
    content = f.read()         # 字符串
```

### 第 4 步：处理大文件（10 分钟）

```python
# ❌ 错误：一次读全部
with open("huge.log") as f:
    content = f.read()        # 可能内存爆炸！

# ✓ 正确：分块读
def process_large_file(filepath, chunk_size=1024 * 1024):
    """每次读 1MB"""
    with open(filepath, "r") as f:
        while True:
            chunk = f.read(chunk_size)
            if not chunk:
                break
            yield chunk

for chunk in process_large_file("huge.log"):
    # 处理每块
    process(chunk)

# ✓ 用迭代器
def read_lines(filepath):
    """逐行迭代"""
    with open(filepath, "r") as f:
        for line in f:
            yield line.strip()

# 统计大文件行数
count = sum(1 for _ in open("huge.log", "r"))
```

### 第 5 步：二进制文件（10 分钟）

```python
# 写二进制
data = bytes([0, 1, 2, 3, 255])
with open("binary.bin", "wb") as f:
    f.write(data)

# 读二进制
with open("binary.bin", "rb") as f:
    data = f.read()             # bytes
    print(data[0])               # 0
    print(data[4])               # 255

# 图片、音频、视频都是二进制
with open("photo.jpg", "rb") as f:
    image_data = f.read()

# 复制二进制文件
with open("source.mp4", "rb") as src:
    with open("dest.mp4", "wb") as dst:
        dst.write(src.read())
```

### 第 6 步：实战 - 日志分析器（10 分钟）

```python
import os

def analyze_log(filepath):
    """分析日志"""
    levels = {"INFO": 0, "WARNING": 0, "ERROR": 0}

    with open(filepath, "r") as f:
        for line_num, line in enumerate(f, 1):
            for level in levels:
                if f"[{level}]" in line:
                    levels[level] += 1
                    break
    return levels, line_num

# 用（log.txt 是 10GB 大文件）
levels, total_lines = analyze_log("app.log")
print(f"总行数：{total_lines}")
for level, count in levels.items():
    print(f"{level}: {count}")
```

### 第 7 步：常见错误（5 分钟）

```python
# 错误 1：忘了 encoding，中文乱码
with open("chinese.txt", "r") as f:    # ⚠️ Windows 默认 GBK
    content = f.read()                  # 可能乱码

# 解决：指定 encoding
with open("chinese.txt", "r", encoding="utf-8") as f:
    content = f.read()

# 错误 2：r+ 模式不会自动创建文件
with open("nonexistent.txt", "r+") as f:  # ❌ FileNotFoundError
    f.write("x")

# 错误 3：b 模式不能用 encoding
with open("file.bin", "rb", encoding="utf-8") as f:  # ❌ ValueError
    pass
```

```python
# 统计文件的行数、单词数、字符数
def file_stats(filepath):
    with open(filepath, "r", encoding="utf-8") as f:
        content = f.read()
    lines = content.split("\n")
    words = content.split()
    return len(lines), len(words), len(content)

# 用

```

```python
import os

def copy_file(src, dst):
    """复制任意文件"""
    with open(src, "rb") as f_src:
        with open(dst, "wb") as f_dst:
            while True:
                chunk = f_src.read(4096)
                if not chunk:
                    break
                f_dst.write(chunk)

copy_file("source.mp4", "dest.mp4")
```
