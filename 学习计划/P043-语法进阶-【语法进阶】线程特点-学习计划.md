## 四、具体学习步骤

### 第 1 步：进程 vs 线程 vs 协程（5 分钟）

```
进程（Process）：
- 独立的内存空间
- 重量级（启动慢）
- 适合 CPU 密集型
- 用 multiprocessing

线程（Thread）：
- 共享进程的内存
- 轻量级
- 适合 I/O 密集型
- 用 threading
- 受 GIL 限制

协程（Coroutine）：
- 单线程内切换
- 极轻量级
- 适合高并发 I/O
- 用 asyncio
```

### 第 2 步：threading 基础（10 分钟）

```python
import threading
import time

def worker(name, duration):
    """线程任务"""
    print(f"线程 {name} 开始")
    time.sleep(duration)              # 模拟 I/O
    print(f"线程 {name} 结束（睡了 {duration}s）")

# 串行
start = time.time()
worker("A", 2)
worker("B", 3)
print(f"串行耗时：{time.time() - start:.1f}s")  # 5s

# 并行
start = time.time()
t1 = threading.Thread(target=worker, args=("A", 2))
t2 = threading.Thread(target=worker, args=("B", 3))
t1.start()
t2.start()
t1.join()
t2.join()
print(f"并行耗时：{time.time() - start:.1f}s")  # 3s
```

### 第 3 步：理解 GIL（10 分钟）

**GIL**（Global Interpreter Lock）：CPython 解释器同一时刻只允许一个线程执行 Python 字节码。

```python
import threading
import time

def cpu_task(n):
    """CPU 密集型"""
    while n > 0:
        n -= 1

# 串行
start = time.time()
cpu_task(50_000_000)
cpu_task(50_000_000)
print(f"串行：{time.time() - start:.2f}s")

# 用多线程（GIL 让它不加速！）
start = time.time()
t1 = threading.Thread(target=cpu_task, args=(50_000_000,))
t2 = threading.Thread(target=cpu_task, args=(50_000_000,))
t1.start(); t2.start()
t1.join(); t2.join()
print(f"多线程：{time.time() - start:.2f}s")  # 几乎一样快！

# 解决：用多进程
import multiprocessing
start = time.time()
p1 = multiprocessing.Process(target=cpu_task, args=(50_000_000,))
p2 = multiprocessing.Process(target=cpu_task, args=(50_000_000,))
p1.start(); p2.start()
p1.join(); p2.join()
print(f"多进程：{time.time() - start:.2f}s")  # 真的快 2 倍！
```

### 第 4 步：Thread 类方式（10 分钟）

```python
import threading
import time

class MyThread(threading.Thread):
    def __init__(self, name, count):
        super().__init__()
        self.name = name
        self.count = count

    def run(self):
        """线程入口"""
        for i in range(self.count):
            print(f"{self.name}: 第 {i+1} 次")
            time.sleep(0.5)

t1 = MyThread("线程A", 3)
t2 = MyThread("线程B", 3)
t1.start()
t2.start()
t1.join()
t2.join()
print("主线程结束")
```

### 第 5 步：实战 - 多线程 URL 检查（10 分钟）

```python
import threading
import requests
import time

urls = [
    "https://www.google.com",
    "https://www.github.com",
    "https://www.python.org",
    "https://httpbin.org",
    "https://www.baidu.com",
]

def check(url):
    """检查 URL 状态"""
    try:
        r = requests.get(url, timeout=5)
        print(f"✓ {url}: {r.status_code} ({len(r.content)} bytes)")
    except Exception as e:
        print(f"✗ {url}: {e}")

# 串行
start = time.time()
for url in urls:
    check(url)
print(f"串行：{time.time() - start:.2f}s")

# 并行
start = time.time()
threads = [threading.Thread(target=check, args=(url,)) for url in urls]
for t in threads: t.start()
for t in threads: t.join()
print(f"并行：{time.time() - start:.2f}s")
# 并行速度提升 N 倍（N = 线程数）
```

### 第 6 步：常见错误（5 分钟）

```python
# 错误 1：忘记 join，主线程提前结束
t = threading.Thread(target=worker, args=("A", 5))
t.start()
# 主线程不等待 t 完成

# 错误 2：共享可变状态不加锁
counter = 0
def increment():
    global counter
    for _ in range(100000):
        counter += 1              # 不是原子操作！

t1 = threading.Thread(target=increment)
t2 = threading.Thread(target=increment)
t1.start(); t2.start(); t1.join(); t2.join()
print(counter)                       # 不一定是 200000

# 错误 3：daemon 线程误用
t = threading.Thread(target=worker)
t.daemon = True                       # 主线程结束立即终止
t.start()
# 主线程结束后 t 也被杀，可能资源未清理
```

## 六、动手练习

### 练习 1：基础多线程

```python
import threading
import time

def task(n):
    time.sleep(0.5)
    print(f"任务 {n} 完成")

# 创建 5 个线程
threads = [threading.Thread(target=task, args=(i,)) for i in range(5)]

# 启动
for t in threads:
    t.start()

# 等待
for t in threads:
    t.join()

print("所有任务完成")
```

### 练习 2：下载多张图片

```python
import threading
import requests

urls = ["https://picsum.photos/200/300"] * 10

def download(url, idx):
    r = requests.get(url)
    with open(f"img_{idx}.jpg", "wb") as f:
        f.write(r.content)

threads = [threading.Thread(target=download, args=(url, i)) for i, url in enumerate(urls)]
for t in threads: t.start()
for t in threads: t.join()
```
