### 第 1 步：理解析构函数 `__del__`（10 分钟）

```python
class File:
    """文件类，模拟资源管理"""
    def __init__(self, path):
        self.path = path
        print(f"打开文件 {path}")
        self.f = open(path, "r")

    def read(self):
        return self.f.read()

    def __del__(self):
        """析构函数（垃圾回收时调用）"""
        print(f"关闭文件 {self.path}")
        self.f.close()

# 用
f = File("data.txt")
print(f.read())
del f                       # 显式删除
# 或程序结束自动删除
```

### 第 2 步：Python 的 `__del__` 不保证被调用（10 分钟）

```python
# ⚠️ __del__ 调用时机不确定！
# - 引用计数为 0 时（del）
# - 垃圾回收时（GC）
# - 程序结束时（不一定）

# 正确做法：用 with 语句（上下文管理器）
class File:
    def __init__(self, path):
        self.path = path
        self.f = open(path, "r")

    def __enter__(self):
        return self                # 返回 self 给 with as

    def __exit__(self, exc_type, exc_val, exc_tb):
        """with 块退出时调用"""
        self.f.close()
        return False              # 不抑制异常

# 用 with
with File("data.txt") as f:
    content = f.read()
# 文件自动关闭
```

### 第 3 步：封装（Encapsulation）（10 分钟）

```python
# Python 没有真正的 private，只有约定

class Account:
    def __init__(self, balance):
        # 双下划线开头：name mangling（会被改成 _Account__balance）
        self.__balance = balance
        # 单下划线开头：约定为"内部使用"，但不强制
        self._id = 12345
        # 普通：公开
        self.owner = "Alice"

a = Account(1000)
print(a.owner)             # Alice（公开）
print(a._id)               # 12345（约定私有）
# print(a.__balance)       # ❌ AttributeError
print(a._Account__balance)  # 1000（name mangling 实际能访问）

# 推荐：用 property 做真正的封装
class Account:
    def __init__(self, balance):
        self._balance = balance

    @property
    def balance(self):
        """getter"""
        return self._balance

    @balance.setter
    def balance(self, value):
        """setter"""
        if value < 0:
            raise ValueError("余额不能为负")
        self._balance = value

a = Account(1000)
print(a.balance)            # 1000（读）
a.balance = 2000            # 写（触发 setter 验证）
# a.balance = -1             # ValueError
```

### 第 4 步：property 的 3 大用途（10 分钟）

```python
class Temperature:
    def __init__(self, celsius=0):
        self._celsius = celsius

    # 1. 只读属性
    @property
    def celsius(self):
        return self._celsius

    # 2. 只写属性
    @celsius.setter
    def celsius(self, value):
        if value < -273.15:
            raise ValueError("低于绝对零度")
        self._celsius = value

    # 3. 计算属性（无 setter）
    @property
    def fahrenheit(self):
        return self._celsius * 9 / 5 + 32

t = Temperature(25)
print(t.celsius)             # 25
print(t.fahrenheit)          # 77.0
t.celsius = 30
print(t.fahrenheit)          # 86.0
# t.celsius = -300            # ValueError
```

### 第 5 步：实战 - 用户类（带封装）（10 分钟）

```python
class User:
    def __init__(self, username, email):
        self._username = username
        self._email = email
        self._password = ""        # 永远不要存明文
        self._is_active = True
        self._created_at = None

    @property
    def username(self):
        return self._username

    @property
    def email(self):
        return self._email

    @email.setter
    def email(self, value):
        if "@" not in value:
            raise ValueError("邮箱格式错误")
        self._email = value

    @property
    def is_active(self):
        return self._is_active

    def deactivate(self):
        self._is_active = False

    def set_password(self, hashed_pwd):
        """实际项目中应存哈希值，不是明文"""
        self._password = hashed_pwd

u = User("alice", "alice@example.com")
print(u.username)             # alice
print(u.is_active)            # True

# u.email = "invalid"          # ❌ ValueError
u.email = "alice@gmail.com"
print(u.email)                # alice@gmail.com

# u._password = "secret"       # 约定私有（不推荐外部访问）
```

### 第 6 步：常见错误（5 分钟）

```python
# 错误 1：析构函数中做耗时操作
class Bad:
    def __del__(self):
        # ❌ 不要在 __del__ 中做 I/O、网络、sleep 等
        requests.post("https://api.example.com/log")
        # 原因：可能死锁、程序卡住

# 错误 2：property 忘记 setter
class Immutable:
    def __init__(self, x):
        self._x = x

    @property
    def x(self):
        return self._x
    # 没 setter → 只读

i = Immutable(5)
i.x = 10                       # ❌ AttributeError
```

class Counter:
    def __init__(self):
        self._count = 0

    def increment(self):
        self._count += 1

    @property
    def count(self):
        return self._count

c = Counter()
c.increment()
c.increment()
c.increment()

### 练习 2：with 语句实现

```python
class Timer:
    def __enter__(self):
        import time
        self.start = time.time()
        return self

    def __exit__(self, *args):
        import time
        elapsed = time.time() - self.start
        print(f"耗时 {elapsed:.2f}s")

with Timer():
    sum(range(10_000_000))
# 耗时 0.05s
```
