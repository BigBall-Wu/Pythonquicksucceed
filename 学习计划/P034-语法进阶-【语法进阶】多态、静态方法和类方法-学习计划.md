### 第 1 步：理解多态（10 分钟）

```python
class Dog:
    def speak(self):
        return "汪！"

class Cat:
    def speak(self):
        return "喵！"

class Duck:
    def speak(self):
        return "嘎！"

# 多态：同样的方法调用，不同对象有不同行为
animals = [Dog(), Cat(), Duck()]
for animal in animals:
    print(animal.speak())       # 汪！/ 喵！/ 嘎！
```

**关键**：调用 `animal.speak()` 时，Python 不在乎 animal 是什么类型，只在乎它有没有 `speak` 方法。这就是"鸭子类型"（duck typing）。

### 第 2 步：3 种方法类型（10 分钟）

```python
class MathUtils:
    pi = 3.14159                    # 类属性

    def __init__(self, name):
        self.name = name           # 实例属性

    # 1. 实例方法（最常用）
    def add(self, a, b):
        """需要 self（实例）"""
        return a + b

    # 2. 类方法
    @classmethod
    def circle_area(cls, r):
        """需要 cls（类），不需实例化"""
        return cls.pi * r ** 2

    # 3. 静态方法
    @staticmethod
    def add_static(a, b):
        """不需要 self/cls，纯粹的工具函数"""
        return a + b

# 用
m = MathUtils("工具")
print(m.add(3, 5))               # 8（实例方法）
print(MathUtils.circle_area(2))   # 12.57（类方法，无需实例）
print(MathUtils.add_static(3, 5))  # 8（静态方法）
```

### 第 3 步：3 种方法的选择（10 分钟）

```python
class User:
    count = 0

    def __init__(self, name, email):
        # 实例方法：需要访问/修改实例状态
        self.name = name
        self.email = email
        User.count += 1

    def get_info(self):
        # 访问 self.name, self.email → 用实例方法
        return f"{self.name} <{self.email}>"

    @classmethod
    def from_dict(cls, d):
        # 工厂方法：基于字典创建实例
        return cls(d["name"], d["email"])

    @classmethod
    def total(cls):
        # 访问类属性 cls.count
        return cls.count

    @staticmethod
    def is_valid_email(email):
        # 纯工具函数：不需要 self/cls
        return "@" in email and "." in email.split("@")[-1]

# 用
u1 = User("Alice", "alice@example.com")
u2 = User.from_dict({"name": "Bob", "email": "bob@example.com"})
print(User.total())               # 2（访问类属性）
print(User.is_valid_email("xxx"))   # False（静态工具方法）
```

### 第 4 步：抽象基类（ABC）（10 分钟）

```python
from abc import ABC, abstractmethod

class Shape(ABC):
    """抽象基类"""
    @abstractmethod
    def area(self):
        pass

    @abstractmethod
    def perimeter(self):
        pass

# 抽象类不能直接实例化
# s = Shape()                       # ❌ TypeError

class Circle(Shape):
    def __init__(self, r):
        self.r = r

    def area(self):
        return 3.14 * self.r ** 2

    def perimeter(self):
        return 2 * 3.14 * self.r

class Rectangle(Shape):
    def __init__(self, w, h):
        self.w = w
        self.h = h

    def area(self):
        return self.w * self.h

    def perimeter(self):
        return 2 * (self.w + self.h)

# 多态：不同形状用统一接口
shapes = [Circle(5), Rectangle(3, 4), Circle(2)]
for s in shapes:
    print(f"面积：{s.area():.2f}, 周长：{s.perimeter():.2f}")
```

### 第 5 步：实战 - 通知系统（10 分钟）

```python
from abc import ABC, abstractmethod

class Notification(ABC):
    """通知基类"""
    @abstractmethod
    def send(self, message):
        pass

class EmailNotification(Notification):
    def __init__(self, email):
        self.email = email

    def send(self, message):
        return f"Email 到 {self.email}: {message}"

class SMSNotification(Notification):
    def __init__(self, phone):
        self.phone = phone

    def send(self, message):
        return f"SMS 到 {self.phone}: {message}"

class PushNotification(Notification):
    def __init__(self, device_id):
        self.device_id = device_id

    def send(self, message):
        return f"Push 到 {self.device_id}: {message}"

# 多态通知
notifiers = [
    EmailNotification("alice@example.com"),
    SMSNotification("13812345678"),
    PushNotification("device-001"),
]

for n in notifiers:
    print(n.send("您的订单已发货"))
```

### 第 6 步：常见错误（5 分钟）

```python
# 错误 1：忘记 @staticmethod 但意图是静态
class Bad:
    def helper(x):                  # ❌ 缺 self
        return x ** 2

b = Bad()
# b.helper(5)                       # ❌ TypeError

# 错误 2：在 staticmethod 中用 self
class Bad2:
    n = 0
    @staticmethod
    def inc():
        # self.n += 1                # ❌ staticmethod 没有 self
        Bad2.n += 1                 # ✓ 显式用类名

# 错误 3：混淆 classmethod 和 staticmethod
class Confused:
    @classmethod                    # 想要静态，但写成 classmethod
    def helper(cls):
        return "hi"
# 这种情况下 helper(cls) 会传 cls，浪费
```

### 练习 2：3 种方法对比

```python
class DateUtils:
    @staticmethod
    def is_leap(year):
        return year % 4 == 0 and (year % 100 != 0 or year % 400 == 0)

    @classmethod
    def from_string(cls, date_str):
        y, m, d = map(int, date_str.split("-"))
        return cls(y, m, d)

print(DateUtils.is_leap(2024))    # True
d = DateUtils.from_string("2024-06-24")
```
