# tests/test_utils/test_cleaner.py
import pytest
import sys
sys.path.insert(0, "src")

from src.utils.cleaner import clean_html, normalize_url, parse_date

class TestCleaner:
    """清洗工具测试"""
    
    def test_clean_html(self):
        """测试 HTML 清理"""
        html = "<p>Hello <b>World</b></p>"
        result = clean_html(html)
        assert result == "Hello World"
    
    def test_clean_html_with_entities(self):
        """测试 HTML 实体"""
        html = "&lt;div&gt;Content&lt;/div&gt;"
        result = clean_html(html)
        assert "&lt;" not in result
    
    def test_clean_html_empty(self):
        """测试空输入"""
        assert clean_html("") == ""
        assert clean_html(None) == ""

class TestURLNormalizer:
    """URL 标准化测试"""
    
    def test_normalize_url_simple(self):
        """测试简单 URL"""
        url = "https://example.com/page"
        result = normalize_url(url)
        assert result == "https://example.com/page"
    
    def test_normalize_url_trailing_slash(self):
        """测试尾部斜杠"""
        url = "https://example.com/page/"
        result = normalize_url(url)
        assert result == "https://example.com/page"

class TestDateParser:
    """日期解析测试"""
    
    def test_parse_date_standard(self):
        """测试标准格式"""
        dt = parse_date("2026-06-24")
        assert dt.year == 2026
        assert dt.month == 6
        assert dt.day == 24
    
    def test_parse_date_chinese(self):
        """测试中文格式"""
        dt = parse_date("2026年06月24日")
        assert dt.year == 2026
    
    def test_parse_date_invalid(self):
        """测试无效日期"""
        assert parse_date("invalid") is None

## 二、Mock 对象

### 2.1 使用 unittest.mock

```python
import pytest
from unittest.mock import Mock, patch, MagicMock
import asyncio

class TestSpiderWithMock:
    """使用 Mock 测试爬虫"""
    
    @pytest.mark.asyncio
    async def test_fetch_success(self):
        """测试成功抓取"""
        from src.spiders.base import BaseSpider, SpiderConfig
        
        # 创建 Mock Session
        mock_response = Mock()
        mock_response.status = 200
        mock_response.text = asyncio.coroutine(lambda: "<html>Test</html>")
        
        mock_session = Mock()
        mock_session.get = Mock(return_value=mock_response.__aenter__)
        
        # 测试
        spider = BaseSpider(SpiderConfig(name="test", start_urls=[]))
        spider.session = mock_session
        
        result = await spider.fetch("https://example.com")
        assert result == "<html>Test</html>"
    
    @pytest.mark.asyncio
    async def test_fetch_with_404(self):
        """测试 404 响应"""
        mock_response = Mock()
        mock_response.status = 404
        
        mock_session = Mock()
        mock_session.get = Mock(return_value=mock_response.__aenter__)
        
        # 断言抛出异常
        with pytest.raises(Exception) as exc_info:
            await spider.fetch("https://example.com/notfound")
        
        assert "404" in str(exc_info.value)
```

### 2.2 patch 装饰器

```python
from unittest.mock import patch, MagicMock

class TestURLManager:
    """URL 管理器测试"""
    
    @patch("src.url_manager.redis.Redis")
    def test_add_url(self, mock_redis):
        """测试添加 URL"""
        from src.url_manager import RedisURLManager
        
        # Mock Redis 方法
        mock_instance = MagicMock()
        mock_instance.sismember.return_value = False
        mock_redis.return_value = mock_instance
        
        manager = RedisURLManager()
        result = manager.add_url("https://example.com")
        
        assert result is True
        mock_instance.lpush.assert_called_once()
    
    @patch("src.url_manager.redis.Redis")
    def test_add_duplicate_url(self, mock_redis):
        """测试添加重复 URL"""
        mock_instance = MagicMock()
        mock_instance.sismember.return_value = True
        mock_redis.return_value = mock_instance
        
        manager = RedisURLManager()
        result = manager.add_url("https://example.com")
        
        assert result is False
```

### 3.2 集成测试用例

```python
# tests/test_integration.py
import pytest

class TestCrawlerIntegration:
    """爬虫集成测试"""
    
    @pytest.mark.asyncio
    async def test_full_crawl_flow(self, url_manager, mongodb):
        """测试完整爬取流程"""
        from src.spiders.news import NewsSpider
        
        # 1. 添加种子 URL
        url_manager.add_url("https://news.example.com/test")
        
        # 2. 运行爬虫
        spider = NewsSpider()
        result = await spider.run()
        
        # 3. 验证结果
        assert result["success"] > 0
        assert result["errors"] == 0
        
        # 4. 验证数据存储
        articles = list(mongodb.articles.find())
        assert len(articles) > 0
    
    def test_url_deduplication(self, url_manager):
        """测试 URL 去重"""
        url = "https://example.com/article"
        
        # 添加两次
        result1 = url_manager.add_url(url)
        result2 = url_manager.add_url(url)
        
        assert result1 is True
        assert result2 is False
        
        # 验证只有一条
        assert url_manager.get_pending_count() == 1
```

## 五、CI/CD 集成

### 5.1 GitHub Actions

```yaml
# .github/workflows/test.yml
name: Test

on:
  push:
    branches: [ main, develop ]
  pull_request:
    branches: [ main ]

jobs:
  test:
    runs-on: ubuntu-latest
    
    services:
      redis:
        image: redis:7-alpine
        ports:
          - 6379:6379
      
      mongodb:
        image: mongo:6
        ports:
          - 27017:27017
    
    steps:
      - uses: actions/checkout@v3
      
      - name: Set up Python
        uses: actions/setup-python@v4
        with:
          python-version: '3.10'
      
      - name: Install dependencies
        run: |
          pip install -r requirements.txt
          pip install pytest pytest-asyncio pytest-cov
      
      - name: Run tests
        run: |
          pytest tests/ --cov=src --cov-report=xml
      
      - name: Upload coverage
        uses: codecov/codecov-action@v3
        with:
          file: ./coverage.xml
```

## 七、监控与日志

### 7.1 结构化日志

```python
import logging
import json
from datetime import datetime

class JSONFormatter(logging.Formatter):
    """JSON 日志格式"""
    
    def format(self, record):
        log_data = {
            "timestamp": datetime.now().isoformat(),
            "level": record.levelname,
            "logger": record.name,
            "message": record.getMessage(),
            "module": record.module,
            "function": record.funcName,
            "line": record.lineno
        }
        
        if record.exc_info:
            log_data["exception"] = self.formatException(record.exc_info)
        
        return json.dumps(log_data)

# 配置
handler = logging.StreamHandler()
handler.setFormatter(JSONFormatter())
logger = logging.getLogger("crawler")
logger.addHandler(handler)
```

### 7.2 健康检查

```python
# health.py
from flask import Flask, jsonify
import redis
from pymongo import MongoClient

app = Flask(__name__)

@app.route("/health")
def health():
    checks = {
        "redis": False,
        "mongodb": False
    }
    
    # 检查 Redis
    try:
        r = redis.Redis()
        r.ping()
        checks["redis"] = True
    except:
        pass
    
    # 检查 MongoDB
    try:
        client = MongoClient()
        client.admin.command("ping")
        checks["mongodb"] = True
    except:
        pass
    
    status = "healthy" if all(checks.values()) else "unhealthy"
    return jsonify({
        "status": status,
        "checks": checks
    }), 200 if status == "healthy" else 503

if __name__ == "__main__":
    app.run(host="0.0.0.0", port=8080)
```

## 九、易错点与避坑指南

1. **测试隔离**：每个测试用例应该独立
2. **Mock 过度**：不要过度依赖 Mock
3. **异步测试**：使用 pytest-asyncio
4. **覆盖率陷阱**：100% 覆盖率不等于无 bug
5. **环境差异**：开发/测试/生产环境要一致
