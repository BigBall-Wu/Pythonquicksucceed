## 四、重点详解

#### 题 8-13：用户简介（**kwargs）
```python
def build_profile(first, last, **user_info):
    """**user_info 收集任意关键字参数"""
    profile = {"first_name": first, "last_name": last}
    profile.update(user_info)
    return profile

me = build_profile("Alice", "Smith", location="Beijing", age=25, job="Engineer")
print(me)
# {'first_name': 'Alice', 'last_name': 'Smith', 'location': 'Beijing', 'age': 25, 'job': 'Engineer'}
```

#### 题 9-6：冰淇淋小店（继承）
```python
class Restaurant:
    def __init__(self, name, cuisine):
        self.name = name
        self.cuisine = cuisine
        self.number_served = 0

    def describe(self):
        print(f"{self.name} serves {self.cuisine}")

    def open(self):
        print(f"{self.name} is now open!")

    def set_number_served(self, n):
        self.number_served = n

    def increment_number_served(self, n):
        self.number_served += n

class IceCreamStand(Restaurant):
    """继承 Restaurant"""
    def __init__(self, name, flavors):
        super().__init__(name, "ice cream")
        self.flavors = flavors

    def show_flavors(self):
        print("Available flavors:")
        for flavor in self.flavors:
            print(f"- {flavor}")

stand = IceCreamStand("Cold Stone", ["vanilla", "chocolate", "strawberry"])
stand.open()
stand.show_flavors()
```
