```python
# 爬虫按实现技术分类：

| 方式 | 适用场景 | 性能 | 难度 |
|------|---------|------|------|
| requests + BeautifulSoup | 简单静态页面 | 中 | 低 |
| requests + lxml/XPath | 高效解析 | 中 | 中 |
| Scrapy | 大型项目、分布式 | 高 | 高 |
| Selenium | JS 渲染页面 | 慢 | 中 |
| Playwright | 现代浏览器自动化 | 较快 | 中 |
| pyppeteer | 无头 Chrome | 中 | 中 |
| Appium | 移动 App | 慢 | 高 |
```

```python
# 爬虫选型决策树

选型决策树 = """
页面是否需要 JS 渲染？
  ├─ 是 → Selenium / Playwright / pyppeteer
  └─ 否
      ├─ 项目规模？
      │   ├─ 小型（<1000页）→ requests + BeautifulSoup
      │   └─ 大型（>1000页）→ Scrapy
      └─ 反爬严重程度？
          ├─ 轻微 → requests + User-Agent
```

### 第 4 步：聚焦爬虫实战（15 分钟）

```python
import requests
from bs4 import BeautifulSoup

def focused_crawler(category_url, max_pages=5):
    """只爬指定分类的页面"""
    results = []
    for page in range(1, max_pages + 1):
        url = f"{category_url}?page={page}"
        resp = requests.get(
            url,
            headers={"User-Agent": "Mozilla/5.0"}
        )
        soup = BeautifulSoup(resp.text, "lxml")
        for item in soup.select(".item"):
            results.append({
                "title": item.select_one(".title").text,
                "link": item.select_one("a")["href"],
                "price": item.select_one(".price").text,
            })
    return results

# 使用
results = focused_crawler("https://example.com/category/books")
print(f"爬取了 {len(results)} 条数据")
```

### 第 5 步：增量爬虫实战（15 分钟）

```python
import os
import json
import hashlib
import requests

class IncrementalCrawler:
    """增量爬虫：只爬新内容"""

    def __init__(self, state_file="crawled.json"):
        self.state_file = state_file
        self.seen = self._load_state()

    def _load_state(self):
        if os.path.exists(self.state_file):
            with open(self.state_file) as f:
                return json.load(f)
        return {}

    def _save_state(self):
        with open(self.state_file, "w") as f:
            json.dump(self.seen, f)

    def is_new(self, url, content):
        """检查内容是否是新内容"""
        h = hashlib.md5(content.encode()).hexdigest()
        return self.seen.get(url) != h

    def save(self, url, content):
        """保存 URL 和内容哈希"""
        self.seen[url] = hashlib.md5(content.encode()).hexdigest()
        self._save_state()

# 使用
crawler = IncrementalCrawler("news_crawled.json")
resp = requests.get("https://example.com/news")

if crawler.is_new("https://example.com/news", resp.text):
    crawler.save("https://example.com/news", resp.text)
    print("发现新内容，处理中...")
    # process_news(resp.text)
else:
    print("内容未变化，跳过")
```
