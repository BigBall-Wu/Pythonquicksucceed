### 第 1 步：`__call__` 让对象可调用（10 分钟）

```python
class Multiplier:
    """乘法器"""
    def __init__(self, factor):
        self.factor = factor

    def __call__(self, x):
        return x * self.factor

double = Multiplier(2)
triple = Multiplier(3)

print(double(5))                  # 10（对象像函数一样被调用）
print(triple(5))                  # 15

# 用作装饰器
class CountCalls:
    def __init__(self, func):
        self.func = func
        self.count = 0

    def __call__(self, *args, **kwargs):
        self.count += 1
        return self.func(*args, **kwargs)

@CountCalls
def greet(name):
    print(f"Hello, {name}")

greet("Alice")
greet("Bob")
print(greet.count)                # 2
```

### 第 2 步：`__iter__` / `__next__` 让类可迭代（10 分钟）

```python
class Countdown:
    """倒计时迭代器"""
    def __init__(self, start):
        self.current = start

    def __iter__(self):
        return self                  # 返回迭代器（自己）

    def __next__(self):
        if self.current <= 0:
            raise StopIteration
        self.current -= 1
        return self.current + 1

# 用 for 循环
for n in Countdown(5):
    print(n, end=" ")              # 5 4 3 2 1
print()
```

### 第 3 步：`__enter__` / `__exit__` 实现 with（10 分钟）

```python
class Timer:
    """计时器上下文管理器"""
    def __enter__(self):
        import time
        self.start = time.time()
        return self                  # 返回值给 as 后的变量

    def __exit__(self, exc_type, exc_val, exc_tb):
        import time
        elapsed = time.time() - self.start
        print(f"耗时：{elapsed:.3f}s")
        return False                 # 不抑制异常

# 用
with Timer():
    sum(range(10_000_000))
# 耗时：0.05s
```

### 第 4 步：`__getattr__` / `__setattr__` 控制属性访问（10 分钟）

```python
class DynamicAttributes:
    """动态属性（所有不存在的属性都返回 None）"""

    def __getattr__(self, name):
        """访问不存在的属性时调用"""
        return f"<{name} 不存在>"

    def __setattr__(self, name, value):
        """设置属性时调用"""
        if name.startswith("_"):
            super().__setattr__(name, value)   # 私有属性正常存
        else:
            print(f"设置 {name} = {value}")
            super().__setattr__(name, value)

obj = DynamicAttributes()
print(obj.foo)                     # <foo 不存在>
print(obj.bar)                     # <bar 不存在>
obj.x = 10                         # 设置 x = 10
```

### 第 5 步：实战 - 链式调用类（10 分钟）

```python
class Query:
    """SQL 查询构建器（链式调用）"""
    def __init__(self):
        self.conditions = []
        self.fields = "*"
        self.table = ""

    def select(self, *fields):
        self.fields = ", ".join(fields)
        return self                  # 返回 self 实现链式

    def from_table(self, table):
        self.table = table
        return self

    def where(self, condition):
        self.conditions.append(condition)
        return self

    def __str__(self):
        sql = f"SELECT {self.fields} FROM {self.table}"
        if self.conditions:
            sql += " WHERE " + " AND ".join(self.conditions)
        return sql

# 链式调用
q = (Query()
     .select("id", "name", "age")
     .from_table("users")
     .where("age > 18")
     .where("status = 'active'"))
print(q)
# SELECT id, name, age FROM users WHERE age > 18 AND status = 'active'
```

### 第 6 步：`__contains__` 和 `__getitem__`（10 分钟）

```python
class Container:
    """自定义容器"""
    def __init__(self, items):
        self.items = list(items)

    def __contains__(self, item):
        """支持 in 操作符"""
        return item in self.items

    def __getitem__(self, index):
        """支持下标访问"""
        return self.items[index]

    def __len__(self):
        return len(self.items)

    def __iter__(self):
        return iter(self.items)

c = Container([1, 2, 3, 4, 5])
print(3 in c)                       # True（用 __contains__）
print(c[2])                        # 3（用 __getitem__）
print(len(c))                      # 5（用 __len__）
for item in c:                     # 用 __iter__
    print(item)
```

### 第 7 步：常见错误（5 分钟）

```python
# 错误 1：`__iter__` 不返回迭代器
class Bad:
    def __iter__(self):
        return self                  # ❌ 应返回新的迭代器对象
# for 会无限循环

# 正确：返回独立的迭代器对象
class Good:
    def __iter__(self):
        return GoodIterator(self.items)

class GoodIterator:
    def __init__(self, items):
        self.items = items
        self.i = 0

    def __next__(self):
        if self.i >= len(self.items):
            raise StopIteration
        v = self.items[self.i]
        self.i += 1
        return v

# 错误 2：`__enter__` 忘 return self
class BadTimer:
    def __enter__(self):
        pass                          # ❌ 没返回值
    def __exit__(self, *args):
        pass

# with BadTimer() as t:           # t 是 None
#     t.something()                # ❌ AttributeError
```

### 练习 2：自定义范围

```python
class MyRange:
    """类似 range 但可自定义步长"""
    def __init__(self, start, stop=None, step=1):
        if stop is None:
            stop = start
            start = 0
        self.start = start
        self.stop = stop
        self.step = step

    def __iter__(self):
        n = self.start
        while n < self.stop:
            yield n
        n += self.step

for i in MyRange(0, 10, 2):
    print(i, end=" ")
```
