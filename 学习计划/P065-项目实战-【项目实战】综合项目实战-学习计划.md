## 一、项目需求

### 1.1 项目背景

我们需要构建一个**新闻聚合爬虫系统**，能够：
- 从多个新闻源自动抓取新闻
- 支持增量更新和全文检索
- 提供 RESTful API 接口
- 支持分布式部署

### 1.2 功能需求

| 模块 | 功能 |
|------|------|
| 爬虫模块 | 多源爬取、自动去重、增量更新 |
| 存储模块 | MongoDB 存储、Elasticsearch 索引 |
| API 模块 | 新闻搜索、分页、分类 |
| 调度模块 | 定时任务、分布式调度 |
| 监控模块 | 状态监控、告警通知 |

### 1.3 技术栈

```yaml
爬虫: Scrapy-Redis + BeautifulSoup + Selenium
存储: MongoDB + Elasticsearch
队列: Redis
API: FastAPI
部署: Docker + Docker Compose
监控: Prometheus + Grafana
```

### 3.1 新闻数据模型

```python
# src/models/news.py
from dataclasses import dataclass, field
from datetime import datetime
from typing import Optional, List

@dataclass
class NewsItem:
    """新闻数据模型"""
    title: str = ""
    url: str = ""
    source: str = ""           # 来源：sina、tencent、toutiao
    author: Optional[str] = ""
    publish_date: Optional[datetime] = None
    content: str = ""
    summary: str = ""
    category: str = ""
    tags: List[str] = field(default_factory=list)
    images: List[str] = field(default_factory=list)
    created_at: datetime = field(default_factory=datetime.now)
    updated_at: datetime = field(default_factory=datetime.now)
    
    @property
    def fingerprint(self) -> str:
        """生成指纹用于去重"""
        import hashlib
        return hashlib.md5(self.url.encode()).hexdigest()
    
    def to_dict(self) -> dict:
        """转换为字典"""
        return {
            "title": self.title,
            "url": self.url,
            "source": self.source,
            "author": self.author,
            "publish_date": self.publish_date.isoformat() if self.publish_date else None,
            "content": self.content,
            "summary": self.summary,
            "category": self.category,
            "tags": self.tags,
            "images": self.images,
            "created_at": self.created_at.isoformat(),
            "updated_at": self.updated_at.isoformat(),
            "fingerprint": self.fingerprint
        }
```

### 3.3 新浪爬虫实现

```python
# src/spiders/sina.py
import scrapy
from datetime import datetime
from src.spiders.base import BaseNewsSpider
from src.models.news import NewsItem

class SinaNewsSpider(BaseNewsSpider):
    """新浪新闻爬虫"""
    
    name = "sina_news"
    allowed_domains = ["news.sina.com.cn"]
    
    def start_requests(self):
        """生成起始请求"""
        categories = ["news", "finance", "tech", "sports", "ent"]
        
        for category in categories:
            url = f"https://news.sina.com.cn/{category}/"
            yield scrapy.Request(url, self.parse_list)
    
    def parse_list(self, response):
        """解析新闻列表"""
        for item in response.css("ul.news-list li"):
            link = item.css("a::attr(href)").get()
            if link:
                yield response.follow(link, self.parse_news)
    
    def parse_news(self, response):
        """解析新闻详情"""
        item = NewsItem()
        item.url = response.url
        item.source = "sina"
        item.title = response.css("h1.main-title::text").get() or ""
        item.author = response.css("span.author::text").get() or ""
        
        date_str = response.css("span.date::text").get()
        if date_str:
            item.publish_date = datetime.strptime(date_str.strip(), "%Y年%m月%d日 %H:%M")
        
        content_parts = response.css("article p::text").getall()
        item.content = "".join(content_parts).strip()
        
        item.summary = item.content[:200] if item.content else ""
        
        item.category = self.get_category(response.url)
        
        yield item
    
    def get_category(self, url):
        """从 URL 提取分类"""
        parts = url.split("/")
        for part in parts:
            if part in ["news", "finance", "tech", "sports", "ent"]:
                return part
        return "other"
```

### 3.4 数据清洗 Pipeline

```python
# src/pipelines/cleaner.py
import scrapy
from itemadapter import ItemAdapter
from src.models.news import NewsItem

class NewsCleanerPipeline:
    """新闻数据清洗"""
    
    def process_item(self, item, spider):
        if isinstance(item, dict):
            item = NewsItem(**item)
        
        adapter = ItemAdapter(item)
        
        # 清理标题
        title = adapter.get("title", "")
        adapter["title"] = self.clean_text(title)
        
        # 清理内容
        content = adapter.get("content", "")
        adapter["content"] = self.clean_text(content)
        
        # 生成摘要
        if not adapter.get("summary") and content:
            adapter["summary"] = content[:200]
        
        # 清理标签
        tags = adapter.get("tags", [])
        if isinstance(tags, str):
            tags = [t.strip() for t in tags.split(",")]
        adapter["tags"] = tags[:10]  # 最多10个标签
        
        return adapter.item
    
    def clean_text(self, text):
        """清理文本"""
        if not text:
            return ""
        
        import re
        # 移除多余空白
        text = re.sub(r"\s+", " ", text)
        # 移除特殊字符
        text = text.strip()
        
        return text
```

### 4.2 搜索接口

```python
# src/api/routes/search.py
from fastapi import APIRouter, Query, HTTPException
from typing import Optional
from datetime import datetime

from src.api.schemas import SearchRequest, SearchResponse, NewsItem

router = APIRouter(prefix="/search", tags=["搜索"])

@router.post("/", response_model=SearchResponse)
async def search_news(
    keyword: str = Query(..., min_length=1, max_length=100),
    category: Optional[str] = None,
    source: Optional[str] = None,
    page: int = Query(1, ge=1),
    page_size: int = Query(20, ge=1, le=100)
):
    """搜索新闻"""
    from src.pipelines.elasticsearch import ESClient
    
    es = ESClient()
    
    query = {"match": {"content": keyword}}
    
    filters = []
    if category:
        filters.append({"term": {"category": category}})
    if source:
        filters.append({"term": {"source": source}})
    
    result = es.search(
        query=query,
        filters=filters,
        from_=(page - 1) * page_size,
        size=page_size
    )
    
    return SearchResponse(
        total=result["total"],
        page=page,
        page_size=page_size,
        items=[NewsItem(**item) for item in result["items"]]
    )

@router.get("/suggestions")
async def get_suggestions(keyword: str = Query(...)):
    """获取搜索建议"""
    from src.utils.redis_client import RedisClient
    
    r = RedisClient()
    suggestions = r.get_suggestions(keyword)
    return {"suggestions": suggestions}
```
