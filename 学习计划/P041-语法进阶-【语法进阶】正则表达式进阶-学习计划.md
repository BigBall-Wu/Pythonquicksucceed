## 一、零宽断言（Lookahead/Lookbehind）

### 1.1 什么是零宽断言

零宽断言是一种特殊的位置匹配，它匹配一个位置，但这个位置本身不占用任何字符宽度（零宽度）。

```
普通字符匹配：  a    b    c    d
              ^^^^    （匹配 4 个字符）

零宽断言：     (?=x)  （只匹配一个位置）
```

### 1.2 四种断言类型

```python
import re

text = "Hello123World456"

# (?=...) 正前瞻：后面是 ...
# 找出后面跟着数字的 "Hello"
result = re.findall(r"Hello(?=\d)", text)
print(result)  # ['Hello']

# (?!...) 负前瞻：后面不是 ...
# 找出后面不跟着数字的 "Hello"
result = re.findall(r"Hello(?!\d)", "HelloWorld")
print(result)  # ['Hello']
result = re.findall(r"Hello(?!\d)", "Hello123")
print(result)  # []

# (?<=...) 正后顾：前面是 ...
# 找出前面是数字的 "World"
result = re.findall(r"(?<=\d)World", text)
print(result)  # ['World']

# (?<!...) 负后顾：前面不是 ...
# 找出前面不是数字的 "World"
result = re.findall(r"(?<!\d)World", "HelloWorld")
print(result)  # ['World']
result = re.findall(r"(?<!\d)World", "123World")
print(result)  # []
```

### 1.3 断言实战应用

#### 应用一：密码验证

```python
import re

def validate_password(password):
    """
    密码规则：
    1. 长度至少8位
    2. 包含数字
    3. 包含小写字母
    4. 包含大写字母
    """
    if len(password) < 8:
        return False, "密码长度至少8位"
    
    if not re.search(r"\d", password):
        return False, "密码必须包含数字"
    
    if not re.search(r"[a-z]", password):
        return False, "密码必须包含小写字母"
    
    if not re.search(r"[A-Z]", password):
        return False, "密码必须包含大写字母"
    
    return True, "密码格式正确"

# 测试
print(validate_password("Pass1234"))  # (True, '密码格式正确')
print(validate_password("pass"))       # (False, '密码长度至少8位')
print(validate_password("PASSWORD123"))# (False, '密码必须包含小写字母')
```

#### 应用二：提取特定内容

```python
import re

# 提取价格（不包括货币符号）
text = "苹果价格：¥5.99，香蕉价格：$2.99"
prices = re.findall(r"(?<=[¥$])\d+\.?\d*", text)
print(prices)  # ['5.99', '2.99']

# 提取 HTML 属性值
html = '<input type="text" name="username" value="admin">'
values = re.findall(r'(?<=\bvalue=")[^"]+', html)
print(values)  # ['admin']

# 提取被引号包围的内容（不使用捕获组）
text = '他说"你好"并说"再见"'
quoted = re.findall(r'(?<=")[^"]+(?=")', text)
print(quoted)  # ['你好', '再见']
```

## 二、贪婪与非贪婪深度理解

### 2.1 匹配原理

```python
import re

text = "<div>内容1</div><div>内容2</div>"

# 贪婪：尽可能多地匹配
result = re.search(r"<div>.*</div>", text)
print(f"贪婪：{result.group()}")
# 输出：<div>内容1</div><div>内容2</div>（全部）

# 非贪婪：尽可能少地匹配
result = re.search(r"<div>.*?</div>", text)
print(f"非贪婪：{result.group()}")
# 输出：<div>内容1</div>（只匹配第一个）
```

### 2.2 量词的贪婪性排名

```python
import re

text = "123456789"

# 贪婪程度从高到低：
print(re.findall(r"\d{3,5}", text))   # 贪婪 ['12345', '6789'] 先取5个
print(re.findall(r"\d{3,5}?", text))  # 非贪婪 ['123', '456', '789'] 先取3个

# * 等价于 {0,}
print(re.findall(r"\d*", text))      # 贪婪 ['123456789', '']
print(re.findall(r"\d*?", text))     # 非贪婪 ['', '', '', '', '', '', '', '', '', '']

# + 等价于 {1,}
print(re.findall(r"\d+", text))      # 贪婪 ['123456789']
print(re.findall(r"\d+?", text))     # 非贪婪 ['1', '2', '3', '4', '5', '6', '7', '8', '9']

# ? 单独使用
print(re.findall(r"\d??", text))     # 非贪婪优先空字符串 ['', '', '', '', '', '', '', '', '', '']
```

### 2.3 何时用贪婪 vs 非贪婪

```python
import re

# 场景一：提取标签内容
html = "<span>文本</span><span>文本2</span>"

# 贪婪：会匹配到最后一个 </span>
result = re.search(r"<span>.*</span>", html)
print(result.group() if result else None)

# 非贪婪：匹配到第一个 </span>
result = re.search(r"<span>.*?</span>", html)
print(result.group())

# 场景二：提取引号内容
text = '"内容1"和"内容2"和"内容3"'

# 贪婪：会匹配 "内容1"和"内容2"和"内容3"
result = re.search(r'"[^"]*"', text)
print(result.group())

# 非贪婪：匹配 "内容1"
result = re.search(r'"[^"]*?"', text)
print(result.group())
```

## 三、环视（Lookaround）高级用法

### 3.1 交叉验证

```python
import re

# 验证密码：必须同时包含字母和数字
def validate_password_lookaround(password):
    # 必须有字母，且有数字
    pattern = r"^(?=.*[a-zA-Z])(?=.*\d).{8,}$"
    return bool(re.match(pattern, password))

# 解释：(?=.*[a-zA-Z]) 前瞻检查：后面任意位置有字母
#       (?=.*\d) 前瞻检查：后面任意位置有数字
#       . {8,} 匹配至少8个任意字符

print(validate_password_lookaround("Pass1234"))  # True
print(validate_password_lookaround("Password"))  # False（无数字）
print(validate_password_lookaround("12345678"))  # False（无字母）

# 验证用户名：必须包含字母、数字和下划线
def validate_username(username):
    pattern = r"^(?=.*[a-zA-Z])(?=.*\d)(?=.*_)\w{6,20}$"
    return bool(re.match(pattern, username))

print(validate_username("user_123"))    # True
print(validate_username("user123"))     # False（无下划线）
```

### 3.2 条件匹配

```python
import re

# 只在特定条件下匹配
text = "版本1.0\n版本2.0\n草稿\n最终版"

# 匹配版本号（非草稿状态）
versions = re.findall(r"^(?!草稿).*版本\d+\.\d+", text, re.MULTILINE)
print(versions)  # ['版本1.0', '版本2.0', '最终版']

# 排除特定模式
numbers = "123abc456def789"
# 不以 789 结尾的数字
result = re.findall(r"\d+(?!789)", numbers)
print(result)  # ['123', '456']

# 只匹配不在引号内的内容
text = "链接：https://a.com 描述：访问链接 链接：https://b.com"
# 提取不在引号内的 URL
result = re.findall(r"(?<![\"'])(https?://\S+)(?![\"'])", text)
print(result)
```

### 3.3 高级分割

```python
import re

# 按特定分隔符分割，但保留分隔符
text = "a+b-c*d/e"

# 分割并保留运算符
result = re.split(r"(?<=\D)(?=\d)|(?<=\d)(?=\D)", text)
print(result)  # ['a', '+', 'b', '-', 'c', '*', 'd', '/', 'e']

# 驼峰转下划线
text = "camelCase toSnakeCase HTTPConnection"
result = re.sub(r"([a-z])([A-Z])", r"\1_\2", text).lower()
print(result)  # camel_case to_snake_case h_t_t_p_connection

# 首字母大写转下划线（但保留连续大写）
def camel_to_snake(s):
    s = re.sub(r"([A-Z]+)([A-Z][a-z])", r"\1_\2", s)
    s = re.sub(r"([a-z])([A-Z])", r"\1_\2", s)
    return s.lower()

print(camel_to_snake("HTTPSConnection"))  # h_t_t_p_s_connection
```

## 四、实战技巧

### 4.1 去除HTML标签与内容

```python
import re

# 移除 script 和 style 标签及其内容
html = """
<script>alert('xss');</script>
<style>.red{color:red;}</style>
<div>可见内容</div>
<script>evil();</script>
"""

# 方法一：移除 script/style 标签及内容
clean = re.sub(r"<script[\s\S]*?</script>", "", html)
clean = re.sub(r"<style[\s\S]*?</style>", "", clean)
print(clean.strip())

# 方法二：移除所有标签，保留文本
text = re.sub(r"<[^>]+>", "", html)
print(text.strip())

# 方法三：移除注释
clean = re.sub(r"<!--[\s\S]*?-->", "", html)
print(clean.strip())
```

### 4.2 敏感信息脱敏

```python
import re

def mask_phone(phone):
    """手机号中间4位脱敏"""
    return re.sub(r"(\d{3})\d{4}(\d{4})", r"\1****\2", phone)

def mask_email(email):
    """邮箱脱敏"""
    return re.sub(r"(\w{2})\w+(@\w+\.\w+)", r"\1***\2", email)

def mask_id_card(id_card):
    """身份证中间生日部分脱敏"""
    return re.sub(r"(\d{6})\d{8}(\d{4})", r"\1********\2", id_card)

def mask_bank_card(card):
    """银行卡号脱敏（保留后4位）"""
    return re.sub(r"\d(?=\d{4})", "*", card)

# 测试
print(mask_phone("13812345678"))       # 138****5678
print(mask_email("user@example.com"))   # us***@example.com
print(mask_id_card("11010119900101123X")) # 110101********0123
print(mask_bank_card("6222021234567890")) # 6222**********7890
```

### 4.3 数据格式标准化

```python
import re

def normalize_date(text):
    """将各种日期格式统一为 YYYY-MM-DD"""
    # 2026/06/24 或 2026-06-24 或 2026.06.24
    text = re.sub(r"(\d{4})[/\-.](\d{1,2})[/\-.](\d{1,2})", r"\1-\2-\3", text)
    # 06/24/2026 → 2026-06-24
    text = re.sub(r"(\d{1,2})/(\d{1,2})/(\d{4})", r"\3-\1-\2", text)
    return text

def normalize_phone(text):
    """统一手机号格式"""
    # 移除所有非数字字符
    digits = re.sub(r"\D", "", text)
    # 格式化
    if len(digits) == 11:
        return f"{digits[:3]}-{digits[3:7]}-{digits[7:]}"
    return text

def normalize_url(text):
    """标准化 URL"""
    # 移除末尾斜杠
    text = re.sub(r"/+$", "", text)
    # 移除 www.
    text = re.sub(r"^https?://www\.", r"https://", text)
    return text

# 测试
print(normalize_date("2026/06/24"))       # 2026-06-24
print(normalize_phone("138-1234-5678"))    # 138-1234-5678
print(normalize_url("https://www.example.com/"))  # https://example.com
```

### 4.4 文本提取增强

```python
import re

def extract_chinese(text):
    """提取纯中文内容"""
    return "".join(re.findall(r"[\u4e00-\u9fa5]+", text))

def extract_numbers_with_unit(text):
    """提取带单位的数字"""
    pattern = r"(\d+\.?\d*)\s*(GB|MB|KB|TB|米|厘米|千克|克|公里|平方米|%)"
    return re.findall(pattern, text)

def extract_citations(text):
    """提取论文引用"""
    # [1] 或 [1,2,3] 格式
    citations = re.findall(r"\[(\d+(?:,\s*\d+)*)\]", text)
    # 展开为独立编号
    all_citations = set()
    for c in citations:
        all_citations.update(int(x.strip()) for x in c.split(","))
    return sorted(all_citations)

def extract_code_blocks(text):
    """提取代码块（Markdown 格式）"""
    return re.findall(r"```(?:\w+)?\n([\s\S]*?)```", text)

# 测试
print(extract_chinese("Hello你好World世界123"))  # 你好世界
print(extract_numbers_with_unit("内存8GB，硬盘512GB，重量1.2kg"))
print(extract_citations("参考文献[1]和[2,3,5]提及..."))  # [1, 2, 3, 5]
```

## 五、性能优化

### 5.1 预编译正则

```python
import re
import time

# 未优化：每次调用都编译
text = "test" * 10000 + "error"
start = time.time()
for _ in range(1000):
    re.search(r"error", text)
print(f"未预编译：{time.time() - start:.4f}秒")

# 优化：预编译
pattern = re.compile(r"error")
start = time.time()
for _ in range(1000):
    pattern.search(text)
print(f"预编译：{time.time() - start:.4f}秒")
```

### 5.2 使用合适的方法

```python
import re

text = "hello world hello python hello"

# search vs match
re.search(r"hello", text)  # 搜索任意位置
re.match(r"hello", text)   # 只匹配开头

# findall vs finditer（大数据时）
# findall 返回列表，finditer 返回迭代器（更省内存）
result = re.finditer(r"\w+", text)
for match in result:
    print(match.group())

# sub 的回调优化
def expensive_replace(match):
    # 复杂处理
    return match.group().upper()

# ✅ 高效
re.sub(r"\w+", expensive_replace, text)

# ❌ 低效（会重复编译）
re.sub(r"\w+", lambda m: expensive_replace(m), text)
```

### 实战：数据清洗管道

```python
import re
from typing import List, Dict, Callable

class DataCleaner:
    """数据清洗管道"""
    
    def __init__(self):
        self.pipelines: List[Callable] = []
    
    def add_step(self, func: Callable) -> "DataCleaner":
        """添加清洗步骤"""
        self.pipelines.append(func)
        return self
    
    def clean(self, data: str) -> str:
        """执行所有清洗步骤"""
        result = data
        for step in self.pipelines:
            result = step(result)
        return result
    
    @staticmethod
    def remove_html():
        return lambda x: re.sub(r"<[^>]+>", "", x)
    
    @staticmethod
    def remove_urls():
        return lambda x: re.sub(r"https?://\S+", "[URL]", x)
    
    @staticmethod
    def remove_emails():
        return lambda x: re.sub(r"\w+@\w+\.\w+", "[EMAIL]", x)
    
    @staticmethod
    def normalize_whitespace():
        return lambda x: re.sub(r"\s+", " ", x).strip()
    
    @staticmethod
    def remove_special_chars(keep_chinese=True):
        def func(x):
            if keep_chinese:
                return re.sub(r"[^\u4e00-\u9fa5a-zA-Z0-9\s]", "", x)
            return re.sub(r"\W", "", x)
        return func
    
    @staticmethod
    def mask_phone():
        return lambda x: re.sub(r"(\d{3})\d{4}(\d{4})", r"\1****\2", x)
    
    @staticmethod
    def mask_email():
        return lambda x: re.sub(r"(\w{2})\w+(@\w+)", r"\1***\2", x)

# 使用示例
cleaner = DataCleaner()
cleaner.add_step(DataCleaner.remove_html())
cleaner.add_step(DataCleaner.remove_urls())
cleaner.add_step(DataCleaner.remove_emails())
cleaner.add_step(DataCleaner.normalize_whitespace())

raw_data = """
<div>联系方式：138-1234-5678</div>
邮箱：user@example.com
访问：https://site.com
价格：$99.99
"""

cleaned = cleaner.clean(raw_data)
print(cleaned)
```

## 八、易错点与避坑指南

1. **后顾断言限制**：Python 中 `(?<=...)` 必须定长
2. **回溯陷阱**：复杂正则可能导致灾难性回溯
3. **Unicode 中文**：`[\u4e00-\u9fa5]` 匹配简体中文
4. **DOTALL 模式**：`[\s\S]*` 替代 `.*` 以匹配换行
5. **编译缓存**：避免在循环内重复编译相同正则
