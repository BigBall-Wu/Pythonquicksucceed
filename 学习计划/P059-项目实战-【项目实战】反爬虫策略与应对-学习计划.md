## 三、IP 限制应对

### 3.1 代理 IP 池

```python
import asyncio
import aiohttp

class ProxyPool:
    """代理 IP 池"""
    
    def __init__(self):
        self.proxies = []
        self.current_index = 0
    
    def add_proxy(self, proxy):
        """添加代理"""
        self.proxies.append(proxy)
    
    def get_proxy(self):
        """获取代理（轮询）"""
        if not self.proxies:
            return None
        proxy = self.proxies[self.current_index]
        self.current_index = (self.current_index + 1) % len(self.proxies)
        return proxy
    
    async def fetch_with_proxy(self, session, url):
        """使用代理请求"""
        proxy = self.get_proxy()
        if proxy:
            async with session.get(url, proxy=f"http://{proxy}") as response:
                return await response.text()
        return await session.get(url).text()

# 使用代理池
proxy_pool = ProxyPool()
proxy_pool.add_proxy("user:password@proxy1.example.com:8080")
proxy_pool.add_proxy("user:password@proxy2.example.com:8080")
```

### 3.3 代理质量检测

```python
import asyncio
import aiohttp

async def check_proxy(proxy, test_url="https://httpbin.org/ip"):
    """检测代理是否可用"""
    try:
        async with aiohttp.ClientSession() as session:
            async with session.get(test_url, proxy=proxy, timeout=10) as response:
                if response.status == 200:
                    return True
    except:
        pass
    return False

async def get_valid_proxies(proxies, max_valid=10):
    """获取有效代理列表"""
    valid_proxies = []
    
    for proxy in proxies:
        if await check_proxy(proxy):
            valid_proxies.append(proxy)
            if len(valid_proxies) >= max_valid:
                break
    
    return valid_proxies
```

### 4.2 打码平台

```python
import base64
import requests

class CaptchaSolver:
    """打码平台接口"""
    
    def __init__(self, api_key, api_url):
        self.api_key = api_key
        self.api_url = api_url
    
    def solve(self, image_bytes):
        """提交验证码并获取结果"""
        # 提交
        data = {
            "api_key": self.api_key,
            "image": base64.b64encode(image_bytes).decode()
        }
        
        response = requests.post(self.api_url + "/solve", json=data)
        result = response.json()
        
        if result["success"]:
            return result["result"]
        else:
            raise Exception(f"打码失败: {result['error']}")
    
    def report_error(self, captcha_id):
        """反馈错误"""
        data = {
            "api_key": self.api_key,
            "captcha_id": captcha_id
        }
        requests.post(self.api_url + "/report", json=data)
```

```python
import random

async def random_delay(min_sec=1, max_sec=3):
    """随机延迟，模拟人类操作"""
    delay = random.uniform(min_sec, max_sec)

async def random_waits(urls):
```

```python
import random
import time

def generate_human_track(start_x, start_y, end_x, end_y):
    """
    生成人类鼠标轨迹
    包含抖动和速度变化
    """
    # 贝塞尔曲线控制点
    control_points = []
    
    # 起始点
    x, y = start_x, start_y
    target_x, target_y = end_x, end_y
    
    # 添加一些随机中间点
    for i in range(random.randint(3, 7)):
        x += (target_x - start_x) / random.randint(2, 4) + random.randint(-20, 20)
        y += (target_y - start_y) / random.randint(2, 4) + random.randint(-20, 20)
        control_points.append((x, y))
    
    track = []
    # 生成轨迹点
    for i in range(100):
        t = i / 100
        point = bezier_curve(start_x, start_y, target_x, target_y, control_points, t)
        track.append(point)
        time.sleep(random.uniform(0.005, 0.015))
    
    return track

def bezier_curve(x0, y0, x1, y1, control_points, t):
    """贝塞尔曲线计算"""
    # 简化版本
    x = x0 + (x1 - x0) * t
    y = y0 + (y1 - y0) * t
    return x + random.randint(-2, 2), y + random.randint(-2, 2)
```

## 七、综合实战

### 7.1 反反爬中间件

```python
import random
from dataclasses import dataclass, field
from typing import List

from src.middlewares.base import BaseMiddleware

@dataclass
class AntiBanConfig:
    """反反爬配置"""
    user_agents: List[str] = field(default_factory=list)
    proxies: List[str] = field(default_factory=list)
    use_delay: bool = True
    min_delay: float = 1.0
    max_delay: float = 3.0
    use_cookie: bool = True
    use_proxy: bool = False

class AntiBanMiddleware(BaseMiddleware):
    """反反爬中间件"""
    
    def __init__(self, config: AntiBanConfig):
        self.config = config
        self.ua_index = 0
        self.proxy_index = 0
    
    def get_random_ua(self):
        if not self.config.user_agents:
            return "Mozilla/5.0"
        self.ua_index = (self.ua_index + 1) % len(self.config.user_agents)
        return self.config.user_agents[self.ua_index]
    
    def get_proxy(self):
        if not self.config.proxies or not self.config.use_proxy:
            return None
        self.proxy_index = (self.proxy_index + 1) % len(self.config.proxies)
        return self.config.proxies[self.proxy_index]
    
    async def before_request(self, request):
        """请求前处理"""
        # 设置 User-Agent
        request.headers["User-Agent"] = self.get_random_ua()
        
        # 设置代理
        proxy = self.get_proxy()
        
        # 随机延迟
        if self.config.use_delay:
            await asyncio.sleep(random.uniform(
                self.config.min_delay,
                self.config.max_delay
            ))
        
        return request, proxy
    
    async def after_response(self, response):
        """响应后处理"""
        # 检查是否被封
        if response.status == 403:
            raise Exception("IP 被封禁")
        if "验证码" in response.text:
            raise Exception("需要验证码")
        
        return response
```

## 九、易错点与避坑指南

1. **代理失效**：定期检测代理有效性
2. **请求过快**：合理设置延迟
3. **检测模式**：单一策略容易被检测，组合使用
4. **验证码复杂度**：简单 OCR 可能无法识别
5. **法律风险**：遵守网站 robots.txt 和使用条款
