## 一、函数递归深度解析

### 1.1 什么是递归

递归就是函数调用自身的过程。想象一下俄罗斯套娃，每个娃娃里面还有一个更小的娃娃，直到最小的那个无法再拆开。

```python
# 最简单的递归：倒计时
def countdown(n):
    if n <= 0:
        print("发射！")
        return
    print(n)
    countdown(n - 1)  # 调用自身

countdown(5)
# 输出：5 4 3 2 1 发射！
```

### 1.2 递归的三个必要条件

#### 条件一：基本情形（Base Case）

没有基本情形，递归会无限进行下去，直到栈溢出。

```python
# ❌ 错误：没有基本情形
def infinite_recursion():
    return infinite_recursion()  # 永远调用自身

# ✅ 正确：有基本情形
def proper_recursion(n):
    if n == 0:  # 基本情形
        return 0
    return n + proper_recursion(n - 1)
```

#### 条件二：递归情形（Recursive Case）

函数在某种条件下调用自身。

```python
def factorial(n):
    if n <= 1:          # 基本情形
        return 1
    return n * factorial(n - 1)  # 递归情形

print(factorial(5))  # 120
```

#### 条件三：向基本情形靠近

每次递归都要使参数更接近基本情形。

```python
# 正确：每次 n-1，更接近 0
def countdown(n):
    if n <= 0:
        return
    countdown(n - 1)  # ✅ n 减小

# ❌ 错误：n 不变，永远无法到达基本情形
def broken_countdown(n):
    if n <= 0:
        return
    broken_countdown(n)  # ❌ n 不变，死循环
```

### 1.3 递归执行过程详解

```python
def factorial(n):
    print(f"进入 factorial({n})")
    if n == 1:
        result = 1
    else:
        result = n * factorial(n - 1)
    print(f"退出 factorial({n}) = {result}")
    return result

factorial(3)
```

**执行过程：**
```
进入 factorial(3)
  进入 factorial(2)
    进入 factorial(1)
    退出 factorial(1) = 1
  退出 factorial(2) = 2
退出 factorial(3) = 6
```

### 1.4 经典递归问题

#### 问题一：斐波那契数列

```python
def fibonacci(n):
    """
    计算斐波那契数列第 n 项
    
    斐波那契数列：1, 1, 2, 3, 5, 8, 13, 21...
    第 n 项 = 第 (n-1) 项 + 第 (n-2) 项
    """
    if n <= 0:
        return 0
    if n == 1 or n == 2:
        return 1
    return fibonacci(n - 1) + fibonacci(n - 2)

# 测试
for i in range(1, 10):
    print(f"F({i}) = {fibonacci(i)}")
```

#### 问题二：汉诺塔

```python
def hanoi(n, source, target, auxiliary):
    """
    汉诺塔问题
    
    Args:
        n: 盘子数量
        source: 源柱子
        target: 目标柱子
        auxiliary: 辅助柱子
    """
    if n == 1:
        print(f"移动盘子 1 从 {source} 到 {target}")
        return
    
    # 将 n-1 个盘子从源柱子移动到辅助柱子
    hanoi(n - 1, source, auxiliary, target)
    
    # 移动最大的盘子到目标柱子
    print(f"移动盘子 {n} 从 {source} 到 {target}")
    
    # 将 n-1 个盘子从辅助柱子移动到目标柱子
    hanoi(n - 1, auxiliary, target, source)

# 3个盘子的解法
hanoi(3, 'A', 'C', 'B')
```

#### 问题三：二分查找（递归版）

```python
def binary_search_recursive(arr, target, left, right):
    """
    二分查找（递归版）
    
    Args:
        arr: 排序后的数组
        target: 查找目标
        left: 查找范围左边界
        right: 查找范围右边界
    
    Returns:
        目标索引，未找到返回 -1
    """
    if left > right:
        return -1  # 基本情形：未找到
    
    mid = (left + right) // 2
    
    if arr[mid] == target:
        return mid  # 找到目标
    elif arr[mid] > target:
        return binary_search_recursive(arr, target, left, mid - 1)
    else:
        return binary_search_recursive(arr, target, mid + 1, right)

# 使用示例
sorted_list = [1, 3, 5, 7, 9, 11, 13, 15]
result = binary_search_recursive(sorted_list, 7, 0, len(sorted_list) - 1)
print(f"找到位置：{result}")  # 3
```

## 二、递归 vs 循环

### 2.1 性能对比

```python
import time

# 递归版本（效率低，有大量重复计算）
def fib_recursive(n):
    if n <= 1:
        return n
    return fib_recursive(n - 1) + fib_recursive(n - 2)

# 循环版本（效率高）
def fib_iterative(n):
    if n <= 1:
        return n
    a, b = 0, 1
    for _ in range(2, n + 1):
        a, b = b, a + b
    return b

# 性能测试
n = 35

start = time.time()
fib_iterative(n)
print(f"循环用时：{time.time() - start:.6f}秒")

start = time.time()
fib_recursive(n)
print(f"递归用时：{time.time() - start:.6f}秒")
```

### 2.2 选择原则

| 场景 | 推荐方式 | 原因 |
|------|----------|------|
| 问题本身是递归定义的 | 递归 | 更直观 |
| 数据规模大 | 循环 | 避免栈溢出 |
| 需要记忆化 | 递归+缓存 | 结合两者优点 |
| 空间受限 | 循环 | 递归占用栈空间 |

### 2.3 尾递归优化

Python 不支持尾递归优化，但了解其原理很重要：

```python
# 普通递归：调用栈逐层累积
def factorial_normal(n, result=1):
    if n == 0:
        return result
    return factorial_normal(n - 1, n * result)  # 这里还有乘法操作

# 尾递归：最后一步就是函数调用
# （Python中仍然会栈溢出，但理论上可以优化）
def factorial_tail(n, result=1):
    if n == 0:
        return result
    return factorial_tail(n - 1, n * result)  # 纯粹返回函数调用
```

## 三、回调函数详解

### 3.1 回调函数的概念

回调函数是将一个函数作为参数传递给另一个函数，被传递的函数称为"回调函数"。

```python
# 回调函数定义
def greet():
    print("Hello!")

# 接收回调的函数
def execute_callback(callback):
    print("准备执行...")
    callback()  # 调用传入的函数
    print("执行完毕")

# 使用
execute_callback(greet)

# 输出：
# 准备执行...
# Hello!
# 执行完毕
```

### 3.2 带参数的回调

```python
# 简单回调
def simple_callback():
    print("回调执行")

def with_callback(callback):
    callback()

# 带参数回调
def process_data(name, callback):
    data = f"处理{name}的数据"
    callback(data)

def print_result(result):
    print(f"结果：{result}")

process_data("Alice", print_result)
# 输出：结果：处理Alice的数据
```

#### 场景三：数据处理管道

```python
def process_data(data, *transforms):
    """数据处理管道"""
    result = data
    for transform in transforms:
        result = transform(result)
    return result

def to_uppercase(text):
    return text.upper()

def add_prefix(text):
    return ">> " + text

def add_suffix(text):
    return text + " <<"

# 使用管道处理
result = process_data("hello", to_uppercase, add_prefix, add_suffix)
print(result)  # >> HELLO <<
```

## 四、闭包与回调结合

### 4.1 闭包基础回顾

```python
def make_multiplier(factor):
    def multiplier(number):
        return number * factor
    return multiplier

# 创建不同的乘法器
double = make_multiplier(2)
triple = make_multiplier(3)

print(double(5))   # 10
print(triple(5))   # 15
```

### 4.2 闭包实现回调工厂

```python
def create_callback(action_name):
    """创建带状态的回调函数"""
    def callback(data):
        print(f"[{action_name}] 处理数据：{data}")
        return f"{action_name}_processed"
    return callback

# 创建不同的回调
on_save = create_callback("保存")
on_delete = create_callback("删除")
on_update = create_callback("更新")

# 使用
on_save({"id": 1, "name": "测试"})
on_delete({"id": 2})
```

### 4.3 计数器闭包

```python
def make_counter():
    """创建计数器"""
    count = 0
    
    def increment():
        nonlocal count
        count += 1
        return count
    
    def decrement():
        nonlocal count
        count -= 1
        return count
    
    def reset():
        nonlocal count
        count = 0
        return count
    
    return {"increment": increment, "decrement": decrement, "reset": reset}

# 使用
counter = make_counter()
print(counter["increment"]())  # 1
print(counter["increment"]())  # 2
print(counter["decrement"]())  # 1
print(counter["reset"]())      # 0
```

## 五、综合实战

### 实战一：文件夹树形结构遍历

```python
import os

def traverse_directory(path, level=0):
    """
    递归遍历目录树
    
    Args:
        path: 目录路径
        level: 当前层级（用于缩进）
    """
    if level > 3:  # 限制深度，防止过深
        return
    
    try:
        items = os.listdir(path)
    except PermissionError:
        print("  " * level + "└─ [权限拒绝]")
        return
    
    for i, item in enumerate(items):
        item_path = os.path.join(path, item)
        prefix = "└─ " if i == len(items) - 1 else "├─ "
        
        if os.path.isdir(item_path):
            print("  " * level + prefix + f"📁 {item}/")
            traverse_directory(item_path, level + 1)
        else:
            size = os.path.getsize(item_path)
            print("  " * level + prefix + f"📄 {item} ({size} bytes)")

# 使用
traverse_directory(".")
```

### 实战二：事件系统

```python
class EventSystem:
    """简单的事件系统"""
    
    def __init__(self):
        self.listeners = {}
    
    def on(self, event_name, callback):
        """注册事件监听器"""
        if event_name not in self.listeners:
            self.listeners[event_name] = []
        self.listeners[event_name].append(callback)
    
    def emit(self, event_name, *args, **kwargs):
        """触发事件"""
        if event_name in self.listeners:
            for callback in self.listeners[event_name]:
                callback(*args, **kwargs)
    
    def off(self, event_name, callback):
        """移除事件监听器"""
        if event_name in self.listeners:
            self.listeners[event_name].remove(callback)

# 使用示例
events = EventSystem()

def on_login(user_id):
    print(f"用户 {user_id} 登录了")

def on_logout(user_id):
    print(f"用户 {user_id} 退出登录")

def on_purchase(user_id, amount):
    print(f"用户 {user_id} 购买了 ${amount} 的商品")

events.on("login", on_login)
events.on("logout", on_logout)
events.on("purchase", on_purchase)

# 触发事件
events.emit("login", user_id=1001)
events.emit("purchase", user_id=1001, amount=99.9)
events.emit("logout", user_id=1001)
```

## 七、易错点与避坑指南

1. **递归没有终止条件**：检查 base case 是否正确设置
2. **递归深度过大**：Python 默认递归深度约 1000，可用 `sys.setrecursionlimit()` 调整（慎用）
3. **尾递归误解**：Python 不支持尾递归优化
4. **闭包变量捕获时机**：闭包捕获的是变量引用，不是值
5. **回调函数作用域**：回调中的 `self` 需要特别注意
