### 第 1 步：函数式 vs 面向对象（10 分钟）

```python
# 函数式爬虫（简单场景）
def scrape_page(url):
    """爬取单个页面"""
    response = requests.get(url)
    return parse(response.text)

def scrape_all(urls):
    """爬取所有页面"""
    results = []
    for url in urls:
        results.append(scrape_page(url))
    return results

# 面向对象爬虫（复杂场景）
class Crawler:
    def __init__(self):
        self.session = requests.Session()
        self.results = []

    def scrape_page(self, url):
        """爬取单个页面"""
        response = self.session.get(url)
        return self.parse(response.text)

    def scrape_all(self, urls):
        """爬取所有页面"""
        for url in urls:
            self.results.append(self.scrape_page(url))
        return self.results

# 面向对象优势：
# 1. 状态共享（session、results）
# 2. 可扩展（继承、多态）
# 3. 易测试（mock 对象）
# 4. 好维护（方法封装）
```

### 第 2 步：设计爬虫基类（15 分钟）

```python
import requests
from abc import ABC, abstractmethod
import time
from pathlib import Path
import logging

# 设置日志
logging.basicConfig(
    level=logging.INFO,
    format="%(asctime)s - %(levelname)s - %(message)s"
)

class BaseCrawler(ABC):
    """爬虫基类"""

    def __init__(self, name="crawler"):
        self.name = name
        self.session = requests.Session()
        self.results = []
        self.failed_urls = []

        # 默认配置
        self.timeout = 10
        self.delay = 2

    @abstractmethod
    def parse(self, html):
        """解析页面，必须被子类实现"""
        pass

    @abstractmethod
    def get_page_url(self, page):
        """生成页面 URL，必须被子类实现"""
        pass

    def fetch(self, url):
        """获取页面"""
        try:
            response = self.session.get(
                url,
                headers=self.get_headers(),
                timeout=self.timeout
            )
            response.raise_for_status()
            response.encoding = "utf-8"
            return response.text
        except Exception as e:
            logging.error(f"获取 {url} 失败: {e}")
            self.failed_urls.append(url)
            return None

    def get_headers(self):
        """获取请求头（可重写）"""
        return {
            "User-Agent": "Mozilla/5.0 (Windows NT 10.0; Win64; x64) ...",
        }

    def crawl(self, start_page=1, end_page=10):
        """爬取多页"""
        for page in range(start_page, end_page + 1):
            url = self.get_page_url(page)
            logging.info(f"爬取第 {page} 页: {url}")

            html = self.fetch(url)
            if html:
                data = self.parse(html)
                self.results.append(data)

            time.sleep(self.delay)  # 避免请求过快

        return self.results
```

### 第 3 步：实现贴吧爬虫类（15 分钟）

```python
from bs4 import BeautifulSoup

class TiebaCrawler(BaseCrawler):
    """贴吧爬虫"""

    def __init__(self, tieba_name, **kwargs):
        super().__init__(name=f"tieba_{tieba_name}")
        self.tieba_name = tieba_name
        self.base_url = f"https://tieba.baidu.com/f?kw={tieba_name}"

        # 从 kwargs 设置配置
        self.timeout = kwargs.get("timeout", 10)
        self.delay = kwargs.get("delay", 2)

    def get_page_url(self, page):
        """生成贴吧分页 URL"""
        pn = (page - 1) * 50
        return f"{self.base_url}&pn={pn}"

    def parse(self, html):
        """解析帖子列表"""
        soup = BeautifulSoup(html, "lxml")
        threads = []

        for item in soup.select(".threadlist_title"):
            title = item.get_text().strip()
            link = item.get("href", "")

            if title and link:
                threads.append({
                    "title": title,
                    "link": f"https://tieba.baidu.com{link}" if not link.startswith("http") else link
                })

        return threads

    def save(self, filepath):
        """保存结果"""
        import json
        all_threads = []
        for page_data in self.results:
            all_threads.extend(page_data)

        with open(filepath, "w", encoding="utf-8") as f:
            json.dump(all_threads, f, ensure_ascii=False, indent=2)

        logging.info(f"保存 {len(all_threads)} 条到 {filepath}")

# 使用
crawler = TiebaCrawler("python", delay=3)
crawler.crawl(start_page=1, end_page=5)
crawler.save("tieba_python.json")
```

### 第 4 步：增量爬取（15 分钟）

```python
import json
import hashlib
from pathlib import Path

class IncrementalCrawler(BaseCrawler):
    """增量爬虫基类"""

    def __init__(self, name="incremental_crawler", state_file=None):
        super().__init__(name=name)
        self.state_file = Path(state_file or f"{name}_state.json")
        self.crawled = self.load_state()
        self.seen_hashes = set(self.crawled.values())

    def content_hash(self, content):
        """计算内容哈希"""
        return hashlib.md5(str(content).encode()).hexdigest()

    def is_new(self, item):
        """检查是否是新内容"""
        h = self.content_hash(item)
        return h not in self.seen_hashes

    def mark_crawled(self, item, identifier):
        """标记为已爬取"""
        h = self.content_hash(item)
        self.seen_hashes.add(h)
        self.crawled[identifier] = h

    def save_state(self):
        """保存状态"""
        with open(self.state_file, "w") as f:
            json.dump(self.crawled, f)

    def load_state(self):
        """加载状态"""
        if self.state_file.exists():
            with open(self.state_file) as f:
                return json.load(f)
        return {}

    def crawl_incremental(self, items):
        """增量爬取"""
        new_items = []
        for item in items:
            identifier = self.get_identifier(item)
            if self.is_new(item):
                new_items.append(item)
                self.mark_crawled(item, identifier)

        self.save_state()
        return new_items


class TiebaIncrementalCrawler(TiebaCrawler, IncrementalCrawler):
    """贴吧增量爬虫"""

    def __init__(self, tieba_name, **kwargs):
        TiebaCrawler.__init__(self, tieba_name, **kwargs)
        IncrementalCrawler.__init__(self, name=f"tieba_{tieba_name}")

    def crawl(self, start_page=1, end_page=10):
        """增量爬取"""
        all_new = []
        for page in range(start_page, end_page + 1):
            url = self.get_page_url(page)
            logging.info(f"爬取第 {page} 页: {url}")

            html = self.fetch(url)
            if html:
                threads = self.parse(html)
                new_threads = self.crawl_incremental(threads)
                all_new.extend(new_threads)
                self.results.append(new_threads)

                logging.info(f"  新帖子: {len(new_threads)}/{len(threads)}")

            time.sleep(self.delay)

        return all_new
```

### 第 5 步：添加日志和错误处理（10 分钟）

```python
import logging
from functools import wraps

# 配置日志
logging.basicConfig(
    level=logging.DEBUG,
    format="%(asctime)s - %(name)s - %(levelname)s - %(message)s",
    handlers=[
        logging.FileHandler("crawler.log", encoding="utf-8"),
        logging.StreamHandler()
    ]
)

def retry_on_failure(max_retries=3, delay=5):
    """失败重试装饰器"""
    def decorator(func):
        @wraps(func)
        def wrapper(*args, **kwargs):
            for i in range(max_retries):
                try:
                    return func(*args, **kwargs)
                except Exception as e:
                    if i == max_retries - 1:
                        raise
                    logging.warning(f"{func.__name__} 失败，{i+1}/{max_retries}: {e}")
                    time.sleep(delay)
        return wrapper
    return decorator

class RobustCrawler(BaseCrawler):
    """带重试和错误处理的爬虫"""

    @retry_on_failure(max_retries=3, delay=5)
    def fetch(self, url):
        """获取页面（带重试）"""
        response = self.session.get(
            url,
            headers=self.get_headers(),
            timeout=self.timeout
        )
        response.raise_for_status()
        response.encoding = "utf-8"
        return response.text

    def crawl_with_report(self, start_page=1, end_page=10):
        """爬取并生成报告"""
        self.crawl(start_page, end_page)

        report = f"""
        爬虫报告
        =========
        名称: {self.name}
        总页数: {end_page - start_page + 1}
        失败 URL: {len(self.failed_urls)}
        失败列表: {self.failed_urls}
        """
        logging.info(report)
```
