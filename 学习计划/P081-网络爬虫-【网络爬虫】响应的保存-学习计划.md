### 第 2 步：保存二进制内容（15 分钟）

```python
import requests

# 下载图片
response = requests.get("https://example.com/image.png")

# 方法 1：一次性写入（小文件）
with open("image.png", "wb") as f:
    f.write(response.content)

# 方法 2：流式写入
with open("image.png", "wb") as f:
    for chunk in response.iter_content(chunk_size=8192):
        f.write(chunk)

# 方法 3：用 stream=True（适合大文件）
response = requests.get("https://example.com/large-image.png", stream=True)
with open("large-image.png", "wb") as f:
    for chunk in response.iter_content(chunk_size=1024):
        f.write(chunk)
```

### 第 3 步：流式下载详解（15 分钟）

```python
import requests

# 为什么要用 stream=True？
# - 不加 stream=True：requests 会在内存中缓存整个响应
# - 适合小文件：代码简单
# - 缺点：大文件会占用大量内存

# 适合 stream=True 的场景：
# - 下载视频（可能几个 GB）
# - 下载大文件（压缩包、镜像）
# - 实时流处理（日志流、音频流）

# 小文件（推荐，不用 stream）
response = requests.get("https://example.com/small-file.png")
with open("file.png", "wb") as f:
    f.write(response.content)

# 大文件（必须用 stream）
response = requests.get("https://example.com/big-file.zip", stream=True)
with open("big-file.zip", "wb") as f:
    for chunk in response.iter_content(chunk_size=8192):
        f.write(chunk)
```

### 第 4 步：带进度条的下载（15 分钟）

```python
import requests
from pathlib import Path

def download_with_progress(url, filename):
    """带进度条的文件下载"""
    response = requests.get(url, stream=True)
    response.raise_for_status()  # 检查 HTTP 错误

    total_size = int(response.headers.get("Content-Length", 0))
    downloaded = 0

    with open(filename, "wb") as f:
        for chunk in response.iter_content(chunk_size=1024 * 1024):  # 1MB
            f.write(chunk)
            downloaded += len(chunk)

            if total_size:
                percent = downloaded / total_size * 100
                print(f"\r下载进度: {percent:.1f}%", end="", flush=True)

    print()  # 换行
    print(f"保存到: {filename}")

# 使用
download_with_progress(
    "https://example.com/big-file.zip",
    "big-file.zip"
)
```

### 第 5 步：完整下载函数封装（10 分钟）

```python
import requests
from pathlib import Path

def download_file(url, filepath=None, chunk_size=8192, headers=None):
    """
    通用文件下载函数

    参数：
        url: 下载链接
        filepath: 保存路径（None 则从 URL 自动生成）
        chunk_size: 每次写入的块大小
        headers: 额外的请求头
    返回：
        文件保存路径
    """
    # 自动生成文件名
    if filepath is None:
        filepath = Path(url).name

    # 发送请求
    kwargs = {"stream": True}
    if headers:
        kwargs["headers"] = headers

    response = requests.get(url, **kwargs)
    response.raise_for_status()

    # 获取文件大小
    total_size = int(response.headers.get("Content-Length", 0))

    # 下载
    downloaded = 0
    with open(filepath, "wb") as f:
        for chunk in response.iter_content(chunk_size=chunk_size):
            if chunk:
                f.write(chunk)
                downloaded += len(chunk)

    return filepath

# 使用示例
download_file(
    "https://example.com/image.jpg",
    "downloaded_image.jpg",
    headers={"User-Agent": "Mozilla/5.0 ..."}
)
```
