## 二、为什么先学这一节

HTTP 请求头是爬虫与服务器"交流"的关键信息。很多网站通过检查请求头来判断请求是否来自浏览器。理解请求头才能：
- 模拟真实浏览器
- 绕过基础反爬
- 正确处理登录认证

## 五、具体学习步骤

### 第 1 步：HTTP 请求头全解（20 分钟）

```python
import requests

# 完整的请求头示例
headers = {
    # === 身份标识 ===
    "User-Agent": "Mozilla/5.0 (Windows NT 10.0; Win64; x64) AppleWebKit/537.36 (KHTML, like Gecko) Chrome/120.0.0.0 Safari/537.36",
    # 浏览器标识，不同网站会检查这个

    "Referer": "https://www.example.com/previous-page",
    # 来源页面，用于防盗链和统计
    # 注意：Referer 拼写错误，是 HTTP 规范的历史遗留

    "Origin": "https://www.example.com",
    # 跨域请求的来源，用于 CORS 验证

    # === 接受格式 ===
    "Accept": "text/html,application/xhtml+xml,application/xml;q=0.9,*/*;q=0.8",
    # 告诉服务器客户端能接受的媒体类型

    "Accept-Language": "zh-CN,zh;q=0.9,en;q=0.8",
    # 能接受的语言，q 表示优先级

    "Accept-Encoding": "gzip, deflate, br",
    # 能接受的压缩格式（br = Brotli）

    # === 缓存控制 ===
    "Cache-Control": "no-cache",
    # 控制缓存行为

    "If-None-Match": '"abc123"',
    # 条件请求：ETag 验证

    "If-Modified-Since": "Wed, 21 Oct 2023 07:28:00 GMT",
    # 条件请求：时间验证

    # === 认证信息 ===
    "Authorization": "Bearer xxx",
    # API 认证，常用的是 Bearer Token

    "Cookie": "session=abc123; token=xyz789",
    # 会话标识，用于登录状态

    # === 内容相关 ===
    "Content-Type": "application/json",
    # 请求体的内容类型
    # 常见值：
    # - application/x-www-form-urlencoded（表单）
    # - multipart/form-data（文件上传）
    # - application/json（JSON）
}

response = requests.get("https://api.example.com", headers=headers)
```

### 第 2 步：常见反爬请求头（15 分钟）

```python
# 网站常用的反爬检查：

反爬检查表 = """
┌──────────────────┬──────────────────────────────────┐
│ Header           │ 作用                             │
├──────────────────┼──────────────────────────────────┤
│ User-Agent       │ 检查是否来自浏览器               │
│ Referer          │ 检查来源是否合法                 │
│ Cookie           │ 检查会话是否有效                 │
│ Authorization    │ 检查 API Token                   │
│ Accept-Language  │ 检查语言设置（区分人类/机器）     │
│ X-Requested-With │ AJAX 请求标记（通常设为 XMLHttpRequest）│
│ X-Forwarded-For  │ 代理服务器转发的真实 IP          │
└──────────────────┴──────────────────────────────────┘
"""

# 基础反爬应对：
headers = {
    "User-Agent": "Mozilla/5.0 (Windows NT 10.0; Win64; x64) ...",
    "Referer": "https://www.target-site.com/",
    "Accept-Language": "zh-CN,zh;q=0.9",
}

# 进阶反爬应对：
headers = {
    "User-Agent": "Mozilla/5.0 ...",
    "Referer": "https://www.target-site.com/",
    "Accept-Language": "zh-CN,zh;q=0.9",
    "Accept": "text/html,application/xhtml+xml...",
    "Accept-Encoding": "gzip, deflate",
    "Connection": "keep-alive",
    "Upgrade-Insecure-Requests": "1",
}
```

### 第 4 步：requests 设置请求头（15 分钟）

```python
import requests

# 方法 1：直接设置 headers 参数
response = requests.get(
    "https://api.example.com",
    headers={
        "User-Agent": "Mozilla/5.0 (Windows NT 10.0; Win64; x64) Chrome/120.0.0.0",
        "Referer": "https://www.example.com/",
    }
)

# 方法 2：用 Session 自动添加 headers
session = requests.Session()
session.headers.update({
    "User-Agent": "Mozilla/5.0 (Windows NT 10.0; Win64; x64) Chrome/120.0.0.0",
    "Referer": "https://www.example.com/",
})

# 所有通过 session 的请求都会自动带上这些 headers
response1 = session.get("https://api.example.com/endpoint1")
response2 = session.get("https://api.example.com/endpoint2")

# 方法 3：默认 headers（全局）
requests.DEFAULT_HEADERS.update({
    "User-Agent": "Mozilla/5.0 MyBot/1.0",
})

# 合并：传入的 headers 会覆盖默认 headers
response = requests.get(
    "https://api.example.com",
    headers={"User-Agent": "Custom/1.0"}  # 这个会覆盖默认的
)
```

### 第 5 步：常见错误和调试（10 分钟）

```python
import requests

# 错误 1：忘记设置 User-Agent
# ❌ 很多网站会拒绝没有 User-Agent 的请求
response = requests.get("https://www.example.com")  # 可能被拒

# ✅ 应该这样
response = requests.get(
    "https://www.example.com",
    headers={"User-Agent": "Mozilla/5.0 ..."}
)

# 错误 2：Referer 不合理
# ❌ 从不存在的页面跳过来
headers = {
    "User-Agent": "Mozilla/5.0 ...",
    "Referer": "https://www.example.com/",
}
# Referer 应该是当前页面的上一个页面

# 错误 3：Cookie 过期
# 登录后 Cookie 会过期，需要重新登录或刷新 Cookie

# 调试技巧：打印实际发送的请求头
session = requests.Session()
session.headers.update({"User-Agent": "TestBot/1.0"})

import logging
logging.basicConfig(level=logging.DEBUG)
response = session.get("https://httpbin.org/headers")
```

```python
class MySession(requests.Session):
    def __init__(self):
        super().__init__()
        self.headers.update({
            "User-Agent": "Mozilla/5.0 (Windows NT 10.0; Win64; x64) ...",
            "Accept-Language": "zh-CN,zh;q=0.9",
        })

# 使用
session = MySession()
response = session.get("https://example.com")
```
